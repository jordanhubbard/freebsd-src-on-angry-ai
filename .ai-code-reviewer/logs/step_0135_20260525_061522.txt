const ABUSEHOST: &str = "whois.abuse.net";
const ANICHOST: &str = "whois.arin.net";
const DENICHOST: &str = "whois.denic.de";
const DKNICHOST: &str = "whois.dk-hostmaster.dk";
const FNICHOST: &str = "whois.afrinic.net";
const GNICHOST: &str = "whois.nic.gov";

const INICHOST: &str = "whois.internic.net";

const LNICHOST: &str = "whois.lacnic.net";

const PNICHOST: &str = "whois.apnic.net";
const QNICHOST_TAIL: &str = ".whois-servers.net";
const RNICHOST: &str = "whois.ripe.net";
const VNICHOST: &str = "whois.verisign-grs.com";

const DEFAULT_PORT: &str = "whois";

const WHOIS_RECURSE: u32 = 0x01;
const WHOIS_QUICK: u32 = 0x02;
const WHOIS_SPAM_ME: u32 = 0x04;

const CHOPSPAM: &str = ">>> Last update of WHOIS database:"

struct WhoisWhere {
    suffix: &'static str,
    server: &'static str,
}

static WHOISWHERE: [WhoisWhere; 8] = [
    WhoisWhere { suffix: "-ARIN", server: ANICHOST },
    WhoisWhere { suffix: "-NICAT", server: "at" },
    WhoisWhere { suffix: "-NORID", server: "no" },
    WhoisWhere { suffix: "-RIPE", server: RNICHOST },
    WhoisWhere { suffix: ".ac.uk", server: "ac.uk" },
    WhoisWhere { suffix: ".gov.uk", server: "ac.uk" },
    WhoisWhere { suffix: "", server: IANAHOST },
    WhoisWhere { suffix: "\0", server: "\0" },
];

struct WhoisReferral {
    prefix: &'static str,
}

static WHOIS_REFERRAL: [WhoisReferral; 6] = [
    WhoisReferral { prefix: "whois:" },
    WhoisReferral { prefix: "Whois Server:" },
    WhoisReferral { prefix: "Registrar WHOIS Server:" },
    WhoisReferral { prefix: "ReferralServer:  whois://" },
    WhoisReferral { prefix: "ReferralServer:  rwhois://" },
    WhoisReferral { prefix: "descr:          region. Please query" },
];

static ACTUALLY_ARIN: [&str; 2] = [
    "netname:        ERX-NETBLOCK\n",
    "netname:        NON-RIPE-NCC-MANAGED-ADDRESS-BLOCK\n",
];

struct TryRir {
    looped: bool,
    host: &'static str,
}

static mut TRY_RIR: [TryRir; 6] = [
    TryRir { looped: false, host: ANICHOST },
    TryRir { looped: false, host: RNICHOST },
    TryRir { looped: false, host: PNICHOST },
    TryRir { looped: false, host: FNICHOST },
    TryRir { looped: false, host: LNICHOST },
    TryRir { looped: false, host: "\0" },
];

fn reset_rir() {
    unsafe {
        for entry in TRY_RIR.iter_mut() {
            entry.looped = false;
        }
    }
}

fn is_host_char(c: char) -> bool {
    c.is_ascii_alphanumeric() || c == '.' || c == '-'
}

fn usage() -> ! {
    eprintln!(
        "usage: whois [-aAbfgiIklmPQrRS] [-c country-code | -h hostname] \
         [-p port] name ..."
    );
    process::exit(64);
}

fn choose_server(domain: &str) -> String {
    let domain_lower = domain.to_lowercase();
    for entry in &WHOISWHERE {
        if entry.suffix == "\0" {
            break;
        }
        let suffix_lower = entry.suffix.to_lowercase();
        if suffix_lower.is_empty() {
            return format!("{}{}", entry.server, QNICHOST_TAIL);
        }
        if domain_lower.len() > suffix_lower.len()
            && domain_lower.ends_with(&suffix_lower)
        {
            if entry.server == "at" || entry.server == "no" || entry.server == "ac.uk" {
                return format!("{}{}", entry.server, QNICHOST_TAIL);
            }
            return entry.server.to_string();
        }
    }
    eprintln!("whois: no default whois server");
    process::exit(70);
}

fn connect_to_host(hostname: &str, port: &str) -> io::Result<TcpStream> {
    let addr_str = format!("{}:{}", hostname, port);
    let mut last_err = None;
    for addr in addr_str.to_socket_addrs()? {
        match TcpStream::connect_timeout(&addr, std::time::Duration::from_secs(3)) {
            Ok(stream) => return Ok(stream),
            Err(e) => last_err = Some(e),
        }
    }
    Err(last_err.unwrap_or_else(|| {
        io::Error::new(io::ErrorKind::ConnectionRefused, "no address resolved")
    }))
}

fn whois(query: &str, hostname: &str, hostport: &str, flags: u32) {
    let stream = match connect_to_host(hostname, hostport) {
        Ok(s) => s,
        Err(e) => {
            eprintln!("whois: connect(): {}", e);
            process::exit(71);
        }
    };

    let mut stream = stream;
    let canonname = hostname;

    let mut request = String::new();
    if (flags & WHOIS_SPAM_ME) == 0
        && (hostname.eq_ignore_ascii_case(DENICHOST)
            || hostname.eq_ignore_ascii_case(&format!("de{}", QNICHOST_TAIL)))
    {
        let idn = query.bytes().any(|b| b > 127);
        if idn {
            request = format!("-T dn{} \r\n", query);
        } else {
            request = format!("-T dn,ace {} \r\n", query);
        }
    } else if (flags & WHOIS_SPAM_ME) == 0
        && (hostname.eq_ignore_ascii_case(DKNICHOST)
            || hostname.eq_ignore_ascii_case(&format!("dk{}", QNICHOST_TAIL)))
    {
        request = format!("--show-handles {} \r\n", query);
    } else if (flags & WHOIS_SPAM_ME) != 0 || query.contains(' ') {
        request = format!("{} \r\n", query);
    } else if hostname.eq_ignore_ascii_case(ANICHOST) {
        let ql = query.to_lowercase();
        if ql.starts_with("as") && query[2..].chars().all(|c| c.is_ascii_digit()) {
            request = format!("+ a {} \r\n", &query[2..]);
        } else {
            request = format!("+ {} \r\n", query);
        }
    } else if canonname.eq_ignore_ascii_case(VNICHOST) {
        request = format!("domain {} \r\n", query);
    } else {
        request = format!("{} \r\n", query);
    }

    if let Err(e) = stream.write_all(request.as_bytes()) {
        eprintln!("whois: write(): {}", e);
        process::exit(71);
    }

    let mut comment: u32 = 0;
    if (flags & WHOIS_SPAM_ME) == 0
        && (hostname.eq_ignore_ascii_case(ANICHOST)
            || hostname.eq_ignore_ascii_case(RNICHOST))
    {
        comment = 2;
    }

    let mut nhost: Option<String> = None;
    let mut nport: Option<String> = None;

    let reader = BufReader::new(stream.try_clone().unwrap());
    for line_result in reader.lines() {
        let line = match line_result {
            Ok(l) => l,
            Err(_) => break,
        };

        // Nominet
        if (flags & WHOIS_SPAM_ME) == 0 && line == "-- " {
            break;
        }

        // RIRs
        if comment == 1 && line.starts_with('#') {
            break;
        } else if comment == 2 {
            if line.chars().next().map_or(false, |c| "#%\r\n".contains(c)) {
                continue;
            } else {
                comment = 1;
            }
        }

        print!("{}", line);

        if (flags & WHOIS_RECURSE) != 0 && nhost.is_none() {
            for referral in &WHOIS_REFERRAL {
                let mut p = line.as_str();
                // Skip leading spaces
                p = p.trim_start();
                if let Some(pos) = p.find(referral.prefix) {
                    let after_prefix = &p[pos + referral.prefix.len()..];
                    let after_prefix = after_prefix.trim_start();
                    let host_end = after_prefix
                        .chars()
                        .position(|c| !is_host_char(c))
                        .unwrap_or(after_prefix.len());
                    let host = &after_prefix[..host_end];
                    if !host.is_empty() {
                        nhost = Some(host.to_string());
                        let rest = &after_prefix[host_end..];
                        if rest.starts_with(':') {
                            let port_str = rest[1..].trim_start();
                            let port_end = port_str
                                .chars()
                                .position(|c| !c.is_ascii_digit())
                                .unwrap_or(port_str.len());
                            let port_digits = &port_str[..port_end];
                            if !port_digits.is_empty() && port_digits.len() < 6 {
                                nport = Some(port_digits.to_string());
                            } else {
                                nport = Some(hostport.to_string());
                                nhost = None;
                            }
                        } else {
                            nport = Some(hostport.to_string());
                        }
                        break;
                    }
                }
            }

            if nhost.is_none() {
                for pattern in &ACTUALLY_ARIN {
                    if line.starts_with(pattern.trim_end_matches('\n')) {
                        nhost = Some(ANICHOST.to_string());
                        nport = Some(hostport.to_string());
                        break;
                    }
                }
            }
        }

        // Verisign etc.
        if (flags & WHOIS_SPAM_ME) == 0 {
            let line_lower = line.to_lowercase();
            let chop_lower = CHOPSPAM.to_lowercase();
            if line_lower.starts_with(&chop_lower)
                || line_lower.starts_with(&chop_lower[4..])
            {
                println!();
                break;
            }
        }
    }

    // Track RIR visits
    let mut f = false;
    unsafe {
        for entry in TRY_RIR.iter_mut() {
            if entry.host == "\0" {
                break;
            }
            if !entry.looped && entry.host.eq_ignore_ascii_case(hostname) {
                entry.looped = true;
            }
            if entry.looped
                && nhost.as_ref().map_or(false, |nh| nh.eq_ignore_ascii_case(entry.host))
            {
                nhost = None;
                nport = None;
                f = true;
            }
        }
    }

    if f {
        unsafe {
            for entry in TRY_RIR.iter() {
                if entry.host == "\0" {
                    break;
                }
                if !entry.looped {
                    nhost = Some(entry.host.to_string());
                    nport = Some(hostport.to_string());
                    break;
                }
            }
        }
    }

    if let Some(ref nh) = nhost {
        if !nh.eq_ignore_ascii_case(hostname) {
            println!("# {}\n", nh);
            let nh = nh.clone();
            let np = nport.clone().unwrap_or_else(|| hostport.to_string());
            whois(query, &nh, &np, flags);
        }
    }
}

    let args: Vec<String> = env::args().collect();
    let mut host: Option<String> = None;
    let mut country: Option<String> = None;
    let mut port = DEFAULT_PORT.to_string();
    let mut flags: u32 = 0;

    let mut i = 1;
    let mut names: Vec<String> = Vec::new();

    while i < args.len() {
        if args[i].starts_with('-') && args[i].len() > 1 {
            // Parse flags
            for ch in args[i][1..].chars() {
                match ch {
                    'a' => host = Some(ANICHOST.to_string()),
                    'A' => host = Some(PNICHOST.to_string()),
                    'b' => host = Some(ABUSEHOST.to_string()),
                    'f' => host = Some(FNICHOST.to_string()),

                    'i' => host = Some(INICHOST.to_string()),
                    'I' => host = Some(IANAHOST.to_string()),
                    'k' => host = Some(KNICHOST.to_string()),
                    'l' => host = Some(LNICHOST.to_string()),
                    'm' => host = Some(MNICHOST.to_string()),
                    'P' => host = Some(PDBHOST.to_string()),
                    'Q' => flags |= WHOIS_QUICK,
                    'r' => host = Some(RNICHOST.to_string()),
                    'R' => flags |= WHOIS_RECURSE,
                    'S' => flags |= WHOIS_SPAM_ME,
                    'c' => {
                        i += 1;
                        if i >= args.len() {
                            usage();
                        }
                        country = Some(args[i].clone());

                    }

            }
        } else {
