fn mkdtemp(template: &str) -> Result<String, String> {
    // Find the XXXXXXXX suffix
    let bytes = template.as_bytes();
    let suffix_start = bytes
        .windows(10)
        .position(|w| w == b"XXXXXXXXXX")
        .ok_or_else(|| "template must contain XXXXXXXX".to_string())?;

    let prefix = &template[..suffix_start];
    let suffix = &template[suffix_start + 10..];

    let mut attempts = 0;
    loop {
        if attempts >= 100 {
            return Err("too many retries".to_string());
        }
        attempts += 1;

        let rand_part = generate_random_string(10);
        let name = format!("{}{}{}", prefix, rand_part, suffix);

        match std::fs::create_dir(&name) {
            Ok(()) => return Ok(name),
            Err(e) if e.kind() == std::io::ErrorKind::AlreadyExists => continue,
            Err(e) => return Err(e.to_string()),
        }
    }
}

fn mkstemp(template: &str) -> Result<String, String> {
    let bytes = template.as_bytes();
    let suffix_start = bytes
        .windows(10)
        .position(|w| w == b"XXXXXXXXXX")
        .ok_or_else(|| "template must contain XXXXXXXX".to_string())?;

    let prefix = &template[..suffix_start];
    let suffix = &template[suffix_start + 10..];

    let mut attempts = 0;
    loop {
        if attempts >= 100 {
            return Err("too many retries".to_string());
        }
        attempts += 1;

        let rand_part = generate_random_string(10);
        let name = format!("{}{}{}", prefix, rand_part, suffix);

        match std::fs::OpenOptions::new()
            .create_new(true)
            .write(true)
            .truncate(true)
            .mode(0o600)
            .open(&name)
        {
            Ok(_) => return Ok(name),
            Err(e) if e.kind() == std::io::ErrorKind::AlreadyExists => continue,
            Err(e) => return Err(e.to_string()),
        }
    }
}

fn generate_random_string(len: usize) -> String {
    const CHARS: &[u8] = b"ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789";
    let mut result = String::with_capacity(len);
    let mut buf = [0u8; 32];

    for chunk in (0..len).steps_by(32) {
        let remaining = std::cmp::min(32, len - chunk);
        match getrandom(&mut buf[..remaining]) {
            Ok(()) => {
                for &b in &buf[..remaining] {
                    result.push(CHARS[(b % CHARS.len() as u8) as usize] as char);
                }
            }
            Err(_) => {
                // Fallback to simple counter-based randomness
                for i in 0..remaining {
                    let idx = ((chunk + i) * 31 + 17) % CHARS.len();
                    result.push(CHARS[idx as usize] as char);
                }
            }
        }
    }
    result
}

fn getrandom(buf: &mut [u8]) -> Result<(), std::io::Error> {
    // Use /dev/urandom for randomness
    let mut f = std::fs::File::open("/dev/urandom")?;
    std::io::Read::read_exact(&mut f, buf)?;
    Ok(())
}
