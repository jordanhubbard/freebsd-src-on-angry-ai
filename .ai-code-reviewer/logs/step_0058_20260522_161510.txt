            let (v, na, e) = parse_int(args, ap); ap = na; if e { err = true; }
            do_int(v, w, pr)
        }
        'o' => {
            let (v, na, e) = parse_uint(args, ap); ap = na; if e { err = true; }
            do_oct(v, w, pr)
        }
        'u' => {
            let (v, na, e) = parse_uint(args, ap); ap = na; if e { err = true; }
            do_int(v as i64, w, pr)
        }
        'x' => {
            let (v, na, e) = parse_uint(args, ap); ap = na; if e { err = true; }
            do_hex(v, w, pr, false)
        }
        'X' => {
            let (v, na, e) = parse_uint(args, ap); ap = na; if e { err = true; }
            do_hex(v, w, pr, true)
        }
        'e'|'E'|'f'|'F'|'g'|'G'|'a'|'A' => {
            let (v, na, e) = parse_float(args, ap); ap = na; if e { err = true; }
            do_flt(v, w, pr, conv)
        }
        _ => {
            eprintln!("printf: illegal format character {}", conv);
            return (String::new(), ap, true, false);
        }
    };
    (out, ap, err, term)
}

fn main() {
    let args: Vec<String> = env::args().skip(1).collect();
    if args.is_empty() { usage(); process::exit(1); }

    let fmt_raw = &args[0];
    let av: Vec<&str> = args[1..].iter().map(|s| s.as_str()).collect();
    let (pfmt, _) = escape(fmt_raw, true);
    let fmt = pfmt.as_str();
    let mut ai = 0;
    let mut rv = 0;
    let out = io::stdout();
    let mut w = out.lock();

    loop {
        let mut fi = 0;
        let mut done = false;
        while fi < fmt.len() {
            if fmt.as_bytes()[fi] == b'%' {
                if fi + 1 < fmt.len() && fmt.as_bytes()[fi+1] == b'%' {
                    if let Err(e) = w.write_all(b"%") { eprintln!("printf: {}", e); process::exit(1); }
                    fi += 2; continue;
                }
                let spec = &fmt[fi..];
                let (res, na, e, t) = process_spec(spec, &av, ai);
                ai = na;
                if e { rv = 1; }
                if let Err(er) = w.write_all(res.as_bytes()) { eprintln!("printf: {}", er); process::exit(1); }
                if t { process::exit(rv); }
                // Find end of spec to advance fi
                let mut sp = 1;
                while sp < spec.len() && "#'-+ 0".contains(spec.as_bytes()[sp] as char) { sp += 1; }
                if sp < spec.len() && spec.as_bytes()[sp] == b'*' { sp += 1; }
                while sp < spec.len() && spec.as_bytes()[sp].is_ascii_digit() { sp += 1; }
                if sp < spec.len() && spec.as_bytes()[sp] == b'.' {
                    sp += 1;
                    if sp < spec.len() && spec.as_bytes()[sp] == b'*' { sp += 1; }
                    while sp < spec.len() && spec.as_bytes()[sp].is_ascii_digit() { sp += 1; }
                }
                if sp < spec.len() && spec.as_bytes()[sp] == b'L' { sp += 1; }
                if sp < spec.len() { sp += 1; }
                fi += sp;
                done = true;
                break;
            }
            fi += 1;
        }
        if fi < fmt.len() {
            if let Err(e) = w.write_all(fmt[fi..].as_bytes()) { eprintln!("printf: {}", e); process::exit(1); }
        }
        if ai >= av.len() { process::exit(rv); }
        if done { break; }
    }
}
