use std::env;
use std::io::{self, Read};
use std::process;

fn usage(progname: &str) {
    eprintln!("usage: {} [-sx] [-n count] [prefix [suffix]]", progname);
    process::exit(1);
}

fn main() {
    let args: Vec<String> = env::args().collect();
    let progname = args.first().map(|s| s.as_str()).unwrap_or("file2c");

    let mut maxcount: usize = 0;
    let mut pretty = false;
    let mut radix = 10;

    let mut i = 1;
    while i < args.len() {
        let arg = &args[i];
        if arg == "-n" {
            i += 1;
            if i >= args.len() {
                usage(progname);
            }
            maxcount = match args[i].parse::<usize>() {
                Ok(v) => v,
                Err(_) => usage(progname),
            };
            i += 1;
        } else if arg == "-s" {
            pretty = true;
            i += 1;
        } else if arg == "-x" {
            radix = 16;
            i += 1;
        } else if arg.starts_with('-') {
            usage(progname);
        } else {
            break;
        }
    }

    let remaining: Vec<&str> = args[i..].iter().map(|s| s.as_str()).collect();
    if remaining.len() > 0 {
        println!("{}", remaining[0]);
    }

    let mut buf = Vec::new();
    if let Err(e) = io::stdin().read_to_end(&mut buf) {
        eprintln!("file2c: read error: {}", e);
        process::exit(1);
    }

    let mut count: usize = 0;
    let mut linepos: usize = 0;

    for &byte in &buf {
        if count > 0 {
            print!(",");
            linepos += 1;

        if (maxcount == 0 && linepos > 70) || (maxcount > 0 && count >= maxcount) {
            println!();

            linepos = 0;

                print!(" ");
                linepos += 1;
            } else {
                print!("\t");
                linepos += 8;
            }

            10 => {
                let printed = format!("{}", byte as i32);
                print!("{}", printed);
                linepos += printed.len();
            }

                print!("{}", printed);
                linepos += printed.len();
            }
            _ => unreachable!(),

    }
