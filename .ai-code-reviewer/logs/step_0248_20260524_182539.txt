use std::env;
use std::fs::File;
use std::io::{self, BufRead, BufReader, Read, Write};
use std::process;

const RANDOM_TYPE_UNSET: i32 = 0;
const RANDOM_TYPE_LINES: i32 = 1;
const RANDOM_TYPE_WORDS: i32 = 2;

fn usage() -> ! {
    eprintln!("usage: random [-elrUuw] [-f filename] [denominator]");
    process::exit(1);
}

/// Generates a random number uniformly in the range [0.0, 1.0).
fn random_unit_float() -> f64 {
    let denom: u64 = 1u64 << 53;
    let mask: u64 = denom - 1;
    let mut rand64: u64 = 0;
    // Use /dev/urandom for randomness
    let mut f = File::open("/dev/urandom").unwrap_or_else(|_| {
        eprintln!("cannot open /dev/urandom");
        process::exit(1);
    });
    if f.read_exact(bytemuck_cast(&mut rand64)).is_err() {
        eprintln!("cannot read from /dev/urandom");
        process::exit(1);
    }
    let masked = rand64 & mask;
    masked as f64 / denom as f64
}

fn bytemuck_cast(buf: &mut u64) -> &mut [u8] {
    unsafe { std::slice::from_raw_parts_mut(buf as *mut u64 as *mut u8, 8) }
}

/// Returns true with probability 1 / denom (a floating point number >= 1).
fn random_uniform_denom(denom: f64) -> bool {
    (denom * random_unit_float()) as u64 == 0
}

fn randomize_fd(
    file: File,
    _type: i32,
    unique: bool,
    denom: f64,
) -> i32 {
    let type_ = if _type == RANDOM_TYPE_UNSET {
        RANDOM_TYPE_LINES
    } else {
        _type
    };

    let mut reader = BufReader::new(file);
    let mut content = String::new();
    if reader.read_to_string(&mut content).is_err() {
        eprintln!("read: {}", io::Error::last_os_error());
        process::exit(1);
    }

    let tokens: Vec<String> = if type_ == RANDOM_TYPE_WORDS {
        content.split_whitespace().map(|s| s.to_string()).collect()
    } else {
        content.lines().map(|s| format!("{}\n", s)).collect()
    };

    let mut indices: Vec<usize> = (0..tokens.len()).collect();
    let mut stdout = io::stdout();

    for _ in 0..indices.len() {
        if indices.is_empty() {
            break;
        }
        let mut buf: [u8; 4] = [0; 4];
        let mut f = File::open("/dev/urandom").unwrap_or_else(|_| {
            eprintln!("cannot open /dev/urandom");
            process::exit(1);
        });
        f.read_exact(&mut buf).unwrap_or_else(|_| {
            eprintln!("cannot read from /dev/urandom");
            process::exit(1);
        });
        let val = u32::from_ne_bytes(buf);
        let selected = (val as usize) % indices.len();
        let idx = indices[selected];

        if random_uniform_denom(denom) {
            if stdout.write_all(tokens[idx].as_bytes()).is_err() {
                eprintln!("printf: {}", io::Error::last_os_error());
                process::exit(1);
            }
        }
        if unique {
            indices.remove(selected);
        }
    }

    if stdout.flush().is_err() {
        eprintln!("flush: {}", io::Error::last_os_error());
        process::exit(1);
    }

    0
}
