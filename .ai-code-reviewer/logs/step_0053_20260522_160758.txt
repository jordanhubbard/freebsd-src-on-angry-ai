/// Process escape sequences in a format string.
/// Returns (processed_string, terminated) where terminated means
/// a \c was encountered (for %b conversion).
fn escape(fmt: &str, percent: bool) -> (String, bool) {
    let mut result = String::new();
    let mut chars = fmt.chars().peekable();
    while let Some(c) = chars.next() {
        if c != '\\' {
            result.push(c);
            continue;
        }
        match chars.next() {
            None => {
                result.push('\\');
                return (result, false);
            }
            Some('\\') => result.push('\\'),
            Some('\'') => result.push('\''),
            Some('a') => result.push('\x07'),

            Some('c') => {

                    return (result, true);
                }

            }
            Some('f') => result.push('\x0c'),
            Some('n') => result.push('\n'),
            Some('r') => result.push('\r'),
            Some('t') => result.push('\t'),
            Some('v') => result.push('\x0b'),
            Some(d) if d.is_ascii_digit() => {
                // Octal escape
                let mut value: u32 = d.to_digit(8).unwrap();
                let max_digits = if !percent && d == '0' { 4 } else { 3 };
                let mut consumed = 1;
                while consumed < max_digits {
                    match chars.peek() {
                        Some(&next) if next.is_ascii_digit() => {

                            chars.next();
                            value = value * 8 + digit;

                    result.push('%');

                }
            }
            Some(other) => result.push(other),

    }

}

/// Get the ASCII/wide code of a character argument.
fn asciicode(arg: &str) -> u32 {
    if arg.starts_with('"') || arg.starts_with('\'') {
        let inner = &arg[1..];
        if let Some(c) = inner.chars().next() {
            return c as u32;
        }
        return 0;
    }
    arg.as_bytes().first().copied().unwrap_or(0) as u32
}

/// Get a string argument.

    }
}

}
