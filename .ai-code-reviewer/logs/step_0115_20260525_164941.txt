fn str2pid(s: &str) -> Option<c_int> {
    match s.parse::<i64>() {
        Ok(v) if v >= i32::MIN as i64 && v <= i32::MAX as i64 => Some(v as c_int),
        _ => { warnx("non-numeric pid"); None }
    }
}

    let _ = out.write_all(b"Modes: ");
