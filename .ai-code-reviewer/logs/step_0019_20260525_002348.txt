			if (error)

			break;

			break;

			if (outfd == -1)
				err(EX_CANTCREAT, "%s", optarg);
			break;

			error = parse_part(optarg);
			if (error)
				errc(EX_DATAERR, error, "partition");
			break;

			if (ep == optarg || *ep != '\0')
