fn recv_from_stdin(buf: &mut [u8]) -> io::Result<usize> {
    let mut stdin = io::stdin();
    let mut remaining = buf;
    let mut total = 0;
    while !remaining.is_empty() {
        match stdin.read(remaining) {
            Ok(0) => return Ok(total),
            Ok(n) => {
                total += n;
                remaining = &mut remaining[n..];
            }
            Err(e) if e.kind() == io::ErrorKind::Interrupted => continue,
            Err(e) => return Err(e),
        }
    }
    Ok(total)
}

fn mailfor(name: &[u8]) {
    // Find '@' to split username from offset
    let at_pos = match name.iter().position(|&b| b == b'@') {
        Some(pos) => pos,
        None => return,
    };

    let username = &name[..at_pos];
    let rest = &name[at_pos + 1..];

    // Parse offset
    let offset_str: String = rest.iter()
        .take_while(|&&b| b != b':')
        .map(|&b| b as char)
        .collect();
    let offset: i64 = offset_str.parse().unwrap_or(0);

    // Determine file path
    let colon_pos = rest.iter().position(|&b| b == b':');
    let mut file_path = String::with_capacity(MAXPATHLEN);
    let mut folder = false;

    if let Some(cp) = colon_pos {
        let file_part = &rest[cp + 1..];
        if !file_part.contains('/') {
            file_path = format!("{}/{}/{}", PATH_MAILDIR, file_part, "");
            folder = true;
        } else {
            file_path = format!("{}/{}", PATH_MAILDIR, String::from_utf8_lossy(username));
        }
    } else {
        file_path = format!("{}/{}", PATH_MAILDIR, String::from_utf8_lossy(username));
    }

    // Look up user in utmpx
    unsafe {
        libc::setutxent();
    }
    loop {
        let utp = unsafe { libc::getutxent() };
        if utp.is_null() {
            break;
        }
        let ut_type = unsafe { (*utp).ut_type };
        if ut_type == libc::USER_PROCESS {
            let ut_user = unsafe { std::ffi::CStr::from_ptr((*utp).ut_user.as_ptr()) };
            if ut_user.to_bytes() == username {
                notify(utp, &file_path, offset, folder);
            }
        }
    }
    unsafe {
        libc::endutxent();
    }
}

fn notify(utp: *mut libc::utmpx, file: &str, offset: i64, folder: bool) {
    unsafe {
        let ut_line = (*utp).ut_line;
        let mut s = std::ffi::CStr::from_ptr(ut_line.as_ptr()).to_bytes();

        // Skip "pts/" prefix
        if s.starts_with(b"pts/") {
            s = &s[4..];
        }

        // Security check: no slashes allowed
        if s.contains(&b'/') {
            libc::syslog(libc::LOG_AUTH | libc::LOG_NOTICE,
                         b"Unexpected `/' in `%s'\0".as_ptr() as *const i8,
                         ut_line.as_ptr() as *const i8);
            return;
        }

        // Build tty path
        let mut tty = [0u8; 20];
        let dev_len = PATH_DEV.len();
        let line_len = s.len();

            return;

        tty[dev_len..dev_len + line_len].copy_from_slice(s);

        let mut stb: libc::stat = std::mem::zeroed();

            dsyslog!(libc::LOG_DEBUG,
                     b"%s: wrong mode on %s\0".as_ptr() as *const i8,
                     (*utp).ut_user.as_ptr() as *const i8,
                     tty_cstr.as_ptr());
            return;
        }

                 b"notify %s on %s\0".as_ptr() as *const i8,
                 (*utp).ut_user.as_ptr() as *const i8,
                 tty_cstr.as_ptr());

            -1 => {
                libc::syslog(libc::LOG_NOTICE, b"fork failed (%m)\0".as_ptr() as *const i8);
                return;

            _ => {

            || libc::setgid((*pw).pw_gid) == -1

            libc::_exit(1);

fn jkfprintf(tp: &mut dyn Write, file: &str, offset: i64, cr: &str) {
    let mut fi = match std::fs::File::open(file) {
        Ok(f) => f,
        Err(_) => return,
    };

    // Seek to offset
    if let Err(_) = fi.seek(io::SeekFrom::Current(offset)) {
        return;
    }

    let mut linecnt = 7;
    let mut charcnt = 560;
    let mut inheader = true;
    let mut line = [0u8; BUFSIZ];

    loop {
        // Read a line
        let mut bytes_read = 0;
        let mut found_newline = false;
        loop {
            let mut buf = [0u8; 1];
            match fi.read(&mut buf) {
                Ok(0) => break,
                Ok(_) => {
                    if bytes_read < BUFSIZ - 1 {
                        line[bytes_read] = buf[0];
                        bytes_read += 1;
                    }
                    if buf[0] == b'\n' {
                        found_newline = true;
                        break;
                    }
                }
                Err(_) => break,
            }
        }
        if bytes_read == 0 && !found_newline {
            break;
        }
        line[bytes_read] = b'\0';

        let line_slice = &line[..bytes_read];

        if inheader {
            if line_slice.first() == Some(&b'\n') {
                inheader = false;
                continue;
            }
            if line_slice.first() == Some(&b' ') || line_slice.first() == Some(&b'\t') {
                continue;
            }
            if !line_slice.starts_with(b"From:") && !line_slice.starts_with(b"Subject:") {
                continue;
            }
        }

        if linecnt <= 0 || charcnt <= 0 {
            let _ = tp.write_all(b"...more...");
            let _ = tp.write_all(cr.as_bytes());
            return;
        }

        // Strip weird characters
        for &ch in line_slice {
            if charcnt <= 0 {
                break;
            }

            let mut out_ch = ch;
            let mut prefix = "";

            if (ch & 0x80) != 0 && ch < 0xA0 {
                prefix = "M-";
                out_ch &= !0x80;
            }

            if (ch & 0x80) == 0 && !is_printable(ch) && !is_space(ch) && ch != b'\a' && ch != b'\b' {
                if is_ctrl(out_ch) {
                    out_ch ^= 0x40;
                    let _ = tp.write_all(b"^");
                }
            }

            if !prefix.is_empty() {
                let _ = tp.write_all(prefix.as_bytes());
            }
            let _ = tp.write_all(&[out_ch]);
            charcnt -= 1;
        }

        let _ = tp.write_all(cr.as_bytes());
        linecnt -= 1;
    }

    let _ = tp.write_all(b"----");
    let _ = tp.write_all(cr.as_bytes());
    let _ = tp.write_all(b"\n");
}

fn is_printable(ch: u8) -> bool {
    ch >= 0x20 && ch < 0x7F
}

fn is_space(ch: u8) -> bool {
    matches!(ch, b' ' | b'\t' | b'\n' | b'\r' | b'\x0C' | b'\x0B')
}

fn is_ctrl(ch: u8) -> bool {
    ch < 0x20 || ch == 0x7F
}
