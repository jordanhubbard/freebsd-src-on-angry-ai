extern "C" fn sig_handler(signo: i32) {
    if signo == KILLSIG.load(Ordering::Relaxed) {
        SIG_TERM.store(true, Ordering::Relaxed);
        return;
    }
    match signo {
        SIGCHLD => SIG_CHLD.store(true, Ordering::Relaxed),
        SIGALRM => SIG_ALRM.store(true, Ordering::Relaxed),
        SIGHUP|SIGINT|SIGQUIT|SIGILL|SIGTRAP|SIGABRT|SIGEMT|SIGFPE|SIGBUS
        |SIGSEGV|SIGSYS|SIGPIPE|SIGTERM|SIGXCPU|SIGXFSZ|SIGVTALRM|SIGPROF
        |SIGUSR1|SIGUSR2 => {
            SIG_TERM_VAL.store(signo, Ordering::Relaxed);
            SIG_TERM.store(true, Ordering::Relaxed);
        }
        _ => {
            SIG_OTHER_VAL.store(signo, Ordering::Relaxed);
            SIG_OTHER.store(true, Ordering::Relaxed);
        }
    }
}
