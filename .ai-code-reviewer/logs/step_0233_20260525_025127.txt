mod qp;
mod uudecode;
mod uuencode;

use std::env;
use std::ffi::OsString;
use std::os::unix::ffi::OsStrExt;
use std::process;

enum Coder {
    Uuencode,
    Uudecode,
    B64encode,
    B64decode,
    Base64,
    Qp,
}

fn search(progname: &str) -> Option<Coder> {
    match progname {
        "uuencode" => Some(Coder::Uuencode),
        "uudecode" => Some(Coder::Uudecode),
        "b64encode" => Some(Coder::B64encode),
        "b64decode" => Some(Coder::B64decode),
        "base64" => Some(Coder::Base64),
        "qp" => Some(Coder::Qp),
        _ => None,
    }
}

fn main() {
    let args: Vec<OsString> = env::args_os().collect();
    let progname = if args.len() > 0 {
        args[0].as_encoded_bytes()
    } else {
        b"bintrans"
    };
    let progname = std::str::from_utf8(progname).unwrap_or("bintrans");

    let coder = search(progname).or_else(|| {
        if args.len() > 1 {
            let subcmd = args[1].as_encoded_bytes();
            let subcmd = std::str::from_utf8(subcmd).unwrap_or("");
            search(subcmd).map(|c| {
                // Skip the subcommand argument
                let rest: Vec<OsString> = args[2..].to_vec();
                let _ = rest; // will be handled by the match below
                c
            })
        } else {
            None
        }
    });

    match coder {
        Some(Coder::Uuencode) | Some(Coder::B64encode) => {
            let args: Vec<String> = env::args().skip(1).collect();
            uuencode::main_encode(&args);
        }
        Some(Coder::Uudecode) | Some(Coder::B64decode) => {
            let args: Vec<String> = env::args().skip(1).collect();
            uudecode::main_decode(&args);
        }
        Some(Coder::Base64) => {
            let args: Vec<String> = env::args().skip(1).collect();
            uuencode::base64_encode_or_decode(&args);
        }
        Some(Coder::Qp) => {
            let args: Vec<String> = env::args().skip(1).collect();
            qp::main_quotedprintable(&args);
        }
        None => {
            eprintln!(
                "usage: {} <uuencode | uudecode> ...\n\
                 {} <b64encode | b64decode> ...\n\
                 {} <base64> ...\n\
                 {} <qp> ...",
                progname, progname, progname, progname
            );
            process::exit(64); // EX_USAGE
        }
    }
}
