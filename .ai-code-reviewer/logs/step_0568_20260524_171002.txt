const TAB: usize = 8;

fn get_terminal_width() -> io::Result<usize> {
    use libc::{ioctl, winsize, TIOCGWINSZ};
    let mut win: winsize = unsafe { std::mem::zeroed() };
    if unsafe { ioctl(1, TIOCGWINSZ, &mut win) } == 0 {
        Ok(win.ws_col as usize)
    } else {
        Err(io::Error::last_os_error())
    }
}

fn input<R: Read>(reader: &mut dyn BufRead, lines: &mut Vec<String>) {
    let mut buf = String::new();
    loop {
        buf.clear();
        match reader.read_line(&mut buf) {
            Ok(0) => break,
            Ok(_) => {
                // Strip trailing newline
                if buf.ends_with('\n') {
                    buf.pop();
                    if buf.ends_with('\r') {
                        buf.pop();
                    }
                }
                // Skip leading whitespace
                let trimmed = buf.trim_start();
                if trimmed.is_empty() {
                    continue;
                }
                lines.push(trimmed.to_string());
            }
            Err(_) => break,
        }
    }
}

fn width(s: &str) -> usize {
    let mut w = 0;
    for c in s.chars() {
        let cw = char_width(c);
        if cw > 0 {
            w += cw;
        }
    }
    w
}

fn char_width(c: char) -> usize {
    // Simple width calculation: most printable chars are 1, CJK etc are 2
    // This is a simplified version of wcwidth
    if c.is_control() || c == '\t' {
        return 0;
    }
    // CJK and other wide characters
    if c >= '\u{1100}' && c <= '\u{115F}' { return 2; }
    if c >= '\u{2E80}' && c <= '\u{A4CF}' && c != '\u{303F}' { return 2; }
    if c >= '\u{AC00}' && c <= '\u{D7A3}' { return 2; }
    if c >= '\u{F900}' && c <= '\u{FAFF}' { return 2; }
    if c >= '\u{FE10}' && c <= '\u{FE19}' { return 2; }
    if c >= '\u{FE30}' && c <= '\u{FE6F}' { return 2; }
    if c >= '\u{FF00}' && c <= '\u{FF60}' { return 2; }
    if c >= '\u{FFE0}' && c <= '\u{FFE6}' { return 2; }
    if c >= '\u{20000}' && c <= '\u{2FFFD}' && c != '\u{E0001}' { return 2; }
    if c >= '\u{30000}' && c <= '\u{3FFFD}' && c != '\u{E0020}' { return 2; }
    1
}

fn roundup(x: usize, y: usize) -> usize {
    ((x + y - 1) / y) * y
}

fn r_columnate(lines: &[String], termwidth: usize, maxlength: usize) {
    let numcols = termwidth / maxlength;
    if numcols == 0 {
        print_lines(lines);
        return;
    }
    let entries = lines.len();
    let numrows = (entries + numcols - 1) / numcols;

    for row in 0..numrows {
        let mut chcnt = 0;
        let mut endcol = maxlength;
        let mut col = 0;
        let mut base = row;
        let mut first = true;

        while col < numcols && base < entries {
            if !first {
                // Print tabs to align
                let mut cnt = roundup(chcnt + 1, TAB);
                while cnt <= endcol {
                    print!("\t");
                    cnt = roundup(cnt + 1, TAB);
                }
                endcol += maxlength;
            }
            print!("{}", lines[base]);
            chcnt += width(&lines[base]);
            first = false;
            col += 1;
            base += numrows;
        }
        println!();
    }
}

fn c_columnate(lines: &[String], termwidth: usize, maxlength: usize) {
    let numcols = termwidth / maxlength;
    if numcols == 0 {
        print_lines(lines);
        return;
    }
    let entries = lines.len();
    let mut lp = 0;
    let mut chcnt = 0;
    let mut col = 0;
    let mut endcol = maxlength;

    while lp < entries {
        if lp > 0 {
            if col < numcols {
                // Print tabs to align
                let mut cnt = roundup(chcnt + 1, TAB);
                while cnt <= endcol {
                    print!("\t");
                    cnt = roundup(cnt + 1, TAB);
                }
                endcol += maxlength;
            }
        }
        print!("{}", lines[lp]);
        chcnt += width(&lines[lp]);
        lp += 1;
        col += 1;
        if col == numcols {
            println!();
            chcnt = 0;
            col = 0;
            endcol = maxlength;
        }
    }
    if chcnt > 0 {
        println!();
    }
}

fn print_lines(lines: &[String]) {
    for line in lines {
        println!("{}", line);
    }
}

fn maketbl(lines: &[String], separator: &str, tblcols: Option<usize>) {
    if lines.is_empty() {
        return;
    }

    // Parse each line into columns
    let mut table: Vec<Vec<String>> = Vec::new();
    let mut max_cols = 0;

    for line in lines {
        let mut cols: Vec<String> = Vec::new();
        let mut current = String::new();
        let mut in_sep = false;

        for c in line.chars() {
            if separator.contains(c) {
                if !in_sep {
                    cols.push(current.clone());
                    current.clear();
                    in_sep = true;
                }
            } else {
                current.push(c);
                in_sep = false;
            }
        }
        if !current.is_empty() || !cols.is_empty() {
            cols.push(current);
        }
        if cols.len() > max_cols {
            max_cols = cols.len();
        }
        table.push(cols);
    }

    // Calculate max width for each column
    let mut col_widths: Vec<usize> = vec![0; max_cols];
    for row in &table {
        for (i, col) in row.iter().enumerate() {
            let w = width(col);
            if i < col_widths.len() && w > col_widths[i] {
                col_widths[i] = w;
            }
        }
    }

    // Print table
    for row in &table {
        for (i, col) in row.iter().enumerate() {
            if i > 0 {
                print!(" ");
            }
            print!("{}", col);
            if i < row.len() - 1 {
                let padding = col_widths[i].saturating_sub(width(col)) + 1;
                for _ in 0..padding {
                    print!(" ");
                }
            }
        }
        println!();
    }
}
