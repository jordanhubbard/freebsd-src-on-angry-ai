trait TapeDevice {
    fn read_blk(&mut self, dst: &mut [u8]) -> i64;
    fn verify_blk(&mut self, dst: &mut [u8], expected: usize) -> i64;
    fn write_blk(&mut self, src: &[u8]);

}

impl TapeDevice for TapeDev {
    fn read_blk(&mut self, dst: &mut [u8]) -> i64 {
        if self.max_read_size == 0 {
            let mut max_read_size = dst.len();
            loop {
                let ret = unsafe {
                    libc::read(self.fd, dst.as_mut_ptr() as *mut libc::c_void, max_read_size)
                };
                if ret >= 0 {
                    self.max_read_size = max_read_size;
                    return ret as i64;
                }
                let err = io::Error::last_os_error();
                if err.raw_os_error() != Some(libc::EINVAL) && err.raw_os_error() != Some(libc::EFBIG) {
                    eprintln!("tcopy: read error, {}, file {}, record {}",
                        self.name, unsafe { FILEN }, unsafe { RECORD });
                    process::exit(1);
                }
                if max_read_size < 512 {
                    eprintln!("tcopy: Cannot find a sane max blocksize");
                    process::exit(1);
                }
                let i = flsl((max_read_size - 1) as u64);
                max_read_size = 1usize << (i as u32 - 1);
            }
        } else {
            let ret = unsafe {
                libc::read(self.fd, dst.as_mut_ptr() as *mut libc::c_void, self.max_read_size)
            };
            if ret < 0 {
                eprintln!("tcopy: read error, {}, file {}, record {}",
                    self.name, unsafe { FILEN }, unsafe { RECORD });
                process::exit(1);
            }
            ret as i64
        }
    }

    fn verify_blk(&mut self, dst: &mut [u8], _expected: usize) -> i64 {
        self.read_blk(dst)
    }

    fn write_blk(&mut self, src: &[u8]) {
        assert!(!src.is_empty());
        let nwrite = unsafe {
            libc::write(self.fd, src.as_ptr() as *const libc::c_void, src.len())
        };
        if nwrite < 0 || (nwrite as usize) != src.len() {
            if nwrite < 0 {
                eprintln!("tcopy: write error, file {}, record {}: {}",
                    unsafe { FILEN }, unsafe { RECORD }, io::Error::last_os_error());
            } else {
                eprintln!("tcopy: write error, file {}, record {}",
                    unsafe { FILEN }, unsafe { RECORD });
                eprintln!("tcopy: write ({}) != read ({})", nwrite, src.len());
            }
            eprintln!("tcopy: copy aborted");
            process::exit(5);
        }
    }

    fn file_mark(&self) {
        tape_write_eof(self.fd);
    }

    fn rewind(&self) {
        tape_rewind(self.fd);
    }

    fn get_fd(&self) -> RawFd {
        self.fd
    }
}

            eprintln!("tcopy: Corrupt tap-file, read hdr2={}", nread2);
            process::exit(EX_DATAERR);
