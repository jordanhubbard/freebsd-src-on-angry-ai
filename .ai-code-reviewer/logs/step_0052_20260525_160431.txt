fn stripsuffix(p: &mut String, suffix: &str) {
    if suffix.is_empty() {
        return;
    }
    let plen = p.len();
    let slen = suffix.len();
    if slen >= plen {
        return;
    }
    let start = plen - slen;
    // Check that suffix matches at the end
    if &p[start..] != suffix {
        return;
    }
    // Ensure the match occurred on a character boundary
    if !p.is_char_boundary(start) {
        return;
    }
    p.truncate(start);
}

fn strcmp_cstr(a: &CStr, b: &CStr) -> c_int {
    let a_bytes = a.to_bytes();
    let b_bytes = b.to_bytes();
    let len = a_bytes.len().min(b_bytes.len());
    for i in 0..len {
        if a_bytes[i] != b_bytes[i] {
            return a_bytes[i] as c_int - b_bytes[i] as c_int;
        }
    }
    (a_bytes.len() as c_int) - (b_bytes.len() as c_int)
}
