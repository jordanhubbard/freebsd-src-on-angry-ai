use std::env;
use std::ffi::CString;
use std::os::unix::fs::DirBuilderExt;
use std::path::Path;
use std::process;

fn usage() {
    eprintln!(
        "usage: mktemp [-d] [-p tmpdir] [-q] [-t prefix] [-u] template \
         ...\n\
         \t       mktemp [-d] [-p tmpdir] [-q] [-u] -t prefix"
    );
    process::exit(1);
}

fn main() {
    let args: Vec<String> = env::args().collect();

    let mut dflag = false;
    let mut qflag = false;
    let mut tflag = false;
    let mut uflag = false;
    let mut tmpdir: Option<String> = None;
    let mut prefix = String::from("mktemp");
    let mut prefer_tmpdir = true;
    let mut optind = 1;

    let mut i = 1;
    while i < args.len() {
        let arg = &args[i];
        if arg == "-d" || arg == "--directory" {
            dflag = true;
            i += 1;
        } else if arg == "-p" {
            // -p tmpdir or -ptmpdir
            if i + 1 < args.len() && !arg.starts_with("-p") || arg.len() == 2 {
                // Need next arg for value
                if i + 1 >= args.len() {
                    // -p with no argument means use TMPDIR
                    tmpdir = env::var("TMPDIR").ok();
                    prefer_tmpdir = false;
                    i += 1;
                } else {
                    let val = &args[i + 1];
                    if val.is_empty() {
                        tmpdir = env::var("TMPDIR").ok();
                    } else {
                        tmpdir = Some(val.clone());
                    }
                    prefer_tmpdir = false;
                    i += 2;
                }
            } else {
                // -ptmpdir combined form
                let val = &arg[2..];
                if val.is_empty() {
                    tmpdir = env::var("TMPDIR").ok();
                } else {
                    tmpdir = Some(val.to_string());
                }
                prefer_tmpdir = false;
                i += 1;
            }
        } else if arg == "--tmpdir" {
            // --tmpdir with optional argument
            if i + 1 < args.len() && !args[i + 1].starts_with('-') {
                tmpdir = Some(args[i + 1].clone());
                prefer_tmpdir = false;
                i += 2;
            } else {
                tmpdir = env::var("TMPDIR").ok();
                prefer_tmpdir = false;
                i += 1;
            }
        } else if arg == "-q" || arg == "--quiet" {
            qflag = true;
            i += 1;
        } else if arg == "-t" {
            if i + 1 >= args.len() {
                usage();
            }
            prefix = args[i + 1].clone();
            tflag = true;
            i += 2;
        } else if arg == "-u" || arg == "--dry-run" {
            uflag = true;
            i += 1;
        } else if arg.starts_with('-') {
            usage();
        } else {
            optind = i;
            break;
        }
    }

    let remaining: Vec<String> = args[optind..].to_vec();

    if !tflag && remaining.is_empty() {
        tflag = true;
        prefix = String::from("tmp");
        prefer_tmpdir = false;
    }

    // Determine tmpdir for -t mode
    let mut name: Option<String> = None;
    if tflag {
        let envtmp = if prefer_tmpdir || tmpdir.is_none() {
            env::var("TMPDIR").ok()
        } else {
            None
        };
        let tmpdir = envtmp.as_ref().or(tmpdir.as_ref());
        let tmpdir = match tmpdir {
            Some(d) => d.as_str(),
            None => "/tmp",
        };
        let suffix = if tmpdir.ends_with('/') { "" } else { "/" };
        name = Some(format!("{}{}{}.XXXXXXXXXX", tmpdir, suffix, prefix));
    }

    let mut ret = 0;

    loop {
        let tmpl = match &name {
            Some(n) => n.clone(),
            None => {
                if remaining.is_empty() {
                    break;
                }
                let arg = &remaining[0];
                let rest = &remaining[1..];
                remaining = rest.to_vec();
                if !tflag {

                        format!("{}/{}", td, arg)
                    } else {
                        arg.clone()
                    }
                } else {
                    arg.clone()

                Ok(created) => {
                    println!("{}", created);

    }

/// Rust equivalent of mkdtemp: create a unique directory from a template
/// ending in XXXXXXXXXX.
fn mkdtemp(template: &str) -> Result<String, std::io::Error> {
    let tmpl_c = CString::new(template)?;
    let result = unsafe { libc_mkdtemp(tmpl_c.as_ptr()) };
    if result.is_null() {
        return Err(std::io::Error::last_os_error());
    }
    let path = unsafe { std::ffi::CStr::from_ptr(result) }
        .to_string_lossy()
        .into_owned();
    unsafe { libc_free(result as *mut std::ffi::c_void) };
    Ok(path)
}

/// Rust equivalent of mkstemp: create a unique file from a template
/// ending in XXXXXXXXXX, then close the fd and return the path.
fn mkstemp(template: &str) -> Result<String, std::io::Error> {
    let tmpl_c = CString::new(template)?;
    let result = unsafe { libc_mkstemp(tmpl_c.as_ptr()) };
    if result.is_null() {
        return Err(std::io::Error::last_os_error());
    }
    let path = unsafe { std::ffi::CStr::from_ptr(result) }
        .to_string_lossy()
        .into_owned();
    Ok(path)
}

extern "C" {
    fn mkdtemp(template: *mut std::ffi::c_char) -> *mut std::ffi::c_char;
    fn mkstemp(template: *mut std::ffi::c_char) -> std::os::raw::c_int;
    fn free(p: *mut std::ffi::c_void);
}

fn libc_mkdtemp(template: *mut std::ffi::c_char) -> *mut std::ffi::c_char {
    unsafe { mkdtemp(template) }
}

fn libc_mkstemp(template: *mut std::ffi::c_char) -> *mut std::ffi::c_char {
    // mkstemp returns int fd, but modifies template in place
    // We need to get the fd and also read back the template
    let fd = unsafe { mkstemp(template) };
    if fd < 0 {
        std::ptr::null_mut()
    } else {
        // Close the fd
        unsafe { libc_close(fd) };
        template
    }
}

fn libc_free(p: *mut std::ffi::c_void) {
    unsafe { free(p) }
}

extern "C" {
    fn close(fd: std::os::raw::c_int) -> std::os::raw::c_int;
}

fn libc_close(fd: std::os::raw::c_int) -> std::os::raw::c_int {
    unsafe { close(fd) }
}
