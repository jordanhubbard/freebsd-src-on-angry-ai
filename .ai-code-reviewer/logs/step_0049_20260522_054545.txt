use std::ffi::CString;
use std::os::raw::{c_char, c_int, mode_t};
use std::process;

const STDIN_FILENO: c_int = 0;
const STDOUT_FILENO: c_int = 1;
const STDERR_FILENO: c_int = 2;

const S_IXUSR: mode_t = 0o100;
const S_IXGRP: mode_t = 0o010;

extern "C" {
    fn ttyname(fd: c_int) -> *mut c_char;
    fn stat(path: *const c_char, sb: *mut libc::stat) -> c_int;
    fn chmod(path: *const c_char, mode: mode_t) -> c_int;
}

fn usage() -> ! {
    eprintln!("usage: biff [n | y | b]");
    process::exit(2);
}

fn main() {
    let args: Vec<String> = std::env::args().collect();

    // biff takes no options, only positional args
    if args.len() > 2 {
        usage();
    }

    // Find tty name
    let mut name_ptr: *mut c_char = std::ptr::null_mut();
    unsafe {
        name_ptr = ttyname(STDIN_FILENO);
        if name_ptr.is_null() {
            name_ptr = ttyname(STDOUT_FILENO);
        }
        if name_ptr.is_null() {
            name_ptr = ttyname(STDERR_FILENO);
        }
        if name_ptr.is_null() {
            eprintln!("biff: unknown tty");
            process::exit(2);
        }
    }

    let name_cstr = unsafe { std::ffi::CStr::from_ptr(name_ptr) };
    let name = name_cstr.to_string_lossy().into_owned();

    // Stat the tty
    let name_cstring = CString::new(name.as_bytes()).unwrap_or_else(|_| usage());
    let mut sb: libc::stat = unsafe { std::mem::zeroed() };
    let ret = unsafe { stat(name_cstring.as_ptr(), &mut sb) };
    if ret != 0 {
        eprintln!("biff: stat: {}", std::io::Error::last_os_error());
        process::exit(2);
    }

    let mode = sb.st_mode;

    if args.len() == 1 {
        // No argument: query mode
        let result = if mode & S_IXUSR != 0 {
            "y"
        } else if mode & S_IXGRP != 0 {
            "b"
        } else {
            "n"
        };
        println!("is {result}");
        process::exit(if mode & (S_IXUSR | S_IXGRP) != 0 { 0 } else { 1 });
    }

    let arg = &args[1];
    let new_mode = match arg.as_str() {
        "n" => mode & !(S_IXUSR | S_IXGRP),
        "y" => (mode & !(S_IXUSR | S_IXGRP)) | S_IXUSR,
        "b" => (mode & !(S_IXUSR | S_IXGRP)) | S_IXGRP,
        _ => usage(),
    };

    if unsafe { chmod(name_cstring.as_ptr(), new_mode) } < 0 {
        eprintln!("biff: {}: {}", name, std::io::Error::last_os_error());
        process::exit(2);
    }

    // After setting, check if any execute bit is set
    // Re-stat to get the actual mode after chmod
    let ret = unsafe { stat(name_cstring.as_ptr(), &mut sb) };
    if ret != 0 {
        eprintln!("biff: stat: {}", std::io::Error::last_os_error());
        process::exit(2);
    }

    process::exit(if sb.st_mode & (S_IXUSR | S_IXGRP) != 0 { 0 } else { 1 });
}
