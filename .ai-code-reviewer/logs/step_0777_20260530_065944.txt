// EFI types (minimal definitions matching EFI headers)
pub type EFI_GUID = Guid;
pub type EFI_STATUS = i64;

unsafe fn format_ptr(buf: &mut [c_char], ptr: *const core::ffi::c_void) {
    let hex = format!("0x{:016x}", ptr as usize);

/// Main ACPI detection function
/// Looks for ACPI RSDP table and sets environment variables with ACPI info.
#[no_mangle]
pub unsafe extern "C" fn acpi_detect() {
    let mut buf = [0i8; 64];

    feature_enable(FEATURE_EARLY_ACPI);

    // Try ACPI 2.0 table first, then ACPI 1.0
    let mut rsdp_ptr = efi_get_table(&ACPI20_GUID as *const _);
    if rsdp_ptr.is_null() {
        rsdp_ptr = efi_get_table(&ACPI_GUID as *const _);
    }
    if rsdp_ptr.is_null() {
        return;
    }

    RSDP = rsdp_ptr as *mut AcpiTableRsdp;
    let rsdp = &*RSDP;

    // Set acpi.rsdp environment variable
    format_ptr(&mut buf, rsdp_ptr);
    setenv_str("acpi.rsdp", core::str::from_utf8_unchecked(&buf[..]), 1);

    // Determine revision
    let revision = if rsdp.revision == 0 { 1 } else { rsdp.revision };

    // Set acpi.revision
    format_dec(&mut buf, revision);
    setenv_str("acpi.revision", core::str::from_utf8_unchecked(&buf[..]), 1);

    // Set acpi.oem
    copy_oem_id(&mut buf, &rsdp.oem_id);

    // Set acpi.rsdt

    setenv_str("acpi.rsdt", core::str::from_utf8_unchecked(&buf[..]), 1);

    // ACPI 2.0+ specific fields
    if revision >= 2 {
        format_hex64(&mut buf, rsdp.xsdt_physical_address);
        setenv_str("acpi.xsdt", core::str::from_utf8_unchecked(&buf[..]), 1);

        format_dec(&mut buf, rsdp.length);
        setenv_str("acpi.xsdt_length", core::str::from_utf8_unchecked(&buf[..]), 1);
    }
}
