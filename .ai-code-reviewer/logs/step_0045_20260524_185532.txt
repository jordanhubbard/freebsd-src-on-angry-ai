struct TapeDev {
    fd: i32,
    name: String,
    direction: Direction,
    max_read_size: usize,
}

impl TapeDev {
    fn new(fd: i32, name: &str, destination: bool) -> Self {
        TapeDev {
            fd,
            name: name.to_string(),
            direction: if destination { Direction::Dst } else { Direction::Src },
            max_read_size: 0,
        }
    }

    fn read_blk(&mut self, dst: &mut [u8]) -> i64 {
        let len = dst.len();
        let mut max_read_size = if self.max_read_size == 0 { len } else { self.max_read_size };
        let mut retval: i64 = -1;

        if self.max_read_size == 0 {
            while max_read_size > 0 {
                let buf = &mut dst[..max_read_size];
                retval = unsafe {
                    libc_syscall(SYS_READ, self.fd, buf.as_mut_ptr() as usize, max_read_size)
                };
                if retval >= 0 {
                    self.max_read_size = max_read_size;
                    break;
                }
                // Check errno for EINVAL/EFBIG
                let errno = unsafe { __errno() };
                if errno != 22 && errno != 76 { // EINVAL, EFBIG
                    self.max_read_size = max_read_size;
                    break;
                }
                if max_read_size < 512 {
                    eprintln!("tcopy: Cannot find a sane max blocksize");
                    process_exit(1);
                }
                // Reduce to next lower power of two
                let bits = (max_read_size as u64).next_power_of_two().trailing_zeros();
                max_read_size = 1usize << (bits.saturating_sub(1));
            }
        } else {
            let buf = &mut dst[..self.max_read_size];
            retval = unsafe {
                libc_syscall(SYS_READ, self.fd, buf.as_mut_ptr() as usize, self.max_read_size)
            };
        }

        if retval < 0 {
            eprintln!(
                "tcopy: read error, {}, file {}, record {}",
                self.name, G_FILN, G_RECORD
            );
            process_exit(1);
        }
        retval
    }

    fn verify_blk(&mut self, dst: &mut [u8], _expected: usize) -> i64 {
        self.read_blk(dst)
    }

    fn write_blk(&mut self, src: &[u8]) {
        assert!(!src.is_empty());
        let nwrite = unsafe {
            libc_syscall(SYS_WRITE, self.fd, src.as_ptr() as usize, src.len())
        };
        if nwrite < 0 || (nwrite as usize) != src.len() {
            if nwrite < 0 {
                eprintln!(
                    "tcopy: write error, file {}, record {}",
                    G_FILN, G_RECORD
                );
            } else {
                eprintln!(
                    "tcopy: write error, file {}, record {}",
                    G_FILN, G_RECORD
                );
                eprintln!("tcopy: write ({}) != read ({})", nwrite, src.len());
            }
            eprintln!("tcopy: copy aborted");
            process_exit(5);
        }
    }

    fn file_mark(&mut self) {
        let mut op = Mtop { mt_op: MTWEOF, mt_count: 1 };
        let ret = unsafe {
            libc_syscall(
                SYS_IOCTL,
                self.fd,
                MTIOCTOP,
                &mut op as *mut Mtop as usize,
            )
        };
        if ret < 0 {
            eprintln!("tcopy: tape op (write file mark)");
            process_exit(6);
        }
    }

    fn rewind(&mut self) {
        let mut op = Mtop { mt_op: MTREW, mt_count: 1 };
        let ret = unsafe {
            libc_syscall(
                SYS_IOCTL,
                self.fd,
                MTIOCTOP,
                &mut op as *mut Mtop as usize,
            )
        };
        if ret < 0 {
            eprintln!("tcopy: tape op (rewind)");
            process_exit(6);
        }
    }
}

}

impl TapFile {

        TapFile {
            base: TapeDev::new(fd, name, dst),
        }
    }

    fn read_blk(&mut self, dst: &mut [u8]) -> i64 {
        let mut lbuf = [0u8; 4];
        let nread = full_read(self.base.fd, &mut lbuf);
        if nread == 0 {
            return 0;
        }
        if nread != 4 {
            eprintln!("tcopy: Corrupt tap-file, read hdr1={}", nread);
            process_exit(17);
        }
        let u = u32::from_le_bytes(lbuf) as u64;
        if u == 0 || (u >> 24) == 0xff {
            return 0;
        }
        if u as usize > dst.len() {
            eprintln!("tcopy: tapfile blocksize too big, 0x{:08x}", u);
            process_exit(17);
        }
        let data_len = u as usize;
        let n = unsafe {
            libc_syscall(
                SYS_READ,
                self.base.fd,
                dst[..data_len].as_mut_ptr() as usize,
                data_len,
            )
        };
        if n < 0 {
            eprintln!("tcopy: read error in tap file");
            process_exit(1);
        }
        n
    }

    fn write_blk(&mut self, src: &[u8]) {
        let len = src.len() as u32;
        let hdr = len.to_le_bytes();
        unsafe {
            libc_syscall(SYS_WRITE, self.base.fd, hdr.as_ptr() as usize, 4);
            libc_syscall(SYS_WRITE, self.base.fd, src.as_ptr() as usize, src.len());
        };
    }

    fn file_mark(&mut self) {
        let eof = 0xffffffffu32.to_le_bytes();
        unsafe {
            libc_syscall(SYS_WRITE, self.base.fd, eof.as_ptr() as usize, 4);
        };
    }

    fn rewind(&mut self) {
        // lseek to beginning
        unsafe {
            libc_syscall(59, self.base.fd, 0, 0); // SYS_LSEEK, offset 0, whence SEEK_SET
        };
    }
}
