fn days_in_month(year: i32, month: i32) -> i32 {
    match month {
        0 | 2 | 4 | 6 | 7 | 9 | 11 => 31,
        3 | 5 | 8 | 10 => 30,
        1 => if isleap(year) { 29 } else { 28 },
        _ => 31,
    }
}

fn compute_yday(tm: &Tm) -> i32 {
    let year = 1900 + tm.tm_year;
    let mut yday = 0;
    for m in 0..tm.tm_mon {
        yday += days_in_month(year, m);
    }
    yday + tm.tm_mday - 1
}

fn mktime(tm: &Tm) -> i64 {
    let year = 1900 + tm.tm_year;
    let mut seconds = 0i64;

    // Days from years
    for y in 1970..year {
        seconds += if isleap(y) { 366 } else { 365 } as i64 * 86400;
    }

    // Days from months
    for m in 0..tm.tm_mon {
        seconds += days_in_month(year, m) as i64 * 86400;
    }

    // Days from day of month
    seconds += (tm.tm_mday - 1) as i64 * 86400;

    // Hours, minutes, seconds
    seconds += tm.tm_hour as i64 * 3600;
    seconds += tm.tm_min as i64 * 60;
    seconds += tm.tm_sec as i64;

    seconds
}

fn gmtime(timestamp: i64) -> Tm {
    let mut rem = timestamp;
    let mut sec = (rem % 60) as i32;
    rem /= 60;

    rem /= 60;
    let mut hour = (rem % 24) as i32;
    rem /= 24;

    let mut year = 1970;

    loop {
        let days_in_year = if isleap(year) { 366 } else { 365 };
        if day_of_year < days_in_year {
            break;
        }
        day_of_year -= days_in_year;
        year += 1;
    }

    let mut month = 0i32;
    loop {
        let dim = days_in_month(year, month);
        if day_of_year < dim {
            break;
        }
        day_of_year -= dim;
        month += 1;
    }

    Tm {
        tm_sec: sec,
        tm_min: min,
        tm_hour: hour,
        tm_mday: (day_of_year + 1) as i32,
