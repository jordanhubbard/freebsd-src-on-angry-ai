fn open_arg(name: &str, is_output: bool, rawfile: bool) -> Box<dyn TapeOps> {
    let flags = if is_output { O_WRONLY | O_CREAT | O_TRUNC } else { O_RDONLY };
    let c_name = CString::new(name).unwrap();
    let fd = unsafe { open(c_name.as_ptr(), flags, 0o666) };
    if fd < 0 { eprintln!("tcopy: {}: {}", name, io::Error::last_os_error()); process::exit(1); }
    let mut st: stat = unsafe { std::mem::zeroed() };
    let is_reg = unsafe { fstat(fd, &mut st) == 0 && (st.st_mode & S_IFMT) == S_IFREG };
    if is_reg && (rawfile || is_output) { Box::new(TapFile::new(fd, name)) }
    else if is_reg { Box::new(TapFile::new(fd, name)) }
    else { Box::new(TapeDev::new(fd, name)) }
}
