fn full_read(fd: RawFd, dst: &mut [u8]) -> io::Result<usize> {
    let mut total = 0;
    while total < dst.len() {
        match unsafe { libc::read(fd, dst[total..].as_mut_ptr() as *mut libc::c_void, dst[total..].len()) } {
            -1 => {
                let err = io::Error::last_os_error();
                if err.kind() == io::ErrorKind::WouldBlock || err.kind() == io::ErrorKind::Interrupted {
                    continue;
                }
                return Err(err);
            }
            0 => return Ok(total),
            n => {
                total += n as usize;
                if total == dst.len() {
                    return Ok(total);
                }
            }
        }
    }
    Ok(total)
}

fn expand_number(s: &str) -> Option<i64> {
    let s = s.trim();
    let (num_str, suffix) = if let Some(pos) = s.find(|c: char| c.is_ascii_alphabetic()) {
        (&s[..pos], &s[pos..])
    } else {
        (s, "")
    };
    let num: i64 = num_str.parse().ok()?;
    match suffix {
        "" | "b" => Some(num),
        "k" | "K" => Some(num * 1024),
        "m" | "M" => Some(num * 1024 * 1024),
        "g" | "G" => Some(num * 1024 * 1024 * 1024),
        _ => None,
    }
}

fn ioctl_mtioctop(fd: RawFd, op: &mtop) -> io::Result<()> {
    unsafe {
        let ret = libc::ioctl(fd, MTIOCTOP, op);
        if ret < 0 { Err(io::Error::last_os_error()) } else { Ok(()) }
    }
}

fn tape_rewind(fd: RawFd) {
    let mut op = mtop { mt_op: MTREW, mt_count: 1 };
    if let Err(e) = ioctl_mtioctop(fd, &op) {
        eprintln!("tcopy: tape op (rewind): {}", e);
        process::exit(6);
    }
}

    let t_start = unsafe { T_START };
    let t_end = unsafe { T_END };
