use std::env;
use std::ffi::CString;
use std::io::{self, Write};
use std::os::unix::process::CommandExt;
use std::process::{self, Command};
use std::str;

fn is_magic_no(p: &[u8], magic: u8) -> bool {
    p.len() >= 2
        && p[0] == magic
        && p[1].is_ascii_digit()
        && p[1] != b'0'
}

fn usage() -> ! {
    let _ = io::stderr().write_all(
        b"usage: apply [-a magic] [-d] [-0123456789] command arguments ...\n",
    );
    process::exit(1);
}

fn main() {
    let args: Vec<String> = env::args().collect();
    let mut debug = false;
    let mut magic: u8 = b'%';
    let mut nargs: Option<usize> = None;

    let mut i = 1;
    while i < args.len() {
        let arg = &args[i];
        if arg.starts_with('-') && arg.len() > 1 {
            match arg.as_str() {
                "-a" => {
                    if i + 1 >= args.len() || args[i + 1].len() != 1 {
                        eprintln!("illegal magic character specification");
                        process::exit(1);
                    }
                    magic = args[i + 1].as_bytes()[0];
                    i += 2;
                    continue;
                }
                "-d" => {
                    debug = true;
                    i += 1;
                    continue;
                }
                _ => {
                    // Check for -0 through -9
                    if arg.len() == 2 && arg.as_bytes()[1].is_ascii_digit() {
                        if nargs.is_some() {
                            eprintln!("only one -# argument may be specified");
                            process::exit(1);
                        }
                        nargs = Some((arg.as_bytes()[1] - b'0') as usize);
                        i += 1;
                        continue;
                    }
                    usage();
                }
            }
        } else {
            break;
        }
    }

    let remaining: Vec<String> = args[i..].to_vec();
    if remaining.len() < 2 {
        usage();
    }

    let cmd_template = &remaining[0];
    let mut args_list: Vec<String> = remaining[1..].to_vec();

    // Find the largest %digit reference in the command
    let mut max_n: usize = 0;
    let cmd_bytes = cmd_template.as_bytes();
    let mut p = 0;
    while p < cmd_bytes.len() {
        if is_magic_no(&cmd_bytes[p..], magic) {
            let digit = (cmd_bytes[p + 1] - b'0') as usize;
            if digit > max_n {
                max_n = digit;
            }
            p += 2;
        } else {
            p += 1;
        }
    }

    // Determine shell
    let shell = match env::var("SHELL") {
        Ok(s) => s,
        Err(_) => "/bin/sh".to_string(),
    };
    let shell_name = shell.rsplit('/').next().unwrap_or(&shell);

    // Determine nargs
    let (cmd, nargs_val) = if max_n == 0 {
        // Build command with %digit references
        let mut cmd = cmd_template.clone();
        let effective_nargs = nargs.unwrap_or(1);
        for j in 1..=effective_nargs {
            cmd.push_str(&format!(" {}{}", magic as char, j));
        }
        let final_nargs = if effective_nargs == 0 { 1 } else { effective_nargs };
        (cmd, final_nargs)
    } else {
        (cmd_template.clone(), max_n)
    };

    // Get ARG_MAX
    let arg_max = unsafe { libc::sysconf(libc::_SC_ARG_MAX) };
    let arg_max = if arg_max < 0 {
        128 * 1024
    } else {
        arg_max as usize
    };

    let mut rval = 0;
    let mut arg_idx = 0;

    while arg_idx + nargs_val <= args_list.len() {
        let mut cmdbuf = String::from("exec ");

        // Expand command references
        let cmd_bytes = cmd.as_bytes();
        let mut p = 0;
        while p < cmd_bytes.len() {
            if is_magic_no(&cmd_bytes[p..], magic) {
                let ref_idx = (cmd_bytes[p + 1] - b'0') as usize;
                let arg_val = &args_list[arg_idx + ref_idx];
                cmdbuf.push_str(arg_val);
                p += 2;
            } else {
                cmdbuf.push(cmd_bytes[p] as char);
                p += 1;
            }
            if cmdbuf.len() > arg_max {
                eprintln!("Argument list too long");
                process::exit(1);
            }
        }

        if debug {
            println!("{}", cmdbuf);
        } else {
            let status = Command::new(&shell)
                .arg("-c")
                .arg(&cmdbuf)
                .status();
            match status {
                Ok(s) => {
                    if !s.success() {
                        rval = 1;
                    }
                }
                Err(e) => {
                    eprintln!("{}: {}", shell, e);
                    rval = 1;
                }
            }
        }

        arg_idx += nargs_val;
    }

    if arg_idx != args_list.len() {
        eprintln!(
            "expecting additional argument{} after \"{}\"",
            if nargs_val > arg_idx { "s" } else { "" },
            args_list[arg_idx].as_str()
        );
        process::exit(1);
    }

    process::exit(rval);
}
