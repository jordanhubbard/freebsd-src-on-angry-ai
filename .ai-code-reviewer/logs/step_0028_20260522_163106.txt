/// Get the ASCII code of the first character of an argument.
/// Handles quoted arguments like 'A' or "A".
fn asciicode(arg: &str) -> u32 {
    if arg.starts_with('"') || arg.starts_with('\'') {
        if arg.len() > 1 {
            return arg[1..].chars().next().map(|c| c as u32).unwrap_or(0);
        }
        return 0;
    }
    arg.as_bytes().first().copied().unwrap_or(0) as u32
}

/// Parse an argument as a signed integer.
/// Returns (value, error_occurred).
fn parse_int_arg(arg: &str) -> (i64, bool) {
    if arg.starts_with('"') || arg.starts_with('\'') {
        return (asciicode(arg) as i64, false);
    }
    // Determine base: 0x/0X = 16, 0o/0O = 8, 0b/0B = 2, 0... = 8, else = 10
    let (base, s) = if arg.starts_with("0x") || arg.starts_with("0X") {
        (16, &arg[2..])
    } else if arg.starts_with("0o") || arg.starts_with("0O") {
        (8, &arg[2..])
    } else if arg.starts_with("0b") || arg.starts_with("0B") {
        (2, &arg[2..])
    } else if arg.starts_with('0') && arg.len() > 1 {
        (8, &arg[1..])
    } else {
        (10, arg)
    };
    match i64::from_str_radix(s, base) {
        Ok(v) => (v, false),
        Err(_) => {
            eprintln!("printf: {}: expected numeric value", arg);
            (0, true)
        }
    }
}
