type iconv_t = *mut std::os::raw::c_void;

const INBUFSIZE: usize = 1024;
const OUTBUFSIZE: usize = INBUFSIZE * 2;

fn usage() -> ! {

        "Usage:\t{} [-cs] -f <from_code> -t <to_code> [file ...]\n\
         \t{} -f <from_code> [-cs] [-t <to_code>] [file ...]\n\
         \t{} -t <to_code> [-cs] [-f <from_code>] [file ...]\n\
         \t{} -l",
        prog, prog, prog, prog
    );
    process::exit(1);
}

fn getprogname() -> String {
    std::env::args().next().unwrap_or_else(|| "iconv".to_string())
}

extern "C" fn do_list(n: usize, list: *const *const i8, _data: *mut std::os::raw::c_void) -> i32 {
    let mut stdout = io::stdout();
    for i in 0..n {
        let ptr = unsafe { *list.add(i) };
        let s = unsafe { CStr::from_ptr(ptr).to_string_lossy().into_owned() };
        if i > 0 {
            let _ = stdout.write_all(b" ");
        }
        let _ = stdout.write_all(s.as_bytes());
    }
    let _ = stdout.write_all(b"\n");
    1
}

fn do_conv(
    mut fp: &dyn Read,
    cd: iconv_t,
    silent: bool,
    hide_invalid: bool,
) -> i32 {
    let mut inbuf = vec![0u8; INBUFSIZE];
    let mut outbuf = vec![0u8; OUTBUFSIZE];
    let mut invalids: u64 = 0;

    // Set discard ilseq if -c flag
    if hide_invalid {
        let arg: i32 = 1;
        // ICONV_SET_DISCARD_ILSEQ = 2 on FreeBSD
        let ret = unsafe {
            libc::iconvctl(cd, 2, &arg as *const _ as *mut std::os::raw::c_void)
        };
        if ret == -1 {
            eprintln!("iconvctl(DISCARD_ILSEQ, 1): {}", io::Error::last_os_error());
            process::exit(1);
        }
    }

    loop {
        let inbytes = match fp.read(&mut inbuf) {
            Ok(0) => break,
            Ok(n) => n,
            Err(e) => {
                eprintln!("fread(): {}", e);
                process::exit(1);
            }
        };

        let mut inptr = inbuf.as_ptr() as *const i8;
        let mut inbytesleft = inbytes;

        while inbytesleft > 0 {
            let mut outptr = outbuf.as_mut_ptr() as *mut i8;
            let mut outbytesleft = OUTBUFSIZE;

            let ret = unsafe {
                iconv(cd,
                      &mut inptr as *mut *const i8,
                      &mut inbytesleft as *mut usize,
                      &mut outptr as *mut *mut i8,
                      &mut outbytesleft as *mut usize)
            };

            let written = OUTBUFSIZE - outbytesleft;
            if written > 0 {
                let _ = io::stdout().write_all(&outbuf[..written]);
            }

            if ret == usize::MAX {
                let err = unsafe { libc::__errno() };
                if err != libc::E2BIG {
                    if err != libc::EINVAL || inptr != inbuf.as_ptr() as *const i8 {
                        eprintln!("iconv(): {}", io::Error::from_raw_os_error(err));
                        process::exit(1);
                    }
                    // Incomplete input character - read more
                    let remaining = inbytesleft;
                    let consumed = inbytes - inbytesleft;
                    unsafe {
                        std::ptr::copy(inbuf.as_ptr().add(consumed), inbuf.as_mut_ptr(), remaining);
                    }
                    match fp.read(&mut inbuf[remaining..]) {
                        Ok(0) => {
                            let _ = io::stdout().flush();
                            eprintln!("unexpected end of file; the last character is incomplete.");
                            process::exit(1);
                        }
                        Ok(n) => {
                            inptr = inbuf.as_ptr() as *const i8;
                            inbytesleft = remaining + n;
                        }
                        Err(e) => {
                            eprintln!("fread(): {}", e);
                            process::exit(1);
                        }
                    }
                }
            }
        }
    }

    // Reset shift state
    {
        let mut outptr = outbuf.as_mut_ptr() as *mut i8;
        let mut outbytesleft = OUTBUFSIZE;
        let ret = unsafe {
            iconv(cd,
                  std::ptr::null_mut(),
                  std::ptr::null_mut(),
                  &mut outptr as *mut *mut i8,
                  &mut outbytesleft as *mut usize)
        };
        if ret == usize::MAX {
            eprintln!("iconv(): {}", io::Error::last_os_error());
            process::exit(1);
        }
        let written = OUTBUFSIZE - outbytesleft;
        if written > 0 {
            let _ = io::stdout().write_all(&outbuf[..written]);
        }
    }

    if invalids > 0 && !silent {
        eprintln!("warning: invalid characters: {}", invalids);
    }

    if invalids > 0 { 1 } else { 0 }
}

fn main() {
    let args: Vec<String> = std::env::args().collect();
    let mut opt_c = false;
    let mut opt_s = false;
    let mut opt_f: Option<&str> = None;
    let mut opt_t: Option<&str> = None;
    let mut list_mode = false;
    let mut optind = 1;

    let mut i = 1;
    while i < args.len() {
        let arg = &args[i];
        if arg == "-c" {
            opt_c = true;
            i += 1;
        } else if arg == "-s" {
            opt_s = true;
            i += 1;
        } else if arg == "-l" || arg == "-L" {
            if opt_s || opt_c || opt_f.is_some() || opt_t.is_some() {
                eprintln!("-l is not allowed with other flags.");
                usage();
            }
            list_mode = true;
            i += 1;
        } else if arg == "-f" || arg == "--from-code" {
            if i + 1 < args.len() {
                opt_f = Some(&args[i + 1]);
                i += 2;
            } else {
                usage();
            }
        } else if arg == "-t" || arg == "--to-code" {
            if i + 1 < args.len() {
                opt_t = Some(&args[i + 1]);
                i += 2;
            } else {
                usage();
            }
        } else if arg.starts_with("--from-code=") {
            opt_f = Some(&arg["--from-code=".len()..]);
            i += 1;
        } else if arg.starts_with("--to-code=") {
            opt_t = Some(&arg["--to-code=".len()..]);
            i += 1;
        } else if arg.starts_with("--") {
            usage();
        } else {
            optind = i;
            break;
        }
    }

    if list_mode {
        unsafe {
            iconvlist(do_list, std::ptr::null_mut());
        }
        process::exit(0);
    }

    let from_code = match opt_f {
        Some(s) => s,
        None => {
            if opt_t.is_none() {
                usage();
            }
            ""
        }
    };
    let to_code = match opt_t {
        Some(s) => s,
        None => {
            if opt_f.is_none() {
                usage();
            }
            ""
        }
    };

    if from_code.is_empty() && to_code.is_empty() {
        usage();
    }

    let from_c = CString::new(from_code).unwrap();
    let to_c = CString::new(to_code).unwrap();

    let cd = unsafe { iconv_open(to_c.as_ptr(), from_c.as_ptr()) };
    if cd.is_null() {
        eprintln!("iconv_open({}, {}): {}", to_code, from_code, io::Error::last_os_error());
        process::exit(1);
    }

    let files: Vec<&str> = args[optind..].iter().map(|s| s.as_str()).collect();

    let res = if files.is_empty() {
        let stdin = io::stdin();
        do_conv(&stdin.lock(), cd, opt_s, opt_c)
    } else {
        let mut res = 0i32;
        for (idx, &file) in files.iter().enumerate() {
            let reader: Box<dyn Read> = if file == "-" {
                Box::new(io::stdin().lock())
            } else {
                match std::fs::File::open(file) {
                    Ok(f) => Box::new(f),
                    Err(e) => {
                        eprintln!("Cannot open '{}': {}", file, e);
                        process::exit(1);
                    }
                }
            };
            res |= do_conv(&*reader, cd, opt_s, opt_c);
            // Reset iconv state between files
            unsafe {
                iconv(cd,
                      std::ptr::null_mut(),
                      std::ptr::null_mut(),
                      std::ptr::null_mut(),
                      std::ptr::null_mut());
            }
        }
        res
    };

    unsafe { iconv_close(cd) };

    process::exit(if res == 0 { 0 } else { 1 });
}
