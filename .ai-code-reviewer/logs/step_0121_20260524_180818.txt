struct TapeDev { fd: RawFd, name: String, max_read_size: usize }

impl TapeDev {

    fn read_blk(&mut self, dst: &mut [u8]) -> ssize_t {
        let len = dst.len();
        let mut retval: ssize_t = -1;
        if self.max_read_size == 0 {
            let mut mrs = len;
            while mrs > 0 {
                retval = unsafe { __read(self.fd, dst.as_mut_ptr() as *mut c_void, mrs) };
                if retval >= 0 { break; }
                let err = std::io::Error::last_os_error().raw_os_error().unwrap_or(-1);

                let i = (mrs - 1).next_power_of_two().trailing_zeros() as usize;
                mrs = 1usize << (i - 1);
            }
            self.max_read_size = mrs;
        } else {
            retval = unsafe { __read(self.fd, dst.as_mut_ptr() as *mut c_void, self.max_read_size) };

fn expand_number(s: &str, result: &mut i64) -> bool {
    let s = s.trim();
    let (digits, suffix) = match s.find(|c: char| !c.is_ascii_digit()) { Some(i) => (&s[..i], &s[i..]), None => (s, "") };
    if digits.is_empty() { return true; }
    let mut val: i64 = match digits.parse() { Ok(v) => v, Err(_) => return true };
    match suffix { "" => {} "b" | "B" => { val *= 512; } "k" | "K" => { val *= 1024; } "m" | "M" => { val *= 1024 * 1024; } "g" | "G" => { val *= 1024 * 1024 * 1024; } _ => return true }
    *result = val; false
}
