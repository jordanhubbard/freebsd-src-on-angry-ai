        "n" => mode & !(S_IXUSR | S_IXGRP),
        "y" => (mode & !(S_IXUSR | S_IXGRP)) | S_IXUSR,
        "b" => (mode & !(S_IXUSR | S_IXGRP)) | S_IXGRP,
        _ => usage(),
    };

    let mut perms = metadata.permissions();
    perms.set_mode(new_mode);
    if std::fs::set_permissions(name_path, perms).is_err() {
        eprintln!(
            "biff: {}: {}",
            name_os.to_string_lossy(),
            std::io::Error::last_os_error()
        );
        process::exit(2);
    }

    // Match original C behavior: exit status uses pre-chmod mode (sb.st_mode)
    process::exit(if mode & (S_IXUSR | S_IXGRP) != 0 { 0 } else { 1 });
}
