use std::ffi::CString;
use std::io::{self, Write};
use std::process;

fn usage() -> ! {
    let _ = io::stderr().write_all(
        b"usage: mknod name\n\
         \       mknod name [b | c] major minor [owner:group]\n",
    );
    process::exit(1);
}

fn err_exit(msg: &str) -> ! {
    eprintln!("{}", msg);
    process::exit(1);
}

fn err_sys(msg: &str, e: io::Error) -> ! {
    eprintln!("{}: {}", msg, e);
    process::exit(1);
}

fn id(name: &str, type_name: &str) -> u64 {
    name.parse::<u64>().unwrap_or_else(|_| {
        err_exit(&format!("{}: illegal {} name", name, type_name));
    })
}

fn a_gid(s: &str) -> u64 {
    if s.is_empty() {
        err_exit("group must be specified when the owner is");
    }
    unsafe {
        let c_name = match CString::new(s) {
            Ok(n) => n,
            Err(_) => err_exit(&format!("invalid group name: {}", s)),
        };
        let result = libc::getgrnam(c_name.as_ptr());
        if !result.is_null() {
            (*result).gr_gid as u64
        } else {
            id(s, "group")
        }
    }
}

fn a_uid(s: &str) -> u32 {
    if s.is_empty() {
        err_exit("owner must be specified when the group is");
    }
    unsafe {
        let c_name = match CString::new(s) {
            Ok(n) => n,
            Err(_) => err_exit(&format!("invalid user name: {}", s)),
        };
        let result = libc::getpwnam(c_name.as_ptr());
        if !result.is_null() {
            (*result).pw_uid
        } else {
            id(s, "user") as u32
        }
    }
}
