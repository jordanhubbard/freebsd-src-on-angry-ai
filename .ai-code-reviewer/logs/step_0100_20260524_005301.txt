use std::cmp::Ordering;
use std::env;
use std::fs::File;
use std::io::{self, BufRead, BufReader, Write, BufWriter};
use std::process;

#[derive(Clone, Copy, PartialEq)]
enum DFlag {
    None,
    NoSep,
    PreSep,
    PostSep,
}

struct Options {
    dflag: DFlag,
    cflag: bool,
    dflag_dup: bool,
    uflag: bool,
    iflag: bool,
    numchars: usize,
    numfields: usize,
}

fn usage() -> ! {
    eprintln!(
        "usage: uniq [-cdiu] [-D[septype]] [-f fields] [-s chars] [input [output]]"
    );
    process::exit(1);
}

fn strtonum(s: &str, low: i64, high: i64) -> Result<i64, String> {
    match s.parse::<i64>() {
        Ok(n) if n >= low && n <= high => Ok(n),
        Ok(_) => Err(format!("out of range for {}..{}", low, high)),
        Err(_) => Err("invalid number".to_string()),
    }
}

/// Convert a line to a comparison key, applying skip and case folding.
fn convert(line: &str, opts: &Options) -> String {
    let mut s = line.trim_end_matches('\n').to_string();

    // Skip fields and characters
    if opts.numfields > 0 || opts.numchars > 0 {
        let mut chars = s.chars();
        let mut fields_skipped = 0;
        let mut in_field = false;

        while fields_skipped < opts.numfields {
            // Skip leading blanks
            while let Some(c) = chars.clone().next() {
                if c.is_whitespace() && (c == ' ' || c == '\t') {
                    chars.next();
                } else {
                    break;
                }
            }
            // Skip non-blanks
            in_field = false;
            while let Some(c) = chars.clone().next() {
                if c.is_whitespace() && (c == ' ' || c == '\t') {
                    break;
                }
                chars.next();
                in_field = true;
            }
            if !in_field {
                break;
            }
            fields_skipped += 1;
        }

        // Skip characters
        for _ in 0..opts.numchars {
            chars.next();
        }

        s = chars.collect();
    }

    // Case folding
    if opts.iflag {
        s = s.to_lowercase();
    }

    s
}

/// Compare two lines using the same logic as the C version.
/// When convert returns empty for both (non-UTF8 fallback), use byte comparison.
fn compare_lines(line1: &str, line2: &str, opts: &Options) -> Ordering {
    let key1 = convert(line1, opts);
    let key2 = convert(line2, opts);
    key1.cmp(&key2)
}

fn show(ofp: &mut BufWriter<File>, line: &str, repeats: i64, cflag: bool) {
    if cflag {
        write!(ofp, "{:4}{}", repeats + 1, line).unwrap();
    } else {
        write!(ofp, "{}", line).unwrap();
    }
}

/// Handle obsolete +N/-N syntax for -s/-f
fn obsolete_args(args: &[String]) -> Vec<String> {
    let mut result = Vec::new();
    for arg in args {
        if arg == "--" {
            result.push(arg.clone());
            continue;
        }
        if !arg.starts_with('-') && !arg.starts_with('+') {
            result.push(arg.clone());
            continue;
        }
        if arg.starts_with("--") {
            result.push(arg.clone());
            continue;
        }
        // Check for +N or -N (digit after sign)
        let rest = &arg[1..];
        if rest.chars().next().map_or(false, |c| c.is_ascii_digit()) {
            let new_opt = if arg.starts_with('+') {
                format!("-s{}", rest)
            } else {
                format!("-f{}", rest)
            };
            result.push(new_opt);
        } else {
            result.push(arg.clone());
        }
    }
    result
}
