fn expand_number_rust(s: &str) -> Option<u64> {
    let mut s = s;
    let mut val: u64 = 0;

    if s.is_empty() {
        return None;
    }

    // Parse numeric part
    let (num_str, suffix) = match s.find(|c: char| !c.is_ascii_digit()) {
        Some(pos) => (&s[..pos], &s[pos..]),
        None => (s, ""),
    };

    if num_str.is_empty() {
        return None;
    }

    val = match num_str.parse::<u64>() {
        Ok(v) => v,
        Err(_) => return None,
    };

    // Apply suffix multiplier
    match suffix {
        "" => {}
        "K" | "k" => {
            val = match val.checked_mul(1024) {
                Some(v) => v,
                None => return None,
            };
        }
        "M" | "m" => {
            val = match val.checked_mul(1024 * 1024) {
                Some(v) => v,
                None => return None,
            };
        }
        "G" | "g" => {
            val = match val.checked_mul(1024 * 1024 * 1024) {
                Some(v) => v,
                None => return None,
            };
        }
        "T" | "t" => {
            val = match val.checked_mul(1024 * 1024 * 1024 * 1024) {
                Some(v) => v,
                None => return None,
            };
        }
        _ => return None,
    }

    Some(val)
}
