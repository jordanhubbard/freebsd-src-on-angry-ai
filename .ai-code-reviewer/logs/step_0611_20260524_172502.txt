fn get_terminal_width() -> Option<usize> {
    #[cfg(unix)]
    {
        use libc::{ioctl, TIOCGWINSZ, winsize};
        let mut win: winsize = unsafe { std::mem::zeroed() };
        if unsafe { ioctl(1, TIOCGWINSZ, &mut win) } == 0 && win.ws_col > 0 {
            return Some(win.ws_col as usize);
        }
    }
    None
}

fn parse_positive_usize(s: &str, name: &str, flag: &str) -> usize {
    match s.parse::<usize>() {
        Ok(v) => v,
        Err(_) => {
            eprintln!("column: invalid {} \"{}\": value out of range", name, s);
            process::exit(1);
        }
    }
}

fn input<R: io::Read>(reader: &mut BufReader<R>, entries: &mut Vec<String>) -> io::Result<()> {
    let mut line = String::new();
    loop {
        line.clear();
        match reader.read_line(&mut line) {
            Ok(0) => break,
            Ok(_) => {}
            Err(e) => return Err(e),
        }
        // Strip trailing newline
        if line.ends_with('\n') {
            line.pop();
            if line.ends_with('\r') {
                line.pop();
            }
        }
        // Skip lines that are all whitespace
        if line.trim().is_empty() {
            continue;
        }
        entries.push(line);
    }
    Ok(())
}

fn width(s: &str) -> usize {
    s.chars().filter(|c| !c.is_control()).map(|c| char_width(*c)).sum()
}

fn char_width(c: char) -> usize {
    // Simple width calculation: CJK and other wide chars are 2, others 1
    // This approximates wcwidth behavior
    if c.is_ascii() {
        1
    } else if c.is_control() {
        0
    } else {
        // Check for wide characters (CJK, etc.)
        let cp = c as u32;
        if (0x1100..=0x115F).contains(&cp) || // Hangul Jamo
           (0x2329..=0x232A).contains(&cp) || // Wide special
           (0x2E80..=0xA4CF).contains(&cp) && cp != 0x303F || // CJK radicals, etc.
           (0xAC00..=0xD7A3).contains(&cp) || // Hangul Syllables
           (0xF900..=0xFAFF).contains(&cp) || // CJK Compatibility Ideographs
           (0xFE10..=0xFE19).contains(&cp) || // Vertical forms
           (0xFE30..=0xFE6F).contains(&cp) || // CJK Compatibility Forms
           (0xFF00..=0xFF60).contains(&cp) || // Fullwidth ASCII variants
           (0xFFE0..=0xFFE6).contains(&cp) || // Fullwidth symbols
           (0x20000..=0x2FFFD).contains(&cp) || // CJK extensions
           (0x30000..=0x3FFFD).contains(&cp) {
            2
        } else {
            1
        }
    }
}

fn roundup(v: usize, align: usize) -> usize {
    (v + align - 1) / align * align
}

fn c_columnate(entries: &[String], termwidth: usize, maxlength: usize) {
    let numcols = termwidth / maxlength;
    let out = io::stdout();
    let mut w = io::BufWriter::new(out.lock());

    let mut chcnt = 0;
    let mut col = 0;
    let mut endcol = maxlength;
    let mut first = true;

    for entry in entries {
        if !first {
            if col >= numcols {
                writeln!(w).unwrap();
                chcnt = 0;
                col = 0;
                endcol = maxlength;
            } else {
                // Print tabs to align
                while roundup(chcnt + 1, TAB) <= endcol {
                    write!(w, "\t").unwrap();
                    chcnt = roundup(chcnt + 1, TAB);
                }
                endcol += maxlength;
            }
        }
        first = false;
        write!(w, "{}", entry).unwrap();
        chcnt += width(entry);
        col += 1;
    }
    if !first {
        writeln!(w).unwrap();
    }
}

fn r_columnate(entries: &[String], termwidth: usize, maxlength: usize) {
    let numcols = termwidth / maxlength;
    let numrows = if numcols > 0 {
        (entries.len() + numcols - 1) / numcols
    } else {
        entries.len()
    };

    let out = io::stdout();
    let mut w = io::BufWriter::new(out.lock());

    for row in 0..numrows {
        let mut endcol = maxlength;
        let mut chcnt = 0;
        let mut col = 0;
        let mut base = row;

        while col < numcols && base < entries.len() {
            write!(w, "{}", entries[base]).unwrap();
            chcnt += width(&entries[base]);
            let next_base = base + numrows;
            if next_base >= entries.len() {
                break;
            }
            while roundup(chcnt + 1, TAB) <= endcol {
                write!(w, "\t").unwrap();
                chcnt = roundup(chcnt + 1, TAB);
            }
            endcol += maxlength;
            base = next_base;
            col += 1;
        }
        writeln!(w).unwrap();
    }
}

fn maketbl(entries: &[String], tblcols: usize, separator: &[char]) {
    if entries.is_empty() {
        return;
    }

    // Parse each entry into columns
    let mut parsed: Vec<Vec<String>> = Vec::new();
    let mut lens: Vec<usize> = Vec::new();

    for entry in entries {
        let mut cols: Vec<String> = Vec::new();
        let mut chars = entry.chars().peekable();
        let mut current = String::new();

        while let Some(&ch) = chars.peek() {
            if separator.contains(&ch) {
                chars.next();
                if !current.is_empty() {
                    cols.push(current.clone());
                    current.clear();
                }
            } else {
                current.push(chars.next().unwrap());
            }
        }
        if !current.is_empty() {
            cols.push(current);
        }

        // Limit columns if tblcols is set
        if tblcols > 0 && cols.len() > tblcols {
            // Rejoin excess columns
            let excess: String = cols[tblcols..].join("");
            cols.truncate(tblcols);
            if let Some(last) = cols.last_mut() {
                last.push_str(&excess);
            }
        }

        // Track max width per column
        for (i, col) in cols.iter().enumerate() {
            let w = width(col);
            while i >= lens.len() {
                lens.push(0);
            }
            if w > lens[i] {
                lens[i] = w;
            }
        }

        parsed.push(cols);
    }

    // Print table
    for row in &parsed {
        for (i, col) in row.iter().enumerate() {
            print!("{}", col);
            if i < row.len() - 1 {
                let pad = lens[i] - width(col) + 2;
                for _ in 0..pad {
                    print!(" ");
                }
            }
        }
        println!();
    }
}

fn usage() {
    eprintln!("usage: column [-tx] [-c columns] [-l tblcols] [-s sep] [file ...]");
    process::exit(1);
}
