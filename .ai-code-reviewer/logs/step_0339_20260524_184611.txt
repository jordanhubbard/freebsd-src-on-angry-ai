static int
contains(ccharp *cpp, const char *s)
{
	ccharp cp;

	if (cpp == NULL)
		return (0);

	while ((cp = *cpp) != NULL) {
		if (strcmp(cp, s) == 0)
			return (1);
		cpp++;
	}
	return (0);
}
