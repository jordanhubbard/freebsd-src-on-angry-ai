use std::os::raw::{c_char, c_int};
use std::process;
use std::time;

fn ctime_r(t: time_t) -> String {
    // Use std::time to format like ctime
    let duration = std::time::Duration::from_secs(t as u64);
    let sys_time = std::time::SystemTime::UNIX_EPOCH + duration;
    let datetime = chrono_like_format(&sys_time);
    datetime
}

fn chrono_like_format(sys_time: &std::time::SystemTime) -> String {
    // ctime format: "Wed Jun 30 21:49:08 1993\n"
    // We need to format manually since we can't use external crates
    use std::time::{SystemTime, UNIX_EPOCH};

    let secs = match sys_time.duration_since(UNIX_EPOCH) {
        Ok(d) => d.as_secs(),
        Err(_) => 0,
    };

    // Convert to UTC time components
    let days = secs / 86400;
    let tod = secs % 86400;
    let hour = tod / 3600;
    let min = (tod % 3600) / 60;
    let sec = tod % 60;

    // Calculate year, month, day from days since epoch
    let (year, month, day, wday) = days_to_ymd(days);

    let weekdays = ["Thu", "Fri", "Sat", "Sun", "Mon", "Tue", "Wed"];
    let months = [
        "Jan", "Feb", "Mar", "Apr", "May", "Jun",
        "Jul", "Aug", "Sep", "Oct", "Nov", "Dec",
    ];

    format!(
        "{} {:>3} {:>2} {:02}:{:02}:{:02} {}\n",
        weekdays[wday as usize],
        months[month],
        day,
        hour,
        min,
        sec,
        year
    )
}

fn days_to_ymd(days_since_epoch: u64) -> (i32, usize, u32, u32) {
    // Algorithm to convert days since Unix epoch to year/month/day
    // Unix epoch is 1970-01-01 (Thursday)
    let mut days = days_since_epoch as i64;
    let mut year: i64 = 1970;

    loop {
        let days_in_year = if is_leap(year) { 366 } else { 365 };
        if days < days_in_year {
            break;
        }
        days -= days_in_year;
        year += 1;
    }

    let month_days: [i64; 12] = if is_leap(year) {
        [31, 29, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31]
    } else {
        [31, 28, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31]
    };

    let mut month: usize = 0;
    let mut day: i64 = days;
    for i in 0..12 {
        if day < month_days[i] {
            month = i;
            break;
        }
        day -= month_days[i];
        month = i + 1;
    }

    // Day of week: 0=Sun, 1=Mon, ..., 6=Sat
    // Unix epoch (1970-01-01) was Thursday = 4
    let dow = (days_since_epoch + 4) % 7;

    (year as i32, month, (day + 1) as u32, dow)
}

fn is_leap(year: i64) -> bool {
    (year % 4 == 0 && year % 100 != 0) || (year % 400 == 0)
}

fn output(u: &utmpx) {
    let user = u.user().to_string_lossy();
    let line = u.line().to_string_lossy();
    let host = u.host().to_string_lossy();
    let t = u.tv_sec();
    let login_time = ctime_r(t);

    // Format: user (10 chars left), tty (8 chars left), from (22 chars left), login-time
    // The ctime output already has a trailing newline, so we strip it
    let login_time_str = login_time.trim_end();

    println!(
        "{:<10} {:<8} {:<22} {}",
        user, line, host, login_time_str
    );
}

            });
        }

        for entry in &entries {
            output(entry);
        }
    }

    process::exit(0);
}
