use std::env;
use std::ffi::CString;
use std::io::{self, Write};
use std::process;

extern "C" {
    fn dirname(path: *const i8) -> *mut i8;
}

fn usage() -> ! {
    let stderr = io::stderr();
    let mut handle = stderr.lock();
    let _ = handle.write_all(b"usage: dirname string [...]\n");
    process::exit(1);
}

fn main() {
    let args: Vec<String> = env::args().skip(1).collect();

    if args.is_empty() {
        usage();
    }

    for arg in &args {
        let c_arg = match CString::new(arg.as_bytes()) {
            Ok(s) => s,
            Err(_) => {
                eprintln!("{}: invalid argument", arg);
                process::exit(1);
            }
        };
        let result = unsafe { dirname(c_arg.as_ptr()) };
        if result.is_null() {
            eprintln!("{}: dirname failed", arg);
            process::exit(1);
        }
        let dirname_str = unsafe { std::ffi::CStr::from_ptr(result) };
        println!("{}", dirname_str.to_string_lossy());
    }
}
