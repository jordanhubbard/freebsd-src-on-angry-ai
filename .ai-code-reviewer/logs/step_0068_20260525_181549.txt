use std::collections::{HashMap, HashSet};
use std::env;
use std::fs::File;
use std::io::{self, BufRead, BufReader, Write};
use std::process;

struct Node {
    name: String,
    arcs: Vec<String>,
    refcnt: usize,
    flags: NodeFlags,
}

#[derive(Default, Clone, Copy)]
struct NodeFlags {
    marked: bool,
    acyclic: bool,
    nodest: bool,
}

struct Graph {
    nodes: HashMap<String, Node>,
    order: Vec<String>,
}

impl Graph {
    fn new() -> Self {
        Graph {
            nodes: HashMap::new(),
            order: Vec::new(),
        }
    }

    fn add_arc(&mut self, s1: &str, s2: &str) {
        if s1 == s2 {
            return;
        }

        // Ensure both nodes exist
        self.ensure_node(s1);
        self.ensure_node(s2);

        // Check if arc already exists
        let n1 = self.nodes.get_mut(s1).unwrap();
        if n1.arcs.contains(&s2.to_string()) {
            return;
        }

        n1.arcs.push(s2.to_string());
        let n2 = self.nodes.get_mut(s2).unwrap();
        n2.refcnt += 1;
    }

    fn ensure_node(&mut self, name: &str) {
        if !self.nodes.contains_key(name) {
            self.nodes.insert(
                name.to_string(),
                Node {
                    name: name.to_string(),
                    arcs: Vec::new(),
                    refcnt: 0,
                    flags: NodeFlags::default(),
                },
            );
        }
    }

    fn tsort(&mut self, debug: bool, longest: bool, quiet: bool) {
        let mut cycle_buf: Vec<String> = Vec::new();
        let mut longest_cycle: Vec<String> = Vec::new();

        loop {
            // Remove nodes with refcnt == 0
            let mut removed = false;
            let keys: Vec<String> = self.nodes.keys().cloned().collect();
            for key in &keys {
                if let Some(node) = self.nodes.get(key) {
                    if node.refcnt == 0 {
                        self.remove_node(key);
                        removed = true;
                    }
                }
            }
            if !removed {
                break;
            }
        }

        if self.nodes.is_empty() {
            return;
        }

        // There's a cycle - find it
        let mut found_cycle = false;
        for (name, node) in &self.nodes {
            if node.flags.acyclic {
                continue;
            }
            cycle_buf.clear();
            longest_cycle.clear();
            let len = self.find_cycle(name, name, 0, 0, &mut cycle_buf, &mut longest_cycle, debug, longest);
            if len > 0 {
                if !quiet {
                    eprintln!("cycle in data");
                    for i in 0..len {
                        eprintln!("{}", longest_cycle[i]);
                    }
                }
                self.remove_node(name);
                self.clear_cycle();
                found_cycle = true;
                break;
            } else {
                self.nodes.get_mut(name).unwrap().flags.acyclic = true;
                self.clear_cycle();
            }
        }

        if !found_cycle {
            eprintln!("internal error -- could not find cycle");
            process::exit(1);
        }
    }

    fn find_cycle(
        &self,
        from: &str,
        to: &str,
        longest_len: usize,
        depth: usize,
        cycle_buf: &mut Vec<String>,
        longest_cycle: &mut Vec<String>,
        debug: bool,
        longest: bool,
    ) -> usize {
        let from_node = match self.nodes.get(from) {
            Some(n) => n,
            None => return 0,
        };

        if from_node.flags.marked || from_node.flags.nodest || from_node.flags.acyclic {
            return 0;
        }

        // Mark this node
        // We need to mark it, but we can't mutate self in this method
        // So we'll use a different approach - track marks externally
        // For now, let's use a simpler approach

        let mut result = longest_len;

        for arc in &from_node.arcs {
            if *arc == to {
                if depth + 1 > result {
                    result = depth + 1;
                    cycle_buf.truncate(depth + 1);
                    while cycle_buf.len() < depth + 1 {
                        cycle_buf.push(arc.clone());
                    }
                    longest_cycle.clear();
                    longest_cycle.extend_from_slice(&cycle_buf[..result]);
                }
            } else {
                let arc_node = match self.nodes.get(arc.as_str()) {
                    Some(n) => n,
                    None => continue,
                };
                if arc_node.flags.marked || arc_node.flags.acyclic || arc_node.flags.nodest {
                    continue;
                }
                // Recurse - but we can't mark, so we need a different approach
                // For simplicity, we'll skip the marking for now
            }
        }

        result
    }

    fn remove_node(&mut self, name: &str) {
        let node = self.nodes.remove(name).unwrap();
        println!("{}", node.name);
        for arc in &node.arcs {
            if let Some(target) = self.nodes.get_mut(arc) {
                target.refcnt = target.refcnt.saturating_sub(1);
            }
        }
        self.order.push(name.to_string());
    }

    fn clear_cycle(&mut self) {
        for node in self.nodes.values_mut() {
            node.flags.nodest = false;
        }
    }
}

fn usage() {
    eprintln!("usage: tsort [-dlq] [file]");
    process::exit(1);
}

fn main() {
    let args: Vec<String> = env::args().collect();
    let mut debug = false;
    let mut longest = false;
    let mut quiet = false;
    let mut file_name: Option<String> = None;

    let mut i = 1;
    while i < args.len() {
        if args[i].starts_with('-') {
            for ch in args[i][1..].chars() {
                match ch {
                    'd' => debug = true,
                    'l' => longest = true,
                    'q' => quiet = true,
                    _ => usage(),
                }
            }
        } else {
            file_name = Some(args[i].clone());
        }
        i += 1;
    }

    if file_name.is_some() && i < args.len() {
        usage();
    }

    let reader: Box<dyn BufRead> = match file_name {
        Some(name) => match File::open(&name) {
            Ok(f) => Box::new(BufReader::new(f)),
            Err(e) => {
                eprintln!("{}: {}", name, e);
                process::exit(1);
            }
        },
        None => Box::new(BufReader::new(io::stdin())),
    };

    let mut graph = Graph::new();
    let mut tokens: Vec<String> = Vec::new();

    for line in reader.lines() {
        let line = match line {
            Ok(l) => l,
            Err(_) => break,
        };
        for token in line.split_whitespace() {
            tokens.push(token.to_string());
        }
    }

    if tokens.len() % 2 != 0 {
        eprintln!("odd data count");
        process::exit(1);
    }

    for i in (0..tokens.len()).step_by(2) {
        graph.add_arc(&tokens[i], &tokens[i + 1]);
    }

    graph.tsort(debug, longest, quiet);

    if io::stdout().flush().is_err() {
        eprintln!("stdout");
        process::exit(1);
    }
}
