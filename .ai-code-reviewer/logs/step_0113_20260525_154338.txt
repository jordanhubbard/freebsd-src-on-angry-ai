fn usage() -> ! {
    let msg = concat!(
        "usage: truncate [-c] -s [+|-|%|/]size[K|k|M|m|G|g|T|t] file ...\n",
        "       truncate [-c] -r rfile file ...\n",

    process::exit(EXIT_FAILURE);
}

fn expand_number_rust(s: &str) -> Option<u64> {
    let mut s = s;
    let mut val: u64 = 0;

    if s.is_empty() {
        return None;
    }

    // Parse numeric part
    let (num_str, suffix) = match s.find(|c: char| !c.is_ascii_digit()) {
        Some(pos) => (&s[..pos], &s[pos..]),
        None => (s, ""),
    };

    if num_str.is_empty() {
        return None;
    }

    val = match num_str.parse::<u64>() {
        Ok(v) => v,
        Err(_) => return None,
    };

    // Apply suffix multiplier
    match suffix {
        "" => {}
        "K" | "k" => {
            val = match val.checked_mul(1024) {
                Some(v) => v,
                None => return None,
            };
        }
        "M" | "m" => {
            val = match val.checked_mul(1024 * 1024) {
                Some(v) => v,
                None => return None,
            };
        }
        "G" | "g" => {
            val = match val.checked_mul(1024 * 1024 * 1024) {
                Some(v) => v,
                None => return None,
            };
        }
        "T" | "t" => {
            val = match val.checked_mul(1024 * 1024 * 1024 * 1024) {
                Some(v) => v,
                None => return None,
            };
        }
        _ => return None,
    }

    Some(val)
}

fn open_file(path: &str, flags: c_int, mode: mode_t) -> io::Result<std::fs::File> {
    let cpath = CString::new(path).map_err(|_| io::Error::new(io::ErrorKind::InvalidInput, "invalid path"))?;
    let fd = unsafe { libc_open(cpath.as_ptr(), flags, mode as u32) };
    if fd < 0 {
        return Err(io::Error::last_os_error());
    }
    Ok(unsafe { std::fs::File::from_raw_fd(fd) })
}

extern "C" {
    fn open(path: *const i8, oflag: c_int, ...) -> c_int;
}

fn libc_open(path: *const i8, oflag: c_int, mode: u32) -> c_int {
    unsafe { open(path, oflag, mode) }
}

fn ftruncate_rust(file: &std::fs::File, size: off_t) -> io::Result<()> {
    let fd = file.as_raw_fd();
    let r = unsafe { libc_ftruncate(fd, size) };
    if r == -1 {
        Err(io::Error::last_os_error())
    } else {
        Ok(())
    }
}
