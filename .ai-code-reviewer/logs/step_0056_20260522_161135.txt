fn escape(fmt: &str, percent: bool) -> (String, bool) {
    let mut result = String::new();
    let mut chars = fmt.chars().peekable();
    while let Some(c) = chars.next() {
        if c != '\\' {
            result.push(c);
            continue;
        }
        match chars.next() {
            None => { result.push('\\'); return (result, false); }
            Some('\\') => result.push('\\'),
            Some('\'') => result.push('\''),
            Some('a') => result.push('\x07'),
            Some('b') => result.push('\x08'),
            Some('c') => { if !percent { return (result, true); } result.push('c'); }
            Some('f') => result.push('\x0c'),
            Some('n') => result.push('\n'),
            Some('r') => result.push('\r'),
            Some('t') => result.push('\t'),
            Some('v') => result.push('\x0b'),
            Some(d) if d.is_ascii_digit() => {
                let mut value: u32 = d.to_digit(8).unwrap();
                let max_d = if !percent && d == '0' { 4 } else { 3 };
                let mut consumed = 1;
                while consumed < max_d {
                    match chars.peek() {
                        Some(&n) if n.is_ascii_digit() => {
                            value = value * 8 + n.to_digit(8).unwrap();
                            chars.next(); consumed += 1;
                        }
                        _ => break,
                    }
                }
                if percent && value == b'%' as u32 {
                    result.push('%'); result.push('%');
                } else {
                    result.push(char::from_u32(value).unwrap_or(d));
                }
            }
            Some(other) => result.push(other),
        }
    }
    (result, false)
}

fn fmt_str(s: &str, width: Option<i32>, precision: Option<i32>) -> String {
    let s = match precision {
        Some(p) if p >= 0 => s.chars().take(p as usize).collect::<String>(),
        _ => s.to_string(),
    };
    match (width, precision) {
        (Some(w), _) if w > 0 => format!("{:>width$}", s, width = w as usize),
        _ => s,
    }
}
