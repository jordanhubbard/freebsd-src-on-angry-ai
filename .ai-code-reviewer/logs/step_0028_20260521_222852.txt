fn parse_duration(s: &str) -> f64 {
    let mut chars = s.chars();
    let suffix = chars.next_back();
    let num_str: String = chars.collect();
    let ret: f64 = match num_str.parse() {
        Ok(n) => n,
        Err(_) => { eprintln!("timeout: duration is not a number"); process::exit(EXIT_INVALID); }
    };
    if ret.is_nan() { eprintln!("timeout: duration is not a number"); process::exit(EXIT_INVALID); }
    let ret = match suffix {
        None | Some('s') => ret, Some('m') => ret * 60.0,
        Some('h') => ret * 3600.0, Some('d') => ret * 86400.0,
        Some(_) => { eprintln!("timeout: duration unit suffix invalid"); process::exit(EXIT_INVALID); }
    };
    if ret < 0.0 || ret >= 100_000_000.0 {
        eprintln!("timeout: duration out of range"); process::exit(EXIT_INVALID);
    }
    ret
}

fn kill_self(signo: i32) -> ! {

    unsafe {
