fn read_char<R: BufRead>(reader: &mut R) -> io::Result<Option<char>> {
    let mut buf = [0u8; 4];
    let mut total = 0u8;
    loop {
        match reader.read(&mut buf[total as usize..]) {
            Ok(0) => {
                if total == 0 {
                    return Ok(None);
                }
                break;
            }
            Ok(n) => {
                total += n as u8;
                if total >= 1 {
                    break;
                }
            }
            Err(e) if e.kind() == io::ErrorKind::Interrupted => continue,
            Err(e) => return Err(e),
        }
    }
    let s = std::str::from_utf8(&buf[..total as usize])
        .unwrap_or("?");
    Ok(s.chars().next())
}

fn write_str<W: Write>(out: &mut W, chars: &[char]) -> io::Result<()> {
    for &c in chars {
        out.write_all(c.encode_utf8(&mut [0; 4]).as_bytes())?;
    }
    Ok(())
}

fn find_last_blank(chars: &[char]) -> Option<usize> {
    for i in (0..chars.len()).rev() {
        if chars[i].is_whitespace() {

    }

}

}
