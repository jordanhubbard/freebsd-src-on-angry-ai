    }

    if let Some(ref sh) = shell {
        if let Some(name) = sh.rsplit('/').next() {
            return getshellbyname(name);
        }
    }
    // Fallback: try sysctl for parent process name
    let mut mib: [libc::c_int; 4] = [
        libc::CTL_KERN, libc::KERN_PROC, libc::KERN_PROC_PID, ppid as libc::c_int,
    ];

    let mut len: size_t = std::mem::size_of::<libc::kinfo_proc>();
    if unsafe { libc::sysctl(mib.as_mut_ptr(), 4, &mut kp as *mut _ as *mut _, &mut len, std::ptr::null(), 0) } != -1 {

                .into_owned()
        };

    1 // SH_SH
}
