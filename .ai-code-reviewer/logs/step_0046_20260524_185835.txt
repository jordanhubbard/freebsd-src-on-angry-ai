fn expand_number(s: &str) -> Option<i64> {
    let s = s.trim();
    let mut mult: i64 = 1;
    let (num, suf) = match s.find(|c: char| c.is_ascii_alphabetic()) {
        Some(p) => (&s[..p], &s[p..]),
        None => (s, ""),
    };
    match suf {
        "" | "b" => mult = 1,
        "k" => mult = 1024,
        "m" => mult = 1024 * 1024,
        "g" => mult = 1024 * 1024 * 1024,
        "t" => mult = 1024 * 1024 * 1024 * 1024,
        _ => return None,
    }
    num.parse::<i64>().ok().map(|n| n * mult)
}

trait TapeOps {
    fn read_blk(&mut self, dst: &mut [u8]) -> i64;
    fn verify_blk(&mut self, dst: &mut [u8], expected: usize) -> i64;
    fn write_blk(&mut self, src: &[u8]);
    fn file_mark(&mut self);
    fn rewind(&mut self);
}

impl TapeOps for TapeDev {
    fn read_blk(&mut self, dst: &mut [u8]) -> i64 {
        let len = dst.len();
        let mut mrs = if self.max_read_size == 0 { len } else { self.max_read_size };
        let mut retval: i64 = -1;
        if self.max_read_size == 0 {
            while mrs > 0 {
                retval = unsafe {
                    libc::syscall(libc::SYS_read, self.fd,
                        dst[..mrs].as_mut_ptr() as *mut libc::c_void, mrs)
                };
                if retval >= 0 || unsafe { __errno() != libc::EINVAL && __errno() != libc::EFBIG } {
                    self.max_read_size = mrs;
                    break;
                }
                if mrs < 512 { eprintln!("tcopy: Cannot find a sane max blocksize"); process::exit(1); }
                let bits = (mrs as u64).next_power_of_two().trailing_zeros();
                mrs = 1usize << (bits.saturating_sub(1));
            }
        } else {
            retval = unsafe {
                libc::syscall(libc::SYS_read, self.fd,
                    dst[..self.max_read_size].as_mut_ptr() as *mut libc::c_void, self.max_read_size)
            };
        }
        if retval < 0 {
            eprintln!("tcopy: read error, {}, file {}, record {}", self.name, G_FILN, G_RECORD);
            process::exit(1);
        }
        retval
    }
    fn verify_blk(&mut self, dst: &mut [u8], _exp: usize) -> i64 { self.read_blk(dst) }
    fn write_blk(&mut self, src: &[u8]) {
        let n = unsafe {
            libc::syscall(libc::SYS_write, self.fd,
                src.as_ptr() as *const libc::c_void, src.len())
        };
        if n < 0 || (n as usize) != src.len() {
            eprintln!("tcopy: write error, file {}, record {}", G_FILN, G_RECORD);
            process::exit(5);
        }
    }
    fn file_mark(&mut self) {
        let mut op: libc::mtio_mtop = unsafe { std::mem::zeroed() };
        op.mt_op = libc::MTWEOF;
        op.mt_count = 1;
        if unsafe { libc::ioctl(self.fd, libc::MTIOCTOP, &mut op) } < 0 {
            eprintln!("tcopy: tape op (write file mark)");
            process::exit(6);
        }
    }
    fn rewind(&mut self) {
        let mut op: libc::mtio_mtop = unsafe { std::mem::zeroed() };
        op.mt_op = libc::MTREW;
        op.mt_count = 1;
        if unsafe { libc::ioctl(self.fd, libc::MTIOCTOP, &mut op) } < 0 {
            eprintln!("tcopy: tape op (rewind)");
            process::exit(6);
        }
    }
}
