/// Set odd parity on each byte of the key.
///
/// This is a pure Rust implementation of DES_set_odd_parity.
/// Each byte of the key is modified so that it has odd parity
/// (the number of 1-bits is odd).
#[no_mangle]
pub extern "C" fn k5_des_fixup_key_parity(keybits: *mut u8) {
    if keybits.is_null() {
        return;
    }
    // DES key is 8 bytes (DES_cblock)
    for i in 0..8 {
        let byte = unsafe { *keybits.add(i) };
        let mut b = byte;
        // Count bits and set parity bit (bit 0) to make odd parity
        let mut count = 0u32;
        let mut tmp = b;
        while tmp != 0 {
            if tmp & 1 != 0 {
                count += 1;
            }
            tmp >>= 1;
        }
        if count % 2 == 0 {
            b |= 1; // Set bit 0 to make odd parity
        } else {
            b &= !1; // Clear bit 0 to keep odd parity
        }
        unsafe {
            *keybits.add(i) = b;
        }
    }
}
