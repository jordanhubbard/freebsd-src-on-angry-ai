fn isleap(y: i32) -> bool {
    (y % 4 == 0 && y % 100 != 0) || y % 400 == 0
}

fn adj360(deg: &mut f64) {
    loop {
        if *deg < 0.0 {
            *deg += 360.0;
        } else if *deg > 360.0 {
            *deg -= 360.0;
        } else {
            break;
        }
    }
}

fn dtor(deg: f64) -> f64 {
    deg * PI / 180.0
}

fn potm(days: f64) -> f64 {
    let mut n = 360.0 * days / 365.2422;
    adj360(&mut n);
    let mut msol = n + EPSILONg - RHOg;
    adj360(&mut msol);

    adj360(&mut l);
