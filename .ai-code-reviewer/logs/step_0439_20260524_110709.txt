pub unsafe extern "C" fn expand_number(buf: *const c_char, num: *mut c_longlong) -> c_int {
    let mut number: u64 = 0;
    let mut neg = false;

    if expand_impl(buf, &mut number, &mut neg) != 0 {
        return -1;
    }

    if neg {
        if number > 0x8000000000000000u64 {
            set_errno(ERANGE);
            return -1;
        }
        *num = -(number as i64);
    } else {
        if number > INT64_MAX as u64 {
            set_errno(ERANGE);
            return -1;
        }
        *num = number as i64;
    }

    0
}

pub unsafe extern "C" fn expand_unsigned(buf: *const c_char, num: *mut c_ulonglong) -> c_int {

    let mut neg = false;

    if expand_impl(buf, &mut number, &mut neg) != 0 {
        return -1;
    }

        return -1;
    }

    0
