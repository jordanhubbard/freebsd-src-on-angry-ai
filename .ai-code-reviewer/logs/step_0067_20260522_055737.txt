use std::env;
use std::ffi::CString;
use std::os::raw::c_char;
use std::process;

const LIBSTDBUF: &str = "/usr/lib/libstdbuf.so";
const LIBSTDBUF32: &str = "/usr/lib32/libstdbuf.so";

extern "C" {
    fn execvp(file: *const c_char, argv: *const *const c_char) -> c_char;
}

fn usage() -> ! {
    eprintln!(
        "usage: stdbuf [-e 0|L|B|<sz>] [-i 0|L|B|<sz>] [-o 0|L|B|<sz>] \
         <cmd> [args ...]"
    );
    process::exit(1);
}

fn appendenv(key: &str, value: &str) {
    let curval = env::var(key).unwrap_or_default();
    let newval = if curval.is_empty() {
        format!("{value}")
    } else {
        format!("{curval}:{value}")
    };
    if env::var(key).is_ok() {
        env::set_var(key, &newval);
    } else {
        env::set_var(key, &newval);
    }
}

fn main() {
    let args: Vec<String> = env::args().collect();

    let mut ibuf: Option<&str> = None;
    let mut obuf: Option<&str> = None;
    let mut ebuf: Option<&str> = None;

    let mut i = 1;
    while i < args.len() {
        match args[i].as_str() {
            "-e" => {
                i += 1;
                if i >= args.len() {
                    usage();
                }
                ebuf = Some(&args[i]);
            }
            "-i" => {
                i += 1;
                if i >= args.len() {
                    usage();
                }
                ibuf = Some(&args[i]);
            }
            "-o" => {
                i += 1;
                if i >= args.len() {
                    usage();
                }
                obuf = Some(&args[i]);
            }
            _ => break,
        }
        i += 1;
    }

    let cmd_args: Vec<String> = args[i..].to_vec();
    if cmd_args.is_empty() {
        process::exit(0);
    }

    if let Some(val) = ibuf {
        env::set_var("_STDBUF_I", val);
    }
    if let Some(val) = obuf {
        env::set_var("_STDBUF_O", val);
    }
    if let Some(val) = ebuf {
        env::set_var("_STDBUF_E", val);
    }

    appendenv("LD_PRELOAD", LIBSTDBUF);
    appendenv("LD_32_PRELOAD", LIBSTDBUF32);

    let cmd_cstr = match CString::new(cmd_args[0].as_bytes()) {
        Ok(c) => c,
        Err(_) => {
            eprintln!("stdbuf: invalid command name");
            process::exit(1);
        }
    };

    let argv: Vec<*const c_char> = cmd_args

    argv_ptrs.push(std::ptr::null());

    unsafe {
        execvp(cmd_cstr.as_ptr(), argv_ptrs.as_ptr());
    }

    eprintln!("stdbuf: {}: {}", cmd_args[0], std::io::Error::last_os_error());
    process::exit(2);
}
