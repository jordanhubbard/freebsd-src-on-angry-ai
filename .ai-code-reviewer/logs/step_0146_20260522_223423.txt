use std::env;
use std::fs::{self, File, OpenOptions};
use std::io;
use std::os::unix::fs::OpenOptionsExt;
use std::path::Path;
use std::process;

fn expand_number(s: &str) -> Option<u64> {
    let mut chars = s.chars().peekable();
    let mut num_str = String::new();
    while let Some(&c) = chars.peek() {
        if c.is_ascii_digit() {
            num_str.push(c);
            chars.next();
        } else {
            break;
        }
    }
    let suffix: String = chars.collect();
    let multiplier: u64 = match suffix.as_str() {
        "" => 1,
        "K" | "k" => 1024,
        "M" | "m" => 1024 * 1024,
        "G" | "g" => 1024 * 1024 * 1024,
        "T" | "t" => 1024 * 1024 * 1024 * 1024,
        _ => return None,
    };
    let n: u64 = num_str.parse().ok()?;
    n.checked_mul(multiplier)
}

fn usage() -> ! {
    eprintln!(
        "{}\n{}\n{}",
        "usage: truncate [-c] -s [+|-|%|/]size[K|k|M|m|G|g|T|t] file ...",
        "       truncate [-c] -r rfile file ...",
        "       truncate [-c] -d [-o offset[K|k|M|m|G|g|T|t]] -l length[K|k|M|m|G|g|T|t] file ..."
    );
    process::exit(1);
}

fn main() {

    let mut do_dealloc = false;
    let mut do_refer = false;
    let mut do_relative = false;
    let mut do_round = false;
    let mut got_size = false;

    let mut off: u64 = 0;
    let mut len: u64 = 0;

        match arg.as_str() {
            "-c" => {
                no_create = true;
                optind += 1;

            "-d" => {
                do_dealloc = true;
                do_refer = false;
                got_size = false;
                optind += 1;

            "-r" => {

                    eprintln!("{}: option requires an argument -- 'r'", prog);
                    process::exit(1);

                do_dealloc = false;
                got_size = false;
                rname = Some(args[optind + 1].clone());
                optind += 2;

            "-s" => {
                if optind + 1 >= args.len() {
                    eprintln!("{}: option requires an argument -- 's'", prog);
                    process::exit(1);
                }
                let sarg = &args[optind + 1];
                if sarg.starts_with('+') || sarg.starts_with('-') {
                    do_relative = true;
                } else if sarg.starts_with('%') || sarg.starts_with('/') {
                    do_round = true;

                    &sarg[1..]
                } else {
                    sarg.as_str()

                    Some(usz) => {
                        if sarg.starts_with('-') || sarg.starts_with('/') {
                            sz = -(usz as i64);
                        } else {
                            sz = usz as i64;

                        got_size = true;
                        do_refer = false;
                        do_dealloc = false;
                    }
                    None => {
                        eprintln!("{}: invalid size argument '{}'", prog, sarg);
                        process::exit(1);
                    }
                }
                optind += 2;
            }
            "-o" => {
                if optind + 1 >= args.len() {
                    eprintln!("{}: option requires an argument -- 'o'", prog);
                    process::exit(1);

                    None => {

                            if round != abs_sz {
                                round -= 1;
                            }
                        } else {
                            round += 1;
                        }
                        current_tsize = if round < 0 { 0 } else { round } * abs_sz;
                    } else {
                        current_tsize = cur_size;
                    }
                }
                Err(e) => {
                    eprintln!("{}: {}: {}", prog, fname, e);
                    error = true;
                    continue;
                }
            }
        }

            current_tsize = 0;
        }

                    r_offset: off,
                    r_len: len,

                    eprintln!(
                        "{}: {}: {}",
                        prog,
                        fname,
                        io::Error::last_os_error()
                    );
                    error = true;
                }
            }
        } else {
            if let Err(e) = file.set_len(current_tsize as u64) {
                eprintln!("{}: {}: {}", prog, fname, e);
                error = true;
            }
        }
    }

    if error {
        process::exit(1);
    }
}
