use std::fs::{self, File, OpenOptions};
use std::io::{self, Write};
use std::os::unix::fs::OpenOptionsExt;

// FreeBSD libutil expand_number equivalent
fn expand_number(s: &str, result: &mut u64) -> bool {
    let mut chars = s.chars().peekable();
    let mut num_str = String::new();
    let mut suffix = String::new();

    while let Some(&c) = chars.peek() {
        if c.is_ascii_digit() {
            num_str.push(c);
            chars.next();
        } else {
            break;
        }
    }

    suffix = chars.collect();

    let multiplier: u64 = match suffix.as_str() {
        "" => 1,
        "K" | "k" => 1024,
        "M" | "m" => 1024 * 1024,
        "G" | "g" => 1024 * 1024 * 1024,
        "T" | "t" => 1024 * 1024 * 1024 * 1024,
        _ => return false,
    };

    let n: u64 = match num_str.parse() {
        Ok(v) => v,
        Err(_) => return false,
    };

    match n.checked_mul(multiplier) {
        Some(v) => {
            *result = v;
            true
        }
        None => false,
    }
}

    eprintln!(

}
