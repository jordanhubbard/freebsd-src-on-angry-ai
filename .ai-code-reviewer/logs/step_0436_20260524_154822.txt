// Rust replacement for LLVM Telemetry library
// Provides basic telemetry framework functionality

use std::collections::VecDeque;
use std::sync::atomic::{AtomicU64, Ordering};

static SESSION_COUNTER: AtomicU64 = AtomicU64::new(0);

pub struct TelemetryInfo {
    pub session_id: u64,
}

impl TelemetryInfo {
    pub fn new() -> Self {
        TelemetryInfo {
            session_id: SESSION_COUNTER.fetch_add(1, Ordering::Relaxed),
        }
    }

    pub fn serialize(&self) -> String {
        format!("SessionId: {}", self.session_id)
    }
}

pub trait Destination {
    fn receive_entry(&self, entry: &TelemetryInfo) -> Result<(), String>;
}

pub struct Manager {
    destinations: VecDeque<Box<dyn Destination>>,
}

impl Manager {
    pub fn new() -> Self {
        Manager {
            destinations: VecDeque::new(),
        }
    }

    pub fn add_destination(&mut self, dest: Box<dyn Destination>) {
        self.destinations.push_back(dest);
    }

    pub fn dispatch(&self, entry: &TelemetryInfo) -> Result<(), String> {
        let mut all_errs = Ok(());
        for dest in &self.destinations {
            if let Err(e) = dest.receive_entry(entry) {
                all_errs = Err(e);
            }
        }
        all_errs
    }
}
