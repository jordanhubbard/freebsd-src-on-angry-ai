fn atoi2(arg: &str) -> (i32, &str) {
    let bytes = arg.as_bytes();
    let val = (bytes[0] - b'0') as i32 * 10 + (bytes[1] - b'0') as i32;
    (val, &arg[2..])
}

fn timeoffset(arg: &str) -> i64 {
    let mut offset: i64 = 0;
    let mut isneg = false;
    let mut s = arg;
    if s.starts_with('-') {
        isneg = true;
        s = &s[1..];
    }
    match s.len() {
        6 => {
            let (v, rest) = atoi2(s);
            offset = v as i64;
            s = rest;
        }
        _ => {}
    }
    if s.len() >= 4 {
        let (v, rest) = atoi2(s);
        offset = offset * 60 + v as i64;
        s = rest;
    }
    if s.len() >= 2 {
        let (v, _) = atoi2(s);
        offset = offset * 60 + v as i64;
    }
    if isneg {
        -offset
    } else {
        offset
    }
}

fn stime_darg(arg: &str, tvp: &mut [libc::timespec; 2]) {
    // Check for at least two colons
    let colon_count = arg.matches(':').count();
    if colon_count < 2 {
        err_exit(1, "out of range or illegal time specification: YYYY-MM-DDThh:mm:SS[.frac][tz]");
    }

    let has_t = arg.contains('T');
    let fmt = if has_t {
        "%Y-%m-%dT%H:%M:%S"
    } else {
        "%Y-%m-%d %H:%M:%S"
    };

    let mut t = unsafe { std::mem::zeroed::<libc::tm>() };
    t.tm_sec = 0;
    t.tm_isdst = -1;

    let c_arg = match CString::new(arg) {
        Ok(s) => s,
        Err(_) => err_exit(1, "out of range or illegal time specification: YYYY-MM-DDThh:mm:SS[.frac][tz]"),
    };
    let c_fmt = match CString::new(fmt) {
        Ok(s) => s,
        Err(_) => err_exit(1, "out of range or illegal time specification: YYYY-MM-DDThh:mm:SS[.frac][tz]"),
    };

    let result = unsafe { libc::strptime(c_arg.as_ptr(), c_fmt.as_ptr(), &mut t) };
    if result.is_null() {
        err_exit(1, "out of range or illegal time specification: YYYY-MM-DDThh:mm:SS[.frac][tz]");
    }

    let remaining = unsafe { std::ffi::CStr::from_ptr(result).to_string_lossy() };

    let mut frac_pos = 0;
    let mut is_utc = false;

    let bytes = remaining.as_bytes();
    if !bytes.is_empty() && (bytes[0] == b'.' || bytes[0] == b',') {
        if bytes.len() > 1 && bytes[1].is_ascii_digit() {
            let mut val: i64 = 100000000;
            let mut i = 1;
            while i < bytes.len() && bytes[i].is_ascii_digit() {
                tvp[0].tv_nsec += val * (bytes[i] - b'0') as i64;
                i += 1;
                val /= 10;
            }
            frac_pos = i;
        }
    }

    if frac_pos < remaining.len() && remaining.as_bytes()[frac_pos] == b'Z' {
        is_utc = true;
        frac_pos += 1;
    }

    if frac_pos < remaining.len() {
        err_exit(1, "out of range or illegal time specification: YYYY-MM-DDThh:mm:SS[.frac][tz]");
    }

    t.tm_yday = -1;
    let sec = if is_utc {
        unsafe { libc::timegm(&mut t) }
    } else {
        unsafe { libc::mktime(&mut t) }
    };
    if t.tm_yday == -1 {
        err_exit(1, "out of range or illegal time specification: YYYY-MM-DDThh:mm:SS[.frac][tz]");
    }

    tvp[0].tv_sec = sec;
    tvp[1] = tvp[0];
}

fn stime_file(fname: &str, tvp: &mut [libc::timespec; 2]) {
    let c_name = match CString::new(fname) {
        Ok(s) => s,
        Err(_) => err_exit(1, fname),
    };
    let mut sb: libc::stat = unsafe { std::mem::zeroed() };
    if unsafe { libc::stat(c_name.as_ptr(), &mut sb) } != 0 {
        err_exit(1, fname);
    }
    tvp[0] = sb.st_atim;
    tvp[1] = sb.st_mtim;
}
