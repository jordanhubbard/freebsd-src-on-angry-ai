fn parse_duration(d: &str) -> f64 {
    let b = d.as_bytes();
    let mut end = b.len();
    let mut suf = '\0';
    for (i, &c) in b.iter().enumerate() {
        if !c.is_ascii_digit() && c != b'.' && c != b'-' && c != b'+' && c != b'e' && c != b'E' {
            end = i; suf = c as char; break;
        }
    }
    let ret: f64 = match d[..end].parse() {
        Ok(v) => v,
        Err(_) => errx(EXIT_INVALID, "duration is not a number"),
    };
    if suf == '\0' { return ret; }
    if end + 1 != b.len() { errx(EXIT_INVALID, "duration unit suffix too long"); }
    let ret = match suf {
        's' => ret, 'm' => ret * 60.0, 'h' => ret * 3600.0, 'd' => ret * 86400.0,
        _ => errx(EXIT_INVALID, "duration unit suffix invalid"),
    };
    if ret < 0.0 || ret >= 100000000.0 { errx(EXIT_INVALID, "duration out of range"); }
    ret
}

extern "C" fn sig_handler(signo: i32) {
    if signo == KILLSIG.load(Ordering::Relaxed) {

                if rk.rk_fpid > 0 { warnx(&format!("failed to signal some processes: first pid={}", rk.rk_fpid)); }
            }
        }
        logv(&format!("signaled {} processes", rk.rk_killed));
    }
    if signo != SIGKILL && signo != SIGSTOP && signo != SIGCONT {
        let cn = signame(SIGCONT);

        }
    }
}

fn kill_self(signo: i32) -> ! {
    let sn = signame(signo);
    logv(&format!("killing self with signal {}({})", sn, signo));
    let mut rl: rlimit = unsafe { std::mem::zeroed() };
    unsafe { setrlimit(RLIMIT_CORE, &rl) };
    unsafe {
        let mut mask: sigset_t = std::mem::zeroed();
        sigemptyset(&mut mask);
        sigprocmask(SIG_SETMASK, &mask, std::ptr::null_mut());
        kill(std::process::id() as i32, signo);
    }
    process::exit(128 + signo as i32);
}
