fn addgroup(grpname: &str) {
    unsafe {
        let egid = getegid();

        let c_grpname = CString::new(grpname).unwrap();
        let mut grp = getgrnam(c_grpname.as_ptr());

        if grp.is_null() {
            let mut ep: *mut c_char = ptr::null_mut();
            let lgid = strtol(c_grpname.as_ptr(), &mut ep, 10);
            if lgid <= 0 || !ep.is_null() && *ep != 0 ||
               (grp = getgrgid(lgid as gid_t)).is_null() {
                warnx(
                    b"%s: bad group name\0".as_ptr() as *const c_char,
                    c_grpname.as_ptr(),
                );
                return;
            }
        }

        // Check if user is a member of the requested group
        let mut dbmember: bool = false;
        if (*PWD).pw_gid == (*grp).gr_gid {
            dbmember = true;
        }
        let mut p = (*grp).gr_mem;
        while !p.is_null() && *(*p as *const *mut c_char) != ptr::null_mut() {
            if strcmp(*p, (*PWD).pw_name) == 0 {
                dbmember = true;
                break;
            }
            p = p.add(1);
        }

        // If not a member and group has a password, prompt
        if !dbmember && *(*grp).gr_passwd != 0 && getuid() != 0 {
            let pass = getpass(b"Password:\0".as_ptr() as *const c_char);
            if pass.is_null() {
                return;
            }
            let cryptpw = crypt(pass, (*grp).gr_passwd);
            if cryptpw.is_null() || strcmp((*grp).gr_passwd, cryptpw) != 0 {
                fprintf(
                    libc_stderr(),
                    b"Sorry\n\0".as_ptr() as *const c_char,
                );
                return;
            }
        }

        let ngrps_max = sysconf(SC_NGROUPS_MAX) as usize;
        let grps = malloc(std::mem::size_of::<gid_t>() * ngrps_max);
        if grps.is_null() {
            err(1, b"malloc\0".as_ptr() as *const c_char);
        }
        let mut ngrps = getgroups(ngrps_max as c_int, grps as *mut gid_t);
        if ngrps < 0 {
            warn(b"getgroups\0".as_ptr() as *const c_char);
            free(grps);
            return;
        }

        // Read groups into a Rust slice for easier manipulation
        let groups_slice: &mut [gid_t] =
            std::slice::from_raw_parts_mut(grps as *mut gid_t, ngrps as usize);

        // Remove requested gid from supp. list if it exists and doesn't match prior egid
        if (*grp).gr_gid != egid && inarray((*grp).gr_gid, groups_slice) {
            if let Some(pos) = groups_slice.iter().position(|&g| g == (*grp).gr_gid) {
                let remaining = ngrps as usize - 1 - pos;
                if remaining > 0 {
                    memmove(
                        groups_slice.as_mut_ptr().add(pos) as *mut c_void,
                        groups_slice.as_ptr().add(pos + 1) as *const c_void,
                        remaining * std::mem::size_of::<gid_t>(),
                    );
                }
                ngrps -= 1;
                // Re-slice
                let _ = std::mem::replace(
                    &mut groups_slice,
                    std::slice::from_raw_parts_mut(grps as *mut gid_t, ngrps as usize),
                );
                priv_start();
                if setgroups(ngrps, grps as *const gid_t) < 0 {
                    priv_end();
                    warn(b"setgroups\0".as_ptr() as *const c_char);
                    free(grps);
                    return;
                }
                priv_end();
            }
        }

        // Set the new effective gid
        priv_start();
        if setgid((*grp).gr_gid) < 0 {
            priv_end();
            warn(b"setgid\0".as_ptr() as *const c_char);
            free(grps);
            return;
        }
        priv_end();

        // Add old effective gid to supp. list if it does not exist
        let groups_slice: &mut [gid_t] =
            std::slice::from_raw_parts_mut(grps as *mut gid_t, ngrps as usize);
        if !inarray(egid, groups_slice) {
            if ngrps as usize == ngrps_max {
                warnx(b"too many groups\0".as_ptr() as *const c_char);
            } else {
                *(grps as *mut gid_t).add(ngrps as usize) = egid;
                ngrps += 1;
                priv_start();
                if setgroups(ngrps, grps as *const gid_t) < 0 {
                    priv_end();
                    warn(b"setgroups\0".as_ptr() as *const c_char);
                    free(grps);
                    return;
                }
                priv_end();
            }
        }

        free(grps);
    }
}

fn loginshell() {
    unsafe {
        let shell = (*PWD).pw_shell;
        let shell_str = if *shell == 0 {
            CString::new(PATH_BSHELL).unwrap()
        } else {
            CStr::from_ptr(shell).to_owned()
        };

        let c_dir = CString::new((*PWD).pw_dir).unwrap();
        if chdir(c_dir.as_ptr()) < 0 {
            warn((*PWD).pw_dir);
            chdir(b"/\0".as_ptr() as *const c_char);
        }

        let term = getenv(b"TERM\0".as_ptr() as *const c_char);
        let ticket = getenv(b"KRBTKFILE\0".as_ptr() as *const c_char);

        let cleanenv = calloc(20, std::mem::size_of::<*mut c_char>());
        if cleanenv.is_null() {
            err(1, b"calloc\0".as_ptr() as *const c_char);
        }
        *(cleanenv as *mut *mut c_char) = ptr::null_mut();
        environ = cleanenv as *mut *mut c_char;

        let lc = login_getpwclass(PWD);
        setusercontext(lc, PWD, (*PWD).pw_uid, LOGIN_SETPATH | LOGIN_SETUMASK | LOGIN_SETENV);
        login_close(lc);

        setenv(b"USER\0".as_ptr() as *const c_char, (*PWD).pw_name, 1);
        setenv(b"SHELL\0".as_ptr() as *const c_char, shell, 1);
        setenv(b"HOME\0".as_ptr() as *const c_char, (*PWD).pw_dir, 1);
        if !term.is_null() {
            setenv(b"TERM\0".as_ptr() as *const c_char, term, 1);
        }
        if !ticket.is_null() {
            setenv(b"KRBTKFILE\0".as_ptr() as *const c_char, ticket, 1);
        }

        // Build "-<shell>" argument
        let shell_path = shell_str.to_string_lossy().into_owned();
        let arg0 = format!("-{}", shell_path);
        let c_arg0 = CString::new(arg0).unwrap();
        let c_shell = shell_str;

        let mut args: [*mut c_char; 2] = [c_arg0.as_ptr() as *mut c_char, ptr::null_mut()];

        execv(c_shell.as_ptr(), args.as_ptr() as *const *mut c_char);
        err(1, shell_str.as_ptr());
    }
}

fn doshell() {

        let shell_str = if *shell == 0 {
            CString::new(PATH_BSHELL).unwrap()
        } else {

        execl(shell_str.as_ptr(), shell_str.as_ptr(), ptr::null_mut());
        err(1, shell_str.as_ptr());
    }
}
