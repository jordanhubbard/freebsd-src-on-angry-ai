use std::env;
use std::ffi::{CStr, CString};
use std::os::raw::{c_char, c_int};
use std::process;

const UFS_ROOTINO: c_int = 2;

// Opaque pointer to C struct uufsd
type UufsdPtr = *mut std::ffi::c_void;

// Opaque pointer to C union dinodep
type DinodepPtr = *mut std::ffi::c_void;

extern "C" {
    fn ufs_disk_fillout(disk: UufsdPtr, fsname: *const c_char) -> c_int;
    fn ufs_disk_close(disk: UufsdPtr) -> c_int;
    fn getinode(disk: UufsdPtr, dp: DinodepPtr, ino: c_int) -> c_int;

    // Our C glue functions
    fn clri_alloc_disk() -> UufsdPtr;
    fn clri_free_disk(disk: UufsdPtr);
    fn clri_clear_inode(disk: UufsdPtr, dp: DinodepPtr) -> c_int;
    fn clri_disk_error(disk: UufsdPtr) -> *const c_char;
}

fn usage() -> ! {
    eprintln!("usage: clri special_device inode_number ...");
    process::exit(1);
}

fn main() {
    let args: Vec<String> = env::args().collect();
    if args.len() < 3 {
        usage();
    }

    let fsname = &args[1];
    let inodes = &args[2..];

    let c_fsname = match CString::new(fsname.as_bytes()) {
        Ok(s) => s,
        Err(_) => {
            eprintln!("Invalid filesystem path");
            process::exit(1);
        }
    };

    let disk = unsafe { clri_alloc_disk() };
    if disk.is_null() {
        eprintln!("Failed to allocate disk structure");
        process::exit(1);
    }

    if unsafe { ufs_disk_fillout(disk, c_fsname.as_ptr()) } == -1 {
        let err = unsafe {
            let eptr = clri_disk_error(disk);
            if !eptr.is_null() {
                CStr::from_ptr(eptr).to_string_lossy().into_owned()
            } else {
                "unknown error".to_string()
            }
        };
        eprintln!("loading superblock: {}", err);
        unsafe { clri_free_disk(disk); }
        process::exit(1);
    }

    let mut exitval: i32 = 0;

    for inode_str in inodes {
        let inonum: c_int = match inode_str.parse() {
            Ok(n) => n,
            Err(_) => {
                eprintln!("{} is not a valid inode number", inode_str);
                exitval = 1;
                continue;
            }
        };

        if inonum < UFS_ROOTINO {
            eprintln!("{} is not a valid inode number", inode_str);
            exitval = 1;
            continue;
        }

        println!("clearing {}", inonum);

        // Allocate a union dinodep on the heap (C union, opaque to Rust)
        // The C union dinodep is two pointers (16 bytes on 64-bit)
        let dp_size = 16usize;
        let dp_align = 8usize;
        let dp = unsafe {
            let ptr = std::alloc::alloc(std::alloc::Layout::from_size_align(dp_size, dp_align).unwrap());
            std::ptr::write_bytes(ptr, 0, dp_size);
            ptr as DinodepPtr
        };

        if unsafe { getinode(disk, dp, inonum) } == -1 {
            let err = unsafe {
                let eptr = clri_disk_error(disk);
                if !eptr.is_null() {
                    CStr::from_ptr(eptr).to_string_lossy().into_owned()
                } else {
                    "unknown error".to_string()
                }
            };
            eprintln!("getinode: {}", err);
            unsafe {
                std::alloc::dealloc(dp as *mut u8, std::alloc::Layout::from_size_align(dp_size, dp_align).unwrap());
            }
            exitval = 1;
            continue;
        }

        if unsafe { clri_clear_inode(disk, dp) } == -1 {
            eprintln!("putinode or fsync failed");
            unsafe {
                std::alloc::dealloc(dp as *mut u8, std::alloc::Layout::from_size_align(dp_size, dp_align).unwrap());
            }
            exitval = 1;
            continue;
        }

        unsafe {
            std::alloc::dealloc(dp as *mut u8, std::alloc::Layout::from_size_align(dp_size, dp_align).unwrap());
        }
    }

    unsafe {
        ufs_disk_close(disk);
        clri_free_disk(disk);
    }
    process::exit(exitval);
}
