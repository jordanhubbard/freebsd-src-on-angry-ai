use std::io::{self, Read, Write};
use std::os::unix::io::AsRawFd;
use std::os::unix::net::UnixStream;
use std::process;
use std::time::{Duration, Instant};

const QUERY: &[u8] = b"\x1b7\x1b[r\x1b[999;999H\x1b[6n\x1b8";

fn usage() -> ! {
    eprintln!("usage: resizewin [-z]");
    process::exit(1);
}

fn main() {
    let args: Vec<String> = std::env::args().collect();
    let mut zflag = false;

    for arg in &args[1..] {
        match arg.as_str() {
            "-z" => zflag = true,
            _ => usage(),
        }
    }

    let fd = match open_tty() {
        Ok(fd) => fd,
        Err(_) => process::exit(1),
    };

    if zflag {
        let mut w = Winsize {
            ws_row: 0,
            ws_col: 0,
            ws_xpixel: 0,
            ws_ypixel: 0,
        };
        if unsafe { libc::ioctl(fd.as_raw_fd(), libc::TIOCGWINSZ, &mut w as *mut Winsize) } == -1 {
            process::exit(1);
        }
        if w.ws_row != 0 && w.ws_col != 0 {
            process::exit(0);
        }
    }

        Ok(t) => t,
        Err(_) => process::exit(1),
    };

        ws_col: 0,
        ws_xpixel: 0,
        ws_ypixel: 0,

        error = 1;

        let mut data = [0u8; 20];
        let start = Instant::now();
        let mut cnt: usize = 0;

        loop {
            match fd.read(&mut data[cnt..cnt + 1]) {
                Ok(0) => {
                    error = 1;
                    break;
                }
                Ok(1) => {
                    if data[cnt] == b'R' {
                        break;
                    }
                    cnt += 1;
                    if cnt >= data.len() - 2 {

                        error = 1;
                        break;

                    continue;
                }

fn parse_cursor_pos(data: &[u8]) -> Result<(u16, u16), ()> {
    // Expected format: ESC[row;colR
    if data.is_empty() || data[0] != 0x1b || data[data.len() - 1] != b'R' {
        return Err(());
    }
    let inner = &data[2..data.len() - 1];
    let s = match std::str::from_utf8(inner) {
        Ok(s) => s,
        Err(_) => return Err(()),
    };
    let parts: Vec<&str> = s.split(';').collect();
    if parts.len() != 2 {
        return Err(());
    }
    let row: u16 = parts[0].parse().map_err(|_| ())?;
    let col: u16 = parts[1].parse().map_err(|_| ())?;
    Ok((row, col))
}

fn open_tty() -> io::Result<UnixStream> {
    // Open /dev/tty with O_RDWR | O_NONBLOCK
    let fd = unsafe {
        libc::open(
            b"/dev/tty\0".as_ptr() as *const libc::c_char,
            libc::O_RDWR | libc::O_NONBLOCK,
        )
    };
    if fd < 0 {
        return Err(io::Error::last_os_error());

}

#[derive(Copy, Clone)]
struct Winsize {
    ws_row: u16,
    ws_col: u16,
    ws_xpixel: u16,
    ws_ypixel: u16,
}

#[repr(C)]
#[derive(Copy, Clone)]
struct Termios {
    c_iflag: libc::c_ulong,
    c_oflag: libc::c_ulong,
    c_cflag: libc::c_ulong,
    c_lflag: libc::c_ulong,
    c_line: libc::cc_t,
    c_cc: [libc::cc_t; libc::NCCS],
}

extern "C" {
    fn tcgetattr(fd: libc::c_int, termios_p: *mut Termios) -> libc::c_int;
    fn tcsetattr(fd: libc::c_int, optional_actions: libc::c_int, termios_p: *const Termios) -> libc::c_int;
}

fn get_termios(fd: libc::c_int) -> Result<Termios, ()> {
    let mut t: Termios = unsafe { std::mem::zeroed() };
    if unsafe { tcgetattr(fd, &mut t) } == -1 {
        Err(())
    } else {
        Ok(t)
    }
}

fn set_termios(fd: libc::c_int, t: &Termios, action: libc::c_int) -> libc::c_int {
    unsafe { tcsetattr(fd, action, t) }
}
