fn compute_diff(a: &[String], b: &[String]) -> Vec<Edit> {
    let n = a.len(); let m = b.len();
    let mut dp = vec![vec![0usize; m+1]; n+1];
    for i in 1..=n { for j in 1..=m {
        dp[i][j] = if a[i-1] == b[j-1] { dp[i-1][j-1]+1 } else { dp[i-1][j].max(dp[i][j-1]) };
    }}
    let mut edits = Vec::new();
    let (mut i, mut j) = (n, m);
    let (mut pd, mut pi) = (0, 0);
    while i > 0 || j > 0 {
        if i > 0 && j > 0 && a[i-1] == b[j-1] {
            if pd > 0 { edits.push(Edit::Del(i-pd, pd)); pd = 0; }
            if pi > 0 { edits.push(Edit::Ins(j-pi, pi)); pi = 0; }
            let mut c = 1;
            while i > c && j > c && a[i-c-1] == b[j-c-1] { c += 1; }
            edits.push(Edit::Eq(c));
            i -= c; j -= c;
        } else if i > 0 && (j == 0 || dp[i-1][j] >= dp[i][j-1]) {
            pd += 1; i -= 1;
        } else {
            pi += 1; j -= 1;
        }
    }
    if pd > 0 { edits.push(Edit::Del(0, pd)); }
    if pi > 0 { edits.push(Edit::Ins(0, pi)); }
    edits.reverse();
    edits
}
