#include <string.h>

static int
ifunc1(void)
{
	return 0xdeadbeef;
}

static int
ifunc2(void)
{
	return 0xbeefdead;
}

static __attribute__((used))
int (*resolve_ifunc(void))(void)
{
	const char *e = getenv("USE_IFUNC2");
	return e && strcmp(e, "1") == 0 ? ifunc2 : ifunc1;
}

__ifunc(ifunc, resolve_ifunc);
