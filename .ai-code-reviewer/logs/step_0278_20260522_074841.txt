fn makemsg(fname: Option<&str>, mbuf: &mut *mut i8) -> bool {
    let mut tmpname = [0i8; 64];
    unsafe {
        let tmpl = CString::new(format!("{}/wall.XXXXXX", PATH_TMP)).unwrap();
        tmpname[..tmpl.as_bytes_with_nul().len()].copy_from_slice(tmpl.as_bytes_with_nul());
    }

    if fd == -1 {
        unsafe { perror(CString::new("can't open temporary file").unwrap().as_ptr()) };
        return false;
    }

    let fp = unsafe { fdopen(fd, b"r+\0".as_ptr() as *const i8) };
    if fp.is_null() {
        unsafe { close(fd) };
        unsafe { perror(CString::new("can't open temporary file").unwrap().as_ptr()) };
        return false;
    }

    let whom_str = if whom.is_null() {
        let uid = unsafe { getuid() };
        let pw = unsafe { getpwuid(uid) };
        if pw.is_null() {
            CString::new("???").unwrap()
        } else {
            let name = unsafe { (*pw).pw_name };
            if name.is_null() {
                CString::new("???").unwrap()
            } else {
                unsafe { CStr::from_ptr(name).to_owned() }
            }
        }
    } else {
        unsafe { CStr::from_ptr(whom).to_owned() }
    };

    let mut hostname = [0i8; MAXHOSTNAMELEN];
    unsafe { gethostname(hostname.as_mut_ptr(), hostname.len()) };
    let hostname_str = unsafe { CStr::from_ptr(hostname.as_ptr()).to_string_lossy().into_owned() };

    let mut now: i64 = 0;
    unsafe { time(&mut now) };
    let lt = unsafe { localtime(&now) };
    let hour = unsafe { (*lt).tm_hour };
    let min = unsafe { (*lt).tm_min };

    let tty = unsafe { ttyname(2) };
    let tty_str = if tty.is_null() {
        "no tty".to_string()
    } else {
        unsafe { CStr::from_ptr(tty).to_string_lossy().into_owned() }
    };

    unsafe {
        let fmt = CString::new("Remote Broadcast Message from %s@%s\n").unwrap();
        fprintf(fp, fmt.as_ptr(), whom_str.as_ptr(), hostname_str.as_ptr());

        let fmt2 = CString::new("        (%s) at %d:%02d ...\n").unwrap();
        fprintf(fp, fmt2.as_ptr(), tty_str.as_ptr(), hour, min);

        putc(b'\n' as i32, fp);
    }

    if let Some(fn_) = fname {
        let fn_cstr = CString::new(fn_).unwrap();
        let stdin_fp = unsafe { freopen(fn_cstr.as_ptr(), b"r\0".as_ptr() as *const i8, stdin_file()) };
        if stdin_fp.is_null() {
            eprintln!("can't read {}", fn_);
            unsafe { fclose(fp) };
            return false;
        }
    }

    let mut lbuf = [0i8; 256];
    loop {
        let line = unsafe { fgets(lbuf.as_mut_ptr(), lbuf.len() as i32, stdin_file()) };
        if line.is_null() {
            break;
        }
        unsafe { fputs(line, fp) };
    }

    unsafe { rewind(fp) };

    let mut sbuf = stat { st_size: 0 };
    if unsafe { fstat(fd, &mut sbuf) } == -1 {
        unsafe { fclose(fp) };
        unsafe { perror(CString::new("can't stat temporary file").unwrap().as_ptr()) };
        return false;
    }

    let mbufsize = sbuf.st_size as usize;
    let buf = unsafe { malloc(mbufsize) };
    if buf.is_null() {
        unsafe { fclose(fp) };
        eprintln!("out of memory");
        return false;
    }

    let n = unsafe { fread(buf as *mut c_void, 1, mbufsize, fp) };
    if n != mbufsize {
        unsafe { fclose(fp) };
        eprintln!("can't read temporary file");
        return false;
    }

    unsafe { fclose(fp) };
    *mbuf = buf;
    true
}
