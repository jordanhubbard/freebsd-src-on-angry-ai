/// Format a u64 as "0x%016x" into a buffer, returns length written
unsafe fn format_hex64(buf: &mut [c_char], val: u64) -> usize {
    let mut i = 0;
    buf[i] = b'0' as c_char; i += 1;
    buf[i] = b'x' as c_char; i += 1;

/// Format a u32 as decimal string into a buffer, returns length written
unsafe fn format_dec(buf: &mut [c_char], val: u32) -> usize {
    if val == 0 {
        buf[0] = b'0' as c_char;
        buf[1] = 0;
        return 2;
    }
    // Collect digits in reverse
    let mut digits = [0u8; 12];
    let mut count = 0;
    let mut v = val;
    while v > 0 {
        digits[count] = b'0' + (v % 10) as u8;
        v /= 10;
        count += 1;
    }
    // Write in correct order
    let mut i = 0;
    for n in (0..count).rev() {
        buf[i] = digits[n] as c_char;
        i += 1;
    }
    buf[i] = 0;
    i + 1
}
