fn chacha20_quarter_round(state: &mut [u32; 16], a: usize, b: usize, c: usize, d: usize) {
    state[a] = state[a].wrapping_add(state[b]);
    state[d] ^= state[a];
    state[d] = state[d].rotate_left(16);

    state[c] = state[c].wrapping_add(state[d]);

    state[c] = state[c].wrapping_add(state[d]);
    state[b] ^= state[c];
    state[b] = state[b].rotate_left(7);
}

fn chacha20_stream(key: &[u8; 32], counter: u64, nonce: &[u8; 8], out: &mut [u8]) {
    let mut pos = 0;
    let mut ctr = counter;
    while pos < out.len() {
        let block = chacha20_block(key, ctr, nonce);
        let remaining = out.len() - pos;
        let to_copy = remaining.min(64);
        out[pos..pos + to_copy].copy_from_slice(&block[..to_copy]);
        pos += to_copy;
        ctr = ctr.wrapping_add(1);
    }
}

fn chacha20_xor_ic(key: &[u8; 32], counter: u64, nonce: &[u8; 8], msg: &[u8], out: &mut [u8]) {
    let mut pos = 0;
    let mut ctr = counter;
    while pos < msg.len() {
        let block = chacha20_block(key, ctr, nonce);
        let remaining = msg.len() - pos;
        let to_copy = remaining.min(64);
        for i in 0..to_copy {
            out[pos + i] = msg[pos + i] ^ block[i];
        }
        pos += to_copy;
        ctr = ctr.wrapping_add(1);
    }
}

    c: *mut u8,
    clen: usize,
    n: *const u8,
    k: *const u8,

    if c.is_null() || n.is_null() || k.is_null() {
        return -1;
    }

    let nonce = std::slice::from_raw_parts(n, NONCEBYTES);
    let key = std::slice::from_raw_parts(k, KEYBYTES);

    key_bytes.copy_from_slice(key);

    nonce_bytes.copy_from_slice(&nonce[..HCHACHA20_INPUTBYTES]);

    0
}
