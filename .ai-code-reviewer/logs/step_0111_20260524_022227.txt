// Rust replacement for des_keys.c
// Implements DES odd parity key fixup directly in Rust,
// avoiding OpenSSL FFI dependency.

/// Set odd parity on each byte of a DES key.
/// Each byte is adjusted so that the total number of 1-bits is odd.
#[no_mangle]
pub extern "C" fn k5_des_fixup_key_parity(keybits: *mut u8) {
    if keybits.is_null() {
        return;
    }
    // DES key is 8 bytes (64 bits, 56 key bits + 8 parity bits)
    for i in 0..8 {
        let mut byte = unsafe { *keybits.add(i) };
        // Set bit 0 (LSB) to make total number of 1-bits odd
        let ones = byte.count_ones();
        if ones % 2 == 0 {
            byte |= 1;
        } else {
            byte &= !1;
        }
        unsafe { *keybits.add(i) = byte };
    }
}
