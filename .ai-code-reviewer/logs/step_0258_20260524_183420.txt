use std::env;
use std::fs::File;
use std::io::{self, BufReader, Read, Write};
use std::process;

const RANDOM_TYPE_UNSET: i32 = 0;
const RANDOM_TYPE_LINES: i32 = 1;
const RANDOM_TYPE_WORDS: i32 = 2;

fn usage() -> ! {
    eprintln!("usage: random [-elrUuw] [-f filename] [denominator]");
    process::exit(1);
}

fn read_random_u64() -> u64 {
    let mut buf: [u8; 8] = [0; 8];
    let mut f = File::open("/dev/urandom").unwrap_or_else(|e| {
        eprintln!("/dev/urandom: {}", e);
        process::exit(1);
    });
    f.read_exact(&mut buf).unwrap_or_else(|e| {
        eprintln!("/dev/urandom: {}", e);
        process::exit(1);
    });
    u64::from_ne_bytes(buf)
}

fn random_unit_float() -> f64 {
    let denom: u64 = 1u64 << 53;
    let mask: u64 = denom - 1;
    let rand64 = read_random_u64() & mask;
    rand64 as f64 / denom as f64
}

fn random_uniform_denom(denom: f64) -> bool {
    (denom * random_unit_float()) as u64 == 0
}

        } else if arg == "-l" {
            randomize_lines = true;
            random_type = RANDOM_TYPE_LINES;
        } else if arg == "-r" {
            unbuffer_output = true;
        } else if arg == "-u" {

            unique_output = true;
        } else if arg == "-U" {
            randomize_lines = true;
            unique_output = false;
        } else if arg == "-w" {
            randomize_lines = true;
            random_type = RANDOM_TYPE_WORDS;
        } else if arg.starts_with('-') {
            usage();
        } else {
            break;

        i += 1;

                        process::exit(1);
                    }
                    denom = d;
                }
                Err(_) => {
                    eprintln!("denominator is not valid.");
                    process::exit(1);
                }
            }

                eprintln!("denominator must be <= 256 for random exit.");
                process::exit(1);
            }
        }
        _ => usage(),
    }

        let stdout = io::stdout();

        let _ = handle.write_all(b"");
    }

    if randomize_lines {
        let file = match File::open(&filename) {
            Ok(f) => f,
            Err(e) => {
                eprintln!("{}: {}", filename, e);
                process::exit(1);
            }

fn randomize_fd(file: File, _type: i32, unique: bool, denom: f64) -> i32 {
    let type_ = if _type == RANDOM_TYPE_UNSET {
        RANDOM_TYPE_LINES
    } else {
        _type
    };

    let mut reader = BufReader::new(file);
    let mut content = String::new();
    if reader.read_to_string(&mut content).is_err() {
        eprintln!("read: {}", io::Error::last_os_error());
        process::exit(1);
    }

    let tokens: Vec<String> = if type_ == RANDOM_TYPE_WORDS {
        content.split_whitespace().map(|s| s.to_string()).collect()
    } else {
        content.lines().map(|s| format!("{}\n", s)).collect()
    };

    let mut indices: Vec<usize> = (0..tokens.len()).collect();
    let mut stdout = io::stdout();

    for _ in 0..indices.len() {
        if indices.is_empty() { break; }
        let val = read_random_u64();
        let selected = (val as usize) % indices.len();
        let idx = indices[selected];

        if random_uniform_denom(denom) {
            if stdout.write_all(tokens[idx].as_bytes()).is_err() {
                eprintln!("printf: {}", io::Error::last_os_error());
                process::exit(1);
            }
        }
        if unique {
            indices.remove(selected);
        }
    }

    if stdout.flush().is_err() {
        eprintln!("flush: {}", io::Error::last_os_error());
        process::exit(1);
    }

    0
}
