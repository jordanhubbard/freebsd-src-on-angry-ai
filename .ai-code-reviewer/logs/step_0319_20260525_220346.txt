fn glob_match_rec(p: &[char], pi: usize, t: &[char], ti: usize) -> bool {
    if pi == p.len() {
        return ti == t.len();
    }
    if p[pi] == '*' {
        for i in ti..=t.len() {
            if glob_match_rec(p, pi + 1, t, i) {
                return true;

        return false;

    if p[pi] == '?' {
        return ti < t.len() && glob_match_rec(p, pi + 1, t, ti + 1);

    if ti < t.len() && p[pi] == t[ti] {

fn parse_script_line(line: &str) -> Option<ScriptCmd> {
    let line = line.trim();
    if line.is_empty() || line.starts_with('#') {
        return None;
    }
    let upper = line.to_uppercase();
    if upper.starts_with("ABORT ") || upper.starts_with("ABORT\t") {
        let s = line.splitn(2, |c: char| c == ' ' || c == '\t')
            .nth(1).unwrap_or("").trim().to_string();
        return Some(ScriptCmd::Abort(s));
    }
    if upper.starts_with("TIMEOUT ") || upper.starts_with("TIMEOUT\t") {
        let s = line.splitn(2, |c: char| c == ' ' || c == '\t')
            .nth(1).unwrap_or("").trim();
        let t: u64 = s.parse().unwrap_or(DEFAULT_TIMEOUT);
        return Some(ScriptCmd::Timeout(t));
    }
    if upper == "ECHO" || upper.starts_with("ECHO ON") {
        return Some(ScriptCmd::Echo(true));
    }
    if upper == "NOECHO" || upper.starts_with("ECHO OFF") {
        return Some(ScriptCmd::Echo(false));
    }
    if upper.starts_with("SAY ") || upper.starts_with("SAY\t") {
        let s = line.splitn(2, |c: char| c == ' ' || c == '\t')
            .nth(1).unwrap_or("").trim().to_string();
        return Some(ScriptCmd::Say(s));
    }
    if upper.starts_with("REPORT ") || upper.starts_with("REPORT\t") {
        let s = line.splitn(2, |c: char| c == ' ' || c == '\t')
            .nth(1).unwrap_or("").trim().to_string();
        return Some(ScriptCmd::Report(s));
    }
    if upper == "CLR_ABORT" {
        return Some(ScriptCmd::ClrAbort);
    }
    if upper == "CLR_REPORT" {
        return Some(ScriptCmd::ClrReport);
    }
    if upper.starts_with("HANGUP ") || upper.starts_with("HANGUP\t") {
        let s = line.splitn(2, |c: char| c == ' ' || c == '\t')
            .nth(1).unwrap_or("").trim().to_uppercase();
        return Some(ScriptCmd::Hup(s != "NO"));
    }
    // Expect/send pair: expect_string send_string
    let parts: Vec<&str> = line.splitn(2, |c: char| c == ' ' || c == '\t').collect();

        return Some(ScriptCmd::Expect(parts[0].trim().to_string()));
    }
    if parts.len() == 1 {
        return Some(ScriptCmd::Expect(parts[0].trim().to_string()));
    }
    None
}

                                        .write_all(format!("got: '{}'\n", trimmed).as_bytes());
                                }
                                // Check aborts first
                                for (ai, ab) in aborts.iter().enumerate() {
                                    let ab_subbed = substitute(ab, &o.phone_num, &o.phone_num2);
                                    if glob_match(&ab_subbed, trimmed) {
                                        if o.verbose {
                                            let _ = io::stderr().lock()
                                                .write_all(format!("abort: '{}'\n", ab).as_bytes());
                                        }
                                        return 4 + ai as i32;
                                    }
                                }
                                if glob_match(&exp_subbed, trimmed) {
                                    break;
                                }
                            }
                        }
                        Err(_) => break,
                    }
                }

                // Find and send the corresponding send string
                for (ci, c) in cmds.iter().enumerate() {
                    if let ScriptCmd::Expect(e) = c {
                        let es = substitute(e, &o.phone_num, &o.phone_num2);
                        if es == exp_subbed {
                            if ci + 1 < cmds.len() {
                                if let ScriptCmd::Send(s) = &cmds[ci + 1] {
                                    let send_subbed = substitute(s, &o.phone_num, &o.phone_num2);
                                    if echo {
                                        let _ = io::stdout().lock().write_all(send_subbed.as_bytes());
                                    } else {
                                        let _ = io::stdout().lock().write_all(send_subbed.as_bytes());
                                    }
                                    let _ = io::stdout().flush();
                                    if o.verbose {
                                        let _ = io::stderr().lock()
                                            .write_all(format!("send: '{}'\n", send_subbed).as_bytes());
                                    }
                                }
                            }
                            break;
                        }
                    }
                }
            }
            ScriptCmd::Send(s) => {
                let send_subbed = substitute(s, &o.phone_num, &o.phone_num2);
                if echo {
                    let _ = io::stdout().lock().write_all(send_subbed.as_bytes());
                } else {
                    let _ = io::stdout().lock().write_all(send_subbed.as_bytes());
                }
                let _ = io::stdout().flush();
                if o.verbose {
                    let _ = io::stderr().lock()
                        .write_all(format!("send: '{}'\n", send_subbed).as_bytes());
                }
            }
        }
    }

    // Write report file
    if let Some(ref rf) = o.report_file {
        if let Ok(mut f) = fs::File::create(rf) {
            for r in &reports {
                let _ = writeln!(f, "{}", r);
            }
        }
    }

    exit_code
}
