// locate.bigram - list bigrams for /usr/libexec/locate.mklocatedb script

use std::io::{self, BufRead, Write};

const LOCATE_PATH_MAX: usize = 1024;
const ASCII_MIN: u8 = 32;
const ASCII_MAX: u8 = 127;

fn main() {
    let mut bigram = [[0u64; 256]; 256];
    let mut buf1 = vec![b' '; LOCATE_PATH_MAX];
    let mut buf2 = vec![0u8; LOCATE_PATH_MAX];
    let mut oldpath = &mut buf1;
    let mut path = &mut buf2;

    let stdin = io::stdin();
    let mut reader = stdin.lock();
    let mut line = String::new();

    loop {
        line.clear();
        match reader.read_line(&mut line) {
            Ok(0) => break,
            Ok(_) => {}
            Err(_) => {
                eprintln!("locate.bigram: stdin");
                std::process::exit(1);
            }
        }

        // Copy line into path buffer
        let line_bytes = line.as_bytes();
        let len = line_bytes.len().min(LOCATE_PATH_MAX - 1);
        path[..len].copy_from_slice(&line_bytes[..len]);
        path[len] = b'\0';

        // skip longest common prefix
        let mut cp = 0;
        let mut old_cp = 0;
        while cp < len && old_cp < oldpath.len() && path[cp] == oldpath[old_cp] {
            if path[cp] == b'\0' {
                break;
            }
            cp += 1;
            old_cp += 1;
        }

        // Count bigrams
        while cp + 1 < len {
            let c1 = path[cp] as usize;
            let c2 = path[cp + 1] as usize;
            bigram[c1][c2] += 1;
            cp += 2;
        }

        // swap pointers
        std::mem::swap(&mut path, &mut oldpath);
    }

    // output, boundary check
    let stdout = io::stdout();
    let mut out = stdout.lock();

        for j in ASCII_MIN..=ASCII_MAX {
            if bigram[i as usize][j as usize] != 0 {
                let _ = writeln!(out, "{} {}{}", bigram[i as usize][j as usize], i as char, j as char);
            }
        }
    }

    std::process::exit(0);
}
