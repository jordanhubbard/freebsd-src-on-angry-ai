use std::ffi::CString;
use std::fs::File;
use std::io::{self, Write};
use std::os::raw::{c_char, c_int, c_uint, c_void};
use std::process;

// FreeBSD uuid_t structure (sys/uuid.h)
#[repr(C)]
struct uuid_t {
    time_low: u32,
    time_mid: u16,
    time_hi_and_version: u16,
    clock_seq_hi_and_reserved: u8,
    clock_seq_low: u8,
    node: [u8; 6],
}

const UUIDGEN_BATCH_MAX: usize = 1024;

// FFI declarations for FreeBSD libutil and libuuid
extern "C" {
    fn uuidgen(uuid: *mut uuid_t, count: c_int) -> c_int;
    fn uuid_to_string(uuid: *const uuid_t, s: *mut *mut c_char, status: *mut c_uint) -> c_uint;
    fn arc4random_buf(buf: *mut c_void, n: usize);
}

const UUID_S_OK: c_uint = 0;

fn usage() -> ! {
    let msg = "usage: uuidgen [-1] [-r] [-n count] [-o filename]\n";
    let _ = io::stderr().write_all(msg.as_bytes());
    process::exit(1);
}

fn uuid_to_compact_string(uuid: &uuid_t) -> String {
    format!(
        "{:08x}{:04x}{:04x}{:02x}{:02x}{:02x}{:02x}{:02x}{:02x}{:02x}{:02x}",
        uuid.time_low,
        uuid.time_mid,
        uuid.time_hi_and_version,
        uuid.clock_seq_hi_and_reserved,
        uuid.clock_seq_low,
        uuid.node[0],
        uuid.node[1],
        uuid.node[2],
        uuid.node[3],
        uuid.node[4],
        uuid.node[5],
    )
}

fn uuidgen_v4(store: &mut [uuid_t]) {
    unsafe {
        arc4random_buf(store.as_mut_ptr() as *mut c_void, store.len() * std::mem::size_of::<uuid_t>());
    }
    for item in store.iter_mut() {
        // Set the two most significant bits of clock_seq_hi_and_reserved
        item.clock_seq_hi_and_reserved &= !(3 << 6);
        item.clock_seq_hi_and_reserved |= 2 << 6;
        // Set the four most significant bits of time_hi_and_version
        item.time_hi_and_version &= !(15 << 12);
        item.time_hi_and_version |= 4 << 12;
    }
}

fn main() {
    let args: Vec<String> = std::env::args().skip(1).collect();

    let mut iterate = false;
    let mut compact = false;
    let mut version = 1;
    let mut count: Option<usize> = None;
    let mut output_file: Option<String> = None;

    let mut i = 0;
    while i < args.len() {
        match args[i].as_str() {
            "-1" => iterate = true,
            "-c" => compact = true,
            "-r" => version = 4,
            "-n" => {
                if count.is_some() {
                    usage();
                }
                i += 1;
                if i >= args.len() {
                    usage();
                }
                let n: usize = match args[i].parse() {
                    Ok(n) if n >= 1 => n,
                    _ => usage(),
                };
                count = Some(n);
            }
            "-o" => {
                if output_file.is_some() {
                    eprintln!("uuidgen: multiple output files not allowed");
                    process::exit(1);
                }
                i += 1;
                if i >= args.len() {
                    usage();
                }
                output_file = Some(args[i].clone());
            }
            _ => usage(),
        }
        i += 1;
    }

    let count = count.unwrap_or(1);

    let mut writer: Box<dyn Write> = match &output_file {
        Some(path) => {
            let f = File::create(path).unwrap_or_else(|e| {
                eprintln!("uuidgen: {}: {}", path, e);
                process::exit(1);
            });
            Box::new(f)
        }
        None => Box::new(io::stdout()),
    };

    let batch_size = std::cmp::min(count, UUIDGEN_BATCH_MAX);
    let mut store: Vec<uuid_t> = vec![unsafe { std::mem::zeroed() }; batch_size];

    let mut remaining = count;
    while remaining > 0 {
        let batch = std::cmp::min(remaining, UUIDGEN_BATCH_MAX);

        if !iterate {
            if version == 1 {
                let ret = unsafe { uuidgen(store.as_mut_ptr(), batch as c_int) };
                if ret != 0 {
                    eprintln!("uuidgen: uuidgen() failed");
                    process::exit(1);
                }
            } else if version == 4 {
                uuidgen_v4(&mut store[..batch]);
            } else {
                eprintln!("uuidgen: unsupported version");
                process::exit(1);
            }
        } else {
            for j in 0..batch {
                if version == 1 {
                    let ret = unsafe { uuidgen(store[j..].as_mut_ptr(), 1) };
                    if ret != 0 {
                        eprintln!("uuidgen: uuidgen() failed");
                        process::exit(1);
                    }
                } else if version == 4 {
                    uuidgen_v4(&mut store[j..j + 1]);
                } else {
                    eprintln!("uuidgen: unsupported version");
                    process::exit(1);
                }
            }
        }

        for j in 0..batch {
            let uuid_str = if compact {
                uuid_to_compact_string(&store[j])
            } else {
                let mut s: *mut c_char = std::ptr::null_mut();
                let mut status: c_uint = 0;
                unsafe {
                    uuid_to_string(&store[j], &mut s, &mut status);
                }
                if status != UUID_S_OK {
                    eprintln!("uuidgen: cannot stringify a UUID");
                    process::exit(1);
                }
                let result = std::ffi::CStr::from_ptr(s).to_string_lossy().into_owned();
                unsafe { libc_free(s as *mut c_void); }
                result
            };
            let _ = writeln!(writer, "{}", uuid_str);
        }

        remaining -= batch;
    }

    let _ = writer.flush();
}

extern "C" {
    fn free(p: *mut c_void);
}

fn libc_free(p: *mut c_void) {
    unsafe { free(p) };
}
