use std::env;
use std::io::{self, Write};
use std::process;

fn usage() -> ! {
    let _ = io::stderr().write_all(b"usage: printenv [name]\n");
    process::exit(1);
}

fn main() {
    let args: Vec<String> = env::args().skip(1).collect();

    if args.is_empty() {
        // Print all environment variables
        for (key, value) in env::vars() {
            if writeln!(io::stdout(), "{}={}", key, value).is_err() {
                eprintln!("printenv: stdout: {}", io::Error::last_os_error());
                process::exit(1);
            }
        }
    } else if args.len() == 1 {
        // Look up a specific environment variable using prefix matching
        // (matching the original C behavior with memcmp)
        let name = &args[0];
        let name_len = name.len();
        let mut found = false;
        for (key, value) in env::vars() {
            if key.len() >= name_len
                && key.as_bytes().starts_with(name.as_bytes())
                && key.as_bytes().get(name_len) == Some(&b'=')
            {
                // Found prefix match - print value after '='
                if writeln!(io::stdout(), "{}", &key[name_len + 1..]).is_err() {
                    eprintln!("printenv: stdout: {}", io::Error::last_os_error());
                    process::exit(1);
                }
                found = true;
                break;
            }
        }
        if !found {
            // Variable not found
            process::exit(1);
        }
    } else {
        // Too many arguments
        usage();
    }

    // Flush stdout and check for errors
    if io::stdout().flush().is_err() {
        eprintln!("printenv: stdout: {}", io::Error::last_os_error());
        process::exit(1);
    }

    process::exit(0);
}
