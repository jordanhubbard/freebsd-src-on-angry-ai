/// Set odd parity on each byte of a DES key.
/// Each byte is adjusted so that the total number of 1-bits is odd.
/// This is the equivalent of OpenSSL's DES_set_odd_parity.
#[no_mangle]
pub unsafe extern "C" fn k5_des_fixup_key_parity(keybits: *mut u8) {
    if keybits.is_null() {
        return;
    }
    // DES key is 8 bytes (DES_cblock)
    for i in 0..8 {
        let byte = *keybits.add(i);
        let mut b = byte;
        // Count bits and adjust parity bit (bit 0) to make odd parity
        let mut count = 0u32;
        let mut tmp = b;
        while tmp != 0 {
            if tmp & 1 != 0 {
                count += 1;
            }
            tmp >>= 1;
        }
        if count % 2 == 0 {
            b ^= 1; // flip LSB to make odd parity
        }
        *keybits.add(i) = b;
    }
}
