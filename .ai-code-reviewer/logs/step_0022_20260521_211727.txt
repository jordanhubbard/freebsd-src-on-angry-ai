fn usage() -> ! {
    let progname = unsafe {
        CStr::from_ptr(getprogname())
            .to_string_lossy()
            .into_owned()
    };
    let msg = format!(
        "Usage: {} [-f | --foreground] [-k time | --kill-after time]\
         [-p | --preserve-status] [-s signal | --signal signal] \
         [-v | --verbose] <duration> <command> [arg ...]\n",
        progname
    );
    let stderr = io::stderr();
    let mut handle = stderr.lock();
    let _ = handle.write_all(msg.as_bytes());
    process::exit(1);
}

fn logv(fmt: &str, args: &str) {
    if !VERBOSE.load(Ordering::Relaxed) {
        return;
    }
    let stderr = io::stderr();
    let mut handle = stderr.lock();
    let _ = handle.write_all(fmt.as_bytes());
    let _ = handle.write_all(b": ");
    let _ = handle.write_all(args.as_bytes());
    let _ = handle.write_all(b"\n");
}

fn parse_duration(duration: &str) -> f64 {
    let bytes = duration.as_bytes();
    let mut end = bytes.len();
    let mut suffix = '\0';

    // Find where the numeric part ends
    for (i, &b) in bytes.iter().enumerate() {
        if !b.is_ascii_digit() && b != b'.' && b != b'-' && b != b'+' && b != b'e' && b != b'E' {
            end = i;
            suffix = b as char;
            break;
        }
    }

    let num_str = &duration[..end];
    let ret: f64 = match num_str.parse() {
        Ok(v) => v,
        Err(_) => errx(EXIT_INVALID, "duration is not a number"),
    };

    if suffix == '\0' {
        return ret;
    }

    // Check that suffix is exactly one character
    if end + 1 != bytes.len() {
        errx(EXIT_INVALID, "duration unit suffix too long");
    }

    let ret = match suffix {
        's' => ret,
        'm' => ret * 60.0,
        'h' => ret * 60.0 * 60.0,
        'd' => ret * 60.0 * 60.0 * 24.0,
        _ => errx(EXIT_INVALID, "duration unit suffix invalid"),
    };

    if ret < 0.0 || ret >= 100000000.0 {
        errx(EXIT_INVALID, "duration out of range");
    }

    ret
}

extern "C" fn sig_handler(signo: i32) {
    let killsig = KILLSIG.load(Ordering::Relaxed);
    if signo == killsig {
        SIG_TERM_VAL.store(signo, Ordering::Relaxed);
        SIG_TERM.store(true, Ordering::Relaxed);
        return;
    }

    match signo {
        SIGCHLD => {
            SIG_CHLD.store(true, Ordering::Relaxed);
        }
        SIGALRM => {
            SIG_ALRM.store(true, Ordering::Relaxed);
        }
        SIGHUP | SIGINT | SIGQUIT | SIGILL | SIGTRAP | SIGABRT | SIGEMT
        | SIGFPE | SIGBUS | SIGSEGV | SIGSYS | SIGPIPE | SIGTERM | SIGXCPU
        | SIGXFSZ | SIGVTALRM | SIGPROF | SIGUSR1 | SIGUSR2 => {
            SIG_TERM_VAL.store(signo, Ordering::Relaxed);
            SIG_TERM.store(true, Ordering::Relaxed);
        }
        _ => {
            SIG_OTHER_VAL.store(signo, Ordering::Relaxed);
            SIG_OTHER.store(true, Ordering::Relaxed);
        }
    }
}
