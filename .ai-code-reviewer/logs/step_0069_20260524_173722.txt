struct TapFile {

impl TapFile {
    fn new(fd: RawFd, name: &str, destination: bool) -> Self {
        TapFile {
            fd,
            name: name.to_string(),
            _direction: if destination { Direction::DST } else { Direction::SRC },
        }
    }

    fn full_read(fd: RawFd, dst: &mut [u8]) -> ssize_t {
        let len = dst.len();
        let mut retval = unsafe { libc::read(fd, dst.as_mut_ptr() as *mut c_void, len) };
        if retval <= 0 || (retval as usize) == len {
            return retval;
        }
        let mut ptr = dst.as_mut_ptr().add(retval as usize);
        let mut left = len - (retval as usize);
        while left > 0 {
            retval = unsafe { libc::read(fd, ptr as *mut c_void, left) };
            if retval <= 0 {
                return retval;
            }
            left -= retval as usize;
            ptr = ptr.add(retval as usize);
        }
        len as ssize_t
    }

    fn read_blk(&mut self, dst: &mut [u8]) -> ssize_t {
        let len = dst.len();
        let mut lbuf = [0u8; 4];
        let nread = Self::full_read(self.fd, &mut lbuf);
        if nread == 0 {
            return 0;
        }

            eprintln!("tcopy: Corrupt tap-file, read hdr1={}", nread);
            process::exit(libc::EX_DATAERR);

                "tcopy: Corrupt tap-file, read hdr2={}, expected={}",
                nread, u

    }
