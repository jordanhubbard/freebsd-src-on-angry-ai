fn addgroup(grpname: &str) {
    unsafe {
        let pwd = &*PWD;
        let c_grpname = match CString::new(grpname) {
            Ok(s) => s,
            Err(_) => {
                warnx(&format!("{}: bad group name", grpname));
                return;
            }
        };

        let mut grp = getgrnam(c_grpname.as_ptr());
        if grp.is_null() {
            // Try as a group id
            match grpname.parse::<i64>() {
                Ok(lgid) if lgid > 0 => {
                    grp = getgrgid(lgid as gid_t);
                    if grp.is_null() {
                        warnx(&format!("{}: bad group name", grpname));
                        return;
                    }
                }
                _ => {
                    warnx(&format!("{}: bad group name", grpname));
                    return;
                }
            }
        }

        let grp = &*grp;

        // Check if user is a member of the group
        let mut dbmember = false;
        if pwd.pw_gid == grp.gr_gid {
            dbmember = true;
        }
        if !dbmember {
            let mut p = grp.gr_mem;
            while !p.is_null() {
                let mem = *p;
                if !mem.is_null() {
                    let mem_str = CStr::from_ptr(mem).to_string_lossy();
                    let name_str = CStr::from_ptr(pwd.pw_name).to_string_lossy();
                    if mem_str == name_str {
                        dbmember = true;
                        break;
                    }
                }
                p = std::ptr::add(p, 1);
            }
        }

        // If not a member and group has a password, prompt for it
        if !dbmember && !grp.gr_passwd.is_null() && (*CStr::from_ptr(grp.gr_passwd)).to_bytes().len() > 0 && getuid() != 0 {
            let prompt = CString::new("Password:").unwrap();
            let pass = getpass(prompt.as_ptr());
            if pass.is_null() {
                return;
            }
            let cryptpw = crypt(pass, grp.gr_passwd);
            if cryptpw.is_null() ||
                CStr::from_ptr(cryptpw).to_bytes() != CStr::from_ptr(grp.gr_passwd).to_bytes() {
                eprintln!("Sorry");
                return;
            }
        }

        let ngrps_max = sysconf(_SC_NGROUPS_MAX) as usize;
        let grps = malloc(std::mem::size_of::<gid_t>() * ngrps_max);
        if grps.is_null() {
            err(1, "malloc");
        }
        let ngrps = getgroups(ngrps_max as c_int, grps as *mut gid_t);
        if ngrps < 0 {
            warn("getgroups");
            free(grps);
            return;
        }
        let ngrps = ngrps as usize;

        let egid = getegid();

        // Read current groups into a slice
        let current_grps: Vec<gid_t> = std::slice::from_raw_parts(grps as *const gid_t, ngrps).to_vec();

        // Remove requested gid from supp. list if it exists and doesn't match prior egid
        if grp.gr_gid != egid && inarray(grp.gr_gid, &current_grps) {
            let mut new_grps: Vec<gid_t> = current_grps.iter().filter(|&&g| g != grp.gr_gid).collect();
            let new_ngrps = new_grps.len();
            priv_start();
            if setgroups(new_ngrps as c_int, new_grps.as_ptr()) < 0 {
                priv_end();
                warn("setgroups");
                free(grps);
                return;
            }
            priv_end();
        }

        // Set the new primary group
        priv_start();
        if setgid(grp.gr_gid) < 0 {
            priv_end();
            warn("setgid");
            free(grps);
            return;
        }
        priv_end();

        // Add old effective gid to supp. list if it does not exist
        if !inarray(egid, &current_grps) {
            if ngrps >= ngrps_max {
                warnx("too many groups");
            } else {
                let mut new_grps = current_grps.clone();
                new_grps.push(egid);
                let new_ngrps = new_grps.len();
                priv_start();
                if setgroups(new_ngrps as c_int, new_grps.as_ptr()) < 0 {
                    priv_end();
                    warn("setgroups");
                    free(grps);
                    return;
                }
                priv_end();
            }
        }

        free(grps);
    }
}

fn loginshell() {
    unsafe {
        let pwd = &*PWD;
        let mut shell = CStr::from_ptr(pwd.pw_shell).to_string_lossy().into_owned();
        if shell.is_empty() {
            shell = "/bin/sh".to_string();
        }
        let shell_cstr = CString::new(shell.as_bytes()).unwrap();

        let dir_cstr = CString::new(CStr::from_ptr(pwd.pw_dir).to_bytes()).unwrap();
        if chdir(dir_cstr.as_ptr()) < 0 {
            warn(CStr::from_ptr(pwd.pw_dir).to_string_lossy().as_ref());
            chdir(CString::new("/").unwrap().as_ptr());
        }

        let term = getenv(b"TERM\0".as_ptr() as *const c_char);
        let ticket = getenv(b"KRBTKFILE\0".as_ptr() as *const c_char);

        // Clear environment
        env::clear();

        let lc = login_getpwclass(pwd as *const passwd);
        setusercontext(lc, pwd as *const passwd, pwd.pw_uid, LOGIN_SETPATH | LOGIN_SETUMASK | LOGIN_SETENV);
        login_close(lc);

        setenv(b"USER\0".as_ptr() as *const c_char, pwd.pw_name, 1);
        setenv(b"SHELL\0".as_ptr() as *const c_char, CString::new(shell.as_bytes()).unwrap().as_ptr(), 1);
        setenv(b"HOME\0".as_ptr() as *const c_char, pwd.pw_dir, 1);

        if !term.is_null() {
            setenv(b"TERM\0".as_ptr() as *const c_char, term, 1);
        }
        if !ticket.is_null() {
            setenv(b"KRBTKFILE\0".as_ptr() as *const c_char, ticket, 1);
        }

        // Build "-<shell>" argument
        let login_shell = format!("-{}", shell);
        let login_shell_cstr = CString::new(login_shell.as_bytes()).unwrap();

        let mut args: [*const c_char; 2] = [login_shell_cstr.as_ptr(), std::ptr::null()];

        execv(shell_cstr.as_ptr(), args.as_ptr());
        err(1, &shell);
    }
}

fn doshell() {
    unsafe {
        let pwd = &*PWD;
        let mut shell = CStr::from_ptr(pwd.pw_shell).to_string_lossy().into_owned();
        if shell.is_empty() {
            shell = "/bin/sh".to_string();
        }
        let shell_cstr = CString::new(shell.as_bytes()).unwrap();
        execl(shell_cstr.as_ptr(), shell_cstr.as_ptr(), std::ptr::null::<c_char>());
        err(1, &shell);
    }
}

fn main() {
    let args: Vec<String> = env::args().collect();
    let argc = args.len() as c_int;
    let argv: Vec<CString> = args.iter().map(|s| CString::new(s.as_bytes()).unwrap()).collect();
    let argv_ptrs: Vec<*const c_char> = argv.iter().map(|s| s.as_ptr()).collect();

    unsafe {
        EUID = geteuid();
        if EUID != 0 {
            warnx("need root permissions to function properly, check setuid bit");
        }
        if seteuid(getuid()) < 0 {
            err(1, "seteuid");
        }

        PWD = getpwuid(getuid());
        if PWD.is_null() {
            errx(1, "unknown user");
        }

        let mut login = 0;
        let optstring = CString::new("-l").unwrap();
        let mut optind_val = 1;

        loop {
            let ch = getopt(argc, argv_ptrs.as_ptr(), optstring.as_ptr());
            if ch == -1 {
                break;
            }
            match ch as i32 {
                b'l' as i32 | (b'-') as i32 => {
                    login = 1;
                }
                _ => {
                    usage();
                }
            }
        }
        optind_val = optind();

        let remaining: Vec<String> = args[optind_val..].to_vec();

        match remaining.len() {
            0 => restoregrps(),
            1 => addgroup(&remaining[0]),
            _ => usage(),
        }

        if seteuid(EUID) < 0 {
            err(1, "seteuid");
        }
        if setuid(getuid()) < 0 {
            err(1, "setuid");
        }

        if login != 0 {
            loginshell();
        } else {
            doshell();
        }

        process::exit(1);
    }
}
