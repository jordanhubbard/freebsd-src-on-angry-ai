//! Rust replacement for green_saver.c - syscons green screensaver module.
//!
//! This is a kernel-adjacent screensaver module. The Rust translation provides
//! equivalent callback functions for init, term, and saver operations.

use std::os::raw::{c_int, c_void};

/// Opaque handle to the video adapter (mirrors video_adapter_t from syscons).
pub struct VideoAdapter(*mut c_void);

/// Display state constants matching V_DISPLAY_* from syscons.
pub const V_DISPLAY_ON: c_int = 0;
pub const V_DISPLAY_STAND_BY: c_int = 1;

/// Error code: no such device (ENODEV).
pub const ENODEV: c_int = 6;

/// FFI declaration for the video display blanking function from syscons.
extern "C" {
    fn vidd_blank_display(adp: *mut VideoAdapter, mode: c_int) -> c_int;
}

/// Blank or unblank the display.
///
/// When `blank` is non-zero, puts the display into standby.
/// When `blank` is zero, turns the display on.
/// Always returns 0 on success.
#[no_mangle]
pub unsafe extern "C" fn green_saver(adp: *mut VideoAdapter, blank: c_int) -> c_int {
    let mode = if blank != 0 { V_DISPLAY_STAND_BY } else { V_DISPLAY_ON };
    vidd_blank_display(adp, mode);
    0
}

/// Initialize the green saver.
///
/// Returns 0 on success, ENODEV if the display cannot be turned on.
#[no_mangle]
pub unsafe extern "C" fn green_init(adp: *mut VideoAdapter) -> c_int {
    if vidd_blank_display(adp, V_DISPLAY_ON) == 0 {
        0
    } else {
        ENODEV
    }
}

/// Terminate the green saver.
///
/// Always returns 0.
#[no_mangle]
pub unsafe extern "C" fn green_term(_adp: *mut VideoAdapter) -> c_int {
    0
}
