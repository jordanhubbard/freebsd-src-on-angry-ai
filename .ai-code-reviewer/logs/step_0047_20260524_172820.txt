struct TapeDev {
    fd: RawFd,
    name: String,
    direction: Direction,
    max_read_size: usize,
}

impl TapeDev {
    fn new(fd: RawFd, name: &str, destination: bool) -> Self {
        TapeDev {
            fd,
            name: name.to_string(),
            direction: if destination { Direction::DST } else { Direction::SRC },
            max_read_size: 0,
        }
    }

    fn read_blk(&mut self, dst: &mut [u8]) -> ssize_t {
        let len = dst.len();
        let mut retval: ssize_t = -1;

        if self.max_read_size == 0 {
            let mut max_read_size = len;
            while max_read_size > 0 {
                let buf = &mut dst[..max_read_size];
                match unsafe { libc::read(self.fd, buf.as_mut_ptr() as *mut c_void, max_read_size) } {
                    r if r >= 0 => {
                        retval = r;
                        break;
                    }
                    _ if unsafe { io::Error::last_os_error().raw_os_error() } != Some(libc::EINVAL)
                        && unsafe { io::Error::last_os_error().raw_os_error() } != Some(libc::EFBIG) =>
                    {
                        retval = unsafe { libc::read(self.fd, buf.as_mut_ptr() as *mut c_void, max_read_size) };
                        break;
                    }
                    _ => {}
                }
                if max_read_size < 512 {
                    eprintln!("tcopy: Cannot find a sane max blocksize");
                    process::exit(1);
                }
                // Reduce to next lower power of two
                let i = (max_read_size - 1).next_power_of_two().trailing_zeros() as usize;
                max_read_size = 1usize << (i - 1);
            }
            self.max_read_size = max_read_size;
        } else {
            let buf = &mut dst[..self.max_read_size];
            retval = unsafe { libc::read(self.fd, buf.as_mut_ptr() as *mut c_void, self.max_read_size) };
        }

        if retval < 0 {
            let err = io::Error::last_os_error();
            eprintln!(
                "tcopy: read error, {}, file {}, record {}: {}",
                self.name, unsafe { FILEN }, unsafe { RECORD }, err
            );
            process::exit(1);
        }
        retval
    }

    fn verify_blk(&mut self, dst: &mut [u8], _expected: usize) -> ssize_t {
        self.read_blk(dst)
    }

    fn write_blk(&self, src: &[u8]) {
        assert!(!src.is_empty());
        let nwrite = unsafe { libc::write(self.fd, src.as_ptr() as *const c_void, src.len()) };
        if nwrite < 0 || (nwrite as usize) != src.len() {
            if nwrite == -1 {
                let err = io::Error::last_os_error();
                eprintln!(
                    "tcopy: write error, file {}, record {}: {}",
                    unsafe { FILEN }, unsafe { RECORD }, err
                );
            } else {
                eprintln!(
                    "tcopy: write error, file {}, record {}",
                    unsafe { FILEN }, unsafe { RECORD }
                );
                eprintln!(
                    "tcopy: write ({}) != read ({})",
                    nwrite,
                    src.len()
                );
            }
            eprintln!("tcopy: copy aborted");
            process::exit(5);
        }
    }

    fn file_mark(&self) {
        let mut op: libc::mtop = unsafe { std::mem::zeroed() };
        op.mt_op = libc::MTWEOF as c_int;
        op.mt_count = 1 as daddr_t;
        if unsafe { libc::ioctl(self.fd, libc::MTIOCTOP, &mut op) } < 0 {
            let err = io::Error::last_os_error();
            eprintln!("tcopy: tape op (write file mark): {}", err);
            process::exit(6);
        }
    }

    fn rewind(&self) {
        let mut op: libc::mtop = unsafe { std::mem::zeroed() };
        op.mt_op = libc::MTREW as c_int;
        op.mt_count = 1 as daddr_t;
        if unsafe { libc::ioctl(self.fd, libc::MTIOCTOP, &mut op) } < 0 {
            let err = io::Error::last_os_error();
            eprintln!("tcopy: tape op (rewind): {}", err);
            process::exit(6);
        }
    }
}

    max_read_size: usize,
}

impl TapFile {
    fn new(fd: RawFd, name: &str, destination: bool) -> Self {
        TapFile {
            fd,
            name: name.to_string(),
            direction: if destination { Direction::DST } else { Direction::SRC },
            max_read_size: 0,
        }
    }

    fn full_read(fd: RawFd, dst: &mut [u8]) -> ssize_t {
        let len = dst.len();
        let mut retval = unsafe { libc::read(fd, dst.as_mut_ptr() as *mut c_void, len) };
        if retval <= 0 || (retval as usize) == len {
            return retval;
        }

        let mut ptr = dst.as_mut_ptr().add(retval as usize);
        let mut left = len - (retval as usize);
        while left > 0 {
            retval = unsafe { libc::read(fd, ptr as *mut c_void, left) };
            if retval <= 0 {
                return retval;
            }
            left -= retval as usize;
            ptr = ptr.add(retval as usize);
        }
        len as ssize_t
    }
