struct TapeDev {
    fd: RawFd,
    name: String,
    _direction: Direction,
    max_read_size: usize,
}

impl TapeDev {
    fn new(fd: RawFd, name: &str, dst: bool) -> Self {
        TapeDev { fd, name: name.to_string(), _direction: if dst { Direction::DST } else { Direction::SRC }, max_read_size: 0 }
    }

    fn read_blk(&mut self, dst: &mut [u8]) -> ssize_t {
        let len = dst.len();
        let mut retval: ssize_t = -1;
        if self.max_read_size == 0 {
            let mut mrs = len;
            while mrs > 0 {
                retval = unsafe { libc::read(self.fd, dst.as_mut_ptr() as *mut c_void, mrs) };
                if retval >= 0 { break; }
                let err = io::Error::last_os_error().raw_os_error().unwrap_or(-1);
                if err != libc::EINVAL && err != libc::EFBIG { break; }
                if mrs < 512 { eprintln!("tcopy: Cannot find a sane max blocksize"); process::exit(1); }
                let i = (mrs - 1).next_power_of_two().trailing_zeros() as usize;
                mrs = 1usize << (i - 1);
            }
            self.max_read_size = mrs;
        } else {
            retval = unsafe { libc::read(self.fd, dst.as_mut_ptr() as *mut c_void, self.max_read_size) };
        }
        if retval < 0 {
            let err = io::Error::last_os_error();
            eprintln!("tcopy: read error, {}, file {}, record {}: {}", self.name, unsafe { FILEN }, unsafe { RECORD }, err);
            process::exit(1);
        }
        retval
    }

    fn verify_blk(&mut self, dst: &mut [u8], _expected: usize) -> ssize_t { self.read_blk(dst) }

    fn write_blk(&self, src: &[u8]) {
        assert!(!src.is_empty());
        let nwrite = unsafe { libc::write(self.fd, src.as_ptr() as *const c_void, src.len()) };
        if nwrite < 0 || (nwrite as usize) != src.len() {
            if nwrite == -1 {
                eprintln!("tcopy: write error, file {}, record {}: {}", unsafe { FILEN }, unsafe { RECORD }, io::Error::last_os_error());
            } else {
                eprintln!("tcopy: write error, file {}, record {}", unsafe { FILEN }, unsafe { RECORD });
                eprintln!("tcopy: write ({}) != read ({})", nwrite, src.len());
            }
            eprintln!("tcopy: copy aborted");
            process::exit(5);
        }
    }

    fn file_mark(&self) {
        let mut op: libc::mtop = unsafe { std::mem::zeroed() };
        op.mt_op = libc::MTWEOF as c_int;
        op.mt_count = 1 as daddr_t;
        if unsafe { libc::ioctl(self.fd, libc::MTIOCTOP, &mut op) } < 0 {
            eprintln!("tcopy: tape op (write file mark): {}", io::Error::last_os_error());
            process::exit(6);
        }
    }

    fn rewind(&self) {
        let mut op: libc::mtop = unsafe { std::mem::zeroed() };

            eprintln!("tcopy: tape op (rewind): {}", io::Error::last_os_error());
            process::exit(6);
        }
    }
}

struct TapFile {
    fd: RawFd,
    name: String,
    _direction: Direction,
}

impl TapFile {
    fn new(fd: RawFd, name: &str, dst: bool) -> Self {
        TapFile { fd, name: name.to_string(), _direction: if dst { Direction::DST } else { Direction::SRC } }
    }

    fn full_read(fd: RawFd, dst: &mut [u8]) -> ssize_t {
        let len = dst.len();
        let mut retval = unsafe { libc::read(fd, dst.as_mut_ptr() as *mut c_void, len) };
        if retval <= 0 || (retval as usize) == len { return retval; }
        let mut ptr = dst.as_mut_ptr().add(retval as usize);
        let mut left = len - (retval as usize);
        while left > 0 {
            retval = unsafe { libc::read(fd, ptr as *mut c_void, left) };
            if retval <= 0 { return retval; }
            left -= retval as usize;
            ptr = ptr.add(retval as usize);
        }
        len as ssize_t
    }

    fn read_blk(&mut self, dst: &mut [u8]) -> ssize_t {
        let len = dst.len();
        let mut lbuf = [0u8; 4];
        let nread = Self::full_read(self.fd, &mut lbuf);
        if nread == 0 { return 0; }
        if (nread as usize) != 4 { eprintln!("tcopy: Corrupt tap-file, read hdr1={}", nread); process::exit(libc::EX_DATAERR); }
        let u = u32::from_le_bytes(lbuf);
        if u == 0 || (u >> 24) == 0xff { return 0; }
        if (u as usize) > len { eprintln!("tcopy: tapfile blocksize too big, 0x{:08x}", u); process::exit(17); }
        let nread = Self::full_read(self.fd, &mut dst[..u as usize]);
        if (nread as usize) != u as usize { eprintln!("tcopy: Corrupt tap-file, read hdr2={}, expected={}", nread, u); process::exit(libc::EX_DATAERR); }
        u as ssize_t
    }

    fn verify_blk(&mut self, dst: &mut [u8], _expected: usize) -> ssize_t { self.read_blk(dst) }

    fn write_blk(&self, src: &[u8]) {
        let hdr = (src.len() as u32).to_le_bytes();
        let nwrite = unsafe { libc::write(self.fd, hdr.as_ptr() as *const c_void, 4) };
        if nwrite != 4 { eprintln!("tcopy: write error, file {}, record {}", unsafe { FILEN }, unsafe { RECORD }); process::exit(5); }
        let nwrite = unsafe { libc::write(self.fd, src.as_ptr() as *const c_void, src.len()) };
        if nwrite < 0 || (nwrite as usize) != src.len() {
            if nwrite == -1 { eprintln!("tcopy: write error, file {}, record {}: {}", unsafe { FILEN }, unsafe { RECORD }, io::Error::last_os_error()); }
            else { eprintln!("tcopy: write error, file {}, record {}", unsafe { FILEN }, unsafe { RECORD }); eprintln!("tcopy: write ({}) != read ({})", nwrite, src.len()); }
            eprintln!("tcopy: copy aborted");
            process::exit(5);
        }
    }

    fn file_mark(&self) {
        let mark = 0xffu32.to_le_bytes();
        let nwrite = unsafe { libc::write(self.fd, mark.as_ptr() as *const c_void, 4) };
        if nwrite != 4 { eprintln!("tcopy: write error, file {}, record {}", unsafe { FILEN }, unsafe { RECORD }); process::exit(5); }
    }

    fn rewind(&self) {
        if unsafe { libc::lseek(self.fd, 0, libc::SEEK_SET) } < 0 {
            eprintln!("tcopy: tape op (rewind): {}", io::Error::last_os_error());
            process::exit(6);
        }
    }
}

enum Tape { Dev(TapeDev), File(TapFile) }

impl Tape {
    fn read_blk(&mut self, dst: &mut [u8]) -> ssize_t { match self { Tape::Dev(d) => d.read_blk(dst), Tape::File(f) => f.read_blk(dst) } }
    fn verify_blk(&mut self, dst: &mut [u8], expected: usize) -> ssize_t { match self { Tape::Dev(d) => d.verify_blk(dst, expected), Tape::File(f) => f.verify_blk(dst, expected) } }
    fn write_blk(&self, src: &[u8]) { match self { Tape::Dev(d) => d.write_blk(src), Tape::File(f) => f.write_blk(src) } }
    fn file_mark(&self) { match self { Tape::Dev(d) => d.file_mark(), Tape::File(f) => f.file_mark() } }
    fn rewind(&self) { match self { Tape::Dev(d) => d.rewind(), Tape::File(f) => f.rewind() } }
    fn set_direction(&mut self, dir: Direction) { match self { Tape::Dev(d) => d._direction = dir, Tape::File(f) => f._direction = dir } }
}
