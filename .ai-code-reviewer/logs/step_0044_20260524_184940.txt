// Global state
static mut G_FILN: i32 = 0;
static mut G_RECORD: u64 = 0;
static mut G_MAXBLK: i64 = 0;

static mut G_LASTNREAD: i64 = NOCOUNT;
static mut G_LASTREC: u64 = 0;
static mut G_FSIZE: u64 = 0;

static mut G_TEND: Option<Instant> = None;

struct TapeDev {
    fd: RawFd,
    name: String,
    direction: Direction,
    max_read_size: usize,
    file: std::fs::File,
}

impl TapeDev {
    fn new(fd: RawFd, name: &str, destination: bool) -> Self {
        TapeDev {
            fd,
            name: name.to_string(),
            direction: if destination { Direction::Dst } else { Direction::Src },
            max_read_size: 0,
            file: unsafe { std::fs::File::from_raw_fd(fd) },
        }
    }

    fn read_blk(&mut self, dst: &mut [u8]) -> std::io::Result<i64> {
        let len = dst.len();
        let mut max_read_size = if self.max_read_size == 0 { len } else { self.max_read_size };

        let mut retval: i64 = -1;
        if self.max_read_size == 0 {
            while max_read_size > 0 {
                let buf = &mut dst[..max_read_size];
                match self.file.read(buf) {
                    Ok(n) => {
                        retval = n as i64;
                        if retval >= 0 {
                            self.max_read_size = max_read_size;
                            break;
                        }
                    }
                    Err(e) => {
                        if e.kind() != io::ErrorKind::InvalidInput {
                            self.max_read_size = max_read_size;
                            break;
                        }
                        if max_read_size < 512 {
                            eprintln!("tcopy: Cannot find a sane max blocksize");
                            process::exit(1);
                        }
                        // Reduce to next lower power of two
                        let i = (max_read_size - 1).next_power_of_two().trailing_zeros() as usize;
                        max_read_size = 1usize.saturating_shl(i.saturating_sub(1));
                    }
                }
            }
        } else {
            let buf = &mut dst[..self.max_read_size];
            retval = self.file.read(buf)? as i64;
        }

        if retval < 0 {
            unsafe {
                eprintln!(
                    "tcopy: read error, {}, file {}, record {}",
                    self.name, G_FILN, G_RECORD
                );
            }
            process::exit(1);
        }
        Ok(retval)
    }

    fn verify_blk(&mut self, dst: &mut [u8], _expected: usize) -> std::io::Result<i64> {
        self.read_blk(dst)
    }

    fn write_blk(&mut self, src: &[u8]) {
        assert!(!src.is_empty());
        match self.file.write_all(src) {
            Ok(()) => {}
            Err(e) => {
                unsafe {
                    eprintln!(
                        "tcopy: write error, file {}, record {}",
                        G_FILN, G_RECORD
                    );
                }
                eprintln!("tcopy: {}", e);
                eprintln!("tcopy: copy aborted");
                process::exit(5);
            }
        }
    }

    fn file_mark(&mut self) {
        // Write file mark using ioctl MTIOCTOP with MTWEOF
        let mut op: mtio_mtop = unsafe { std::mem::zeroed() };
        op.mt_op = MTWEOF;
        op.mt_count = 1;
        let ret = unsafe {
            libc::ioctl(self.fd, libc::MTIOCTOP as _, &mut op as *mut _ as *mut libc::c_void)
        };
        if ret < 0 {
            eprintln!("tcopy: tape op (write file mark): {}", io::Error::last_os_error());
            process::exit(6);
        }
    }

    fn rewind(&mut self) {
        let mut op: mtio_mtop = unsafe { std::mem::zeroed() };
        op.mt_op = MTREW;
        op.mt_count = 1;
        let ret = unsafe {
            libc::ioctl(self.fd, libc::MTIOCTOP as _, &mut op as *mut _ as *mut libc::c_void)
        };
        if ret < 0 {
            eprintln!("tcopy: tape op (rewind): {}", io::Error::last_os_error());
            process::exit(6);
        }
    }
}

impl Drop for TapeDev {
    fn drop(&mut self) {
        // Prevent double-free since we took ownership of the fd
        std::ptr::null_mut();
    }
}

struct TapFile {
    base: TapeDev,
}

impl TapFile {
    fn new(fd: RawFd, name: &str, dst: bool) -> Self {
        TapFile {
            base: TapeDev::new(fd, name, dst),
        }
    }

    fn read_blk(&mut self, dst: &mut [u8]) -> std::io::Result<i64> {
        let mut lbuf = [0u8; 4];
        let nread = full_read(&mut self.base.file, &mut lbuf)?;
        if nread == 0 {
            return Ok(0);
        }
        if nread != 4 {
            eprintln!("tcopy: Corrupt tap-file, read hdr1={}", nread);
            process::exit(17);
        }
        let u = u32::from_le_bytes(lbuf) as u64;
        if u == 0 || (u >> 24) == 0xff {
            return Ok(0);
        }
        if u as usize > dst.len() {
            eprintln!("tcopy: tapfile blocksize too big, 0x{:08x}", u);
            process::exit(17);
        }
        let data_len = u as usize;
        let n = self.base.file.read(&mut dst[..data_len])?;
        Ok(n as i64)
    }

    fn write_blk(&mut self, src: &[u8]) {
        let len = src.len() as u32;
        let hdr = len.to_le_bytes();
        self.base.file.write_all(&hdr).unwrap();
        self.base.file.write_all(src).unwrap();
    }

    fn file_mark(&mut self) {
        // Write EOF marker: 0xffffffff
        let eof = 0xffffffffu32.to_le_bytes();
        self.base.file.write_all(&eof).unwrap();
    }

    fn rewind(&mut self) {
        self.base.file.seek(SeekFrom::Start(0)).unwrap();
    }
}

fn full_read<R: Read>(r: &mut R, dst: &mut [u8]) -> std::io::Result<usize> {
    let mut total = 0;
    while total < dst.len() {
        match r.read(&mut dst[total..]) {
            Ok(0) => return Ok(total),
            Ok(n) => total += n,
            Err(e) => return Err(e),
        }
    }
    Ok(total)
}

fn report_total<W: io::Write>(mut w: W) {
    unsafe {
        let t_start = G_TSTART.unwrap();
        let t_end = G_TEND.unwrap();
        let dur = t_end.duration_since(t_start).as_secs_f64();
        let tot = G_TSIZE + G_FSIZE;
        write!(w, "total length: {} bytes", tot).unwrap();
        write!(w, " time: {:.0} s", dur).unwrap();
        let tot_kb = tot / 1024;
        write!(w, " rate: {:.1} kB/s", (tot_kb as f64) / dur).unwrap();
        writeln!(w).unwrap();
    }
}

fn progress(nread: i64) {
    unsafe {
        if nread != G_LASTNREAD {
            if G_LASTNREAD != 0 && G_LASTNREAD != NOCOUNT {
                if G_LASTREC == 0 && nread == 0 {
                    write!(G_MSG.unwrap(), "{} records\n", G_RECORD).unwrap();
                } else if G_RECORD - G_LASTREC > 1 {
                    write!(
                        G_MSG.unwrap(),
                        "records {} to {}\n",
                        G_LASTREC, G_RECORD
                    )
                    .unwrap();
                } else {
                    write!(G_MSG.unwrap(), "record {}\n", G_LASTREC).unwrap();
                }
            }
            if nread != 0 {
                write!(G_MSG.unwrap(), "file {}: block size {} ", G_FILN, nread).unwrap();
                G_MSG.unwrap().flush().unwrap();
            }
            G_LASTREC = G_RECORD;
        }
        if nread > 0 {
            G_FSIZE += nread as u64;
            G_RECORD += 1;
        } else {
            if G_LASTNREAD <= 0 && G_LASTNREAD != NOCOUNT {
                write!(G_MSG.unwrap(), "eot\n").unwrap();
                return;
            }
            write!(
                G_MSG.unwrap(),
                "file {}: eof after {} records: {} bytes\n",
                G_FILN, G_RECORD, G_FSIZE
            )
            .unwrap();
            G_FILN += 1;
            G_TSIZE += G_FSIZE;
            G_FSIZE = 0;
            G_RECORD = 0;
            G_LASTREC = 0;
            G_LASTNREAD = 0;
        }
        G_LASTNREAD = nread;
    }
}

fn check_want_info() {
    if WANT_INFO.swap(false, Ordering::SeqCst) {
        unsafe {
            G_TEND = Some(Instant::now());
            eprint!("tcopy: file {} record {} ", G_FILN, G_RECORD);
            report_total(io::stderr());
        }
    }
}

fn expand_number(s: &str) -> Option<i64> {
    let s = s.trim();
    let mut multiplier: i64 = 1;
    let (num_str, suffix) = if let Some(pos) = s.find(|c: char| c.is_ascii_alphabetic()) {
        (&s[..pos], &s[pos..])
    } else {
        (s, "")
    };

    match suffix {
        "" | "b" => multiplier = 1,
        "k" => multiplier = 1024,
        "m" => multiplier = 1024 * 1024,
        "g" => multiplier = 1024 * 1024 * 1024,
        "t" => multiplier = 1024 * 1024 * 1024 * 1024,
        _ => return None,
    }

    num_str.parse::<i64>().ok().map(|n| n * multiplier)
}

fn open_arg(name: &str, is_output: bool, rawfile: bool) -> Box<dyn TapeOps> {
    let flags = if is_output {
        libc::O_WRONLY | libc::O_CREAT | libc::O_TRUNC
    } else {
        libc::O_RDONLY
    };
    let mode = libc::S_IRUSR | libc::S_IWUSR | libc::S_IRGRP | libc::S_IWGRP | libc::S_IROTH | libc::S_IWOTH;

    let c_name = CString::new(name).unwrap();
    let fd = unsafe { libc::open(c_name.as_ptr(), flags, mode) };
    if fd < 0 {
        eprintln!("tcopy: {}: {}", name, io::Error::last_os_error());
        process::exit(1);
    }

    // Check if it's a regular file (tap file format)
    let mut statbuf: libc::stat = unsafe { std::mem::zeroed() };
    if unsafe { libc::fstat(fd, &mut statbuf) } == 0 && (statbuf.st_mode & libc::S_IFMT) == libc::S_IFREG {
        if rawfile || is_output {
            return Box::new(TapFile::new(fd, name, is_output));
        }
        return Box::new(TapFile::new(fd, name, is_output));
    }

    Box::new(TapeDev::new(fd, name, is_output))
}
