# $FreeBSD$
#
# Rust replacement for CheckerDependencyHandling clang plugin

PROG=	CheckerDependencyHandlingAnalyzerPlugin
SRCS=	lib.rs

CARGO_DIR=	${.CURDIR}
CARGO_TARGET_DIR?=	${OBJDIR}/cargo-target

# Build the Rust cdylib
all: ${PROG}.so

${PROG}.so: ${CARGO_DIR}/Cargo.toml ${CARGO_DIR}/src/lib.rs
	@echo "Building Rust CheckerDependencyHandling plugin..."
	@mkdir -p ${CARGO_TARGET_DIR}
	cargo build --manifest-path ${CARGO_DIR}/Cargo.toml \
	    --target-dir ${CARGO_TARGET_DIR} --release
	@cp ${CARGO_TARGET_DIR}/release/libchecker_dependency_handling.so ${.OBJDIR}/${PROG}.so

clean:
	@rm -f ${.OBJDIR}/${PROG}.so
