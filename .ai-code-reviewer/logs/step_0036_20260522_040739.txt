type XdrProc = extern "C" fn(*mut std::os::raw::c_void, *mut std::os::raw::c_void) -> c_int;

extern "C" {
    fn yp_match(
        domain: *const c_char,
        map: *const c_char,
        key: *const c_char,
        key_len: c_int,

}

fn usage() -> ! {
    eprintln!(
        "{}\n{}",
        "usage: ypmatch [-kt] [-d domainname] key ... mapname",
        "       ypmatch -x"
    );
    process::exit(1);
}

fn main() {
    let args: Vec<String> = std::env::args().collect();
    let mut domainname: *mut c_char = std::ptr::null_mut();
    let mut notrans: bool = false;
    let mut key: bool = false;
    let mut show_aliases: bool = false;

    let mut optind: usize = 1;
    while optind < args.len() {
        let arg = &args[optind];
        if arg.starts_with('-') && arg.len() > 1 {
            if arg == "-x" {
                show_aliases = true;
                optind += 1;
                continue;
            }
            if arg == "-t" {
                notrans = true;
                optind += 1;
                continue;
            }
            if arg == "-k" {
                key = true;
                optind += 1;
                continue;
            }
            if arg.starts_with("-d") {
                if arg.len() > 2 {
                    let d = CString::new(&arg[2..]).unwrap();
                    domainname = d.into_raw();
                } else if optind + 1 < args.len() {
                    optind += 1;
                    let d = CString::new(&args[optind]).unwrap();
                    domainname = d.into_raw();
                }
                optind += 1;
                continue;
            }
            usage();
        } else {
            break;
        }
    }

    if show_aliases {
        for a in YPALIASES {
            println!("Use \"{}\" for \"{}\"", a.alias, a.name);
        }
        process::exit(0);
    }

    // We need at least key(s) + mapname after options
    if args.len() - optind < 2 {
        usage();
    }

    if domainname.is_null() {
        let ret = unsafe { yp_get_default_domain(&mut domainname) };
        if ret != 0 {
            eprintln!("can't get default domain");
            process::exit(1);
        }
    }

    let inmap_cstr = CString::new(args[args.len() - 1].as_str()).unwrap();
    let mut inmap_ptr = inmap_cstr.as_ptr();

    if !notrans {
        for a in YPALIASES {
            if args[args.len() - 1] == a.alias {
                inmap_ptr = CString::new(a.name).unwrap().as_ptr();
                break;
            }
        }
    }

    let mut rval: i32 = 0;
    for i in optind..args.len() - 1 {
        let inkey = &args[i];
        let inkey_cstr = CString::new(inkey.as_bytes()).unwrap();
        let key_len = inkey.len() as c_int;

        let mut outbuf: *mut c_char = std::ptr::null_mut();
        let mut outbuflen: c_int = 0;

        let r = unsafe {
            yp_match(domainname, inmap_ptr, inkey_cstr.as_ptr(), key_len, &mut outbuf, &mut outbuflen)
        };

        match r {
            0 => {
                if key {
                    print!("{}: ", inkey);
                }
                let val = unsafe {
                    std::str::from_utf8_unchecked(std::slice::from_raw_parts(outbuf as *const u8, outbuflen as usize))
                };
                println!("{}", val);
                unsafe {
                    xdr_free(xdr_string, outbuf as *mut std::os::raw::c_void);
                }
            }
            YPERR_YPBIND => {
                eprintln!("not running ypbind");
                process::exit(1);
            }
            _ => {
                let errstr = unsafe {
                    CStr::from_ptr(yperr_string(r)).to_string_lossy().into_owned()
                };
                eprintln!(
                    "can't match key {} in map {}. reason: {}",
                    inkey,
                    unsafe { CStr::from_ptr(inmap_ptr).to_string_lossy() },
                    errstr
                );
                rval = 1;
            }
        }
    }
    process::exit(rval);
}
