fn warnx(msg: &str) {
    let _ = io::stderr().write_all(format!("{}\n", msg).as_bytes());
}

fn err(code: c_int, msg: &str) -> ! {
    let err = get_errno();
    let err_str = io::Error::from_raw_os_error(err).to_string();
    let _ = io::stderr().write_all(format!("{}: {}\n", msg, err_str).as_bytes());
    process::exit(code);
}

fn str2pid(s: &str) -> Option<c_int> {
    match s.parse::<i64>() {
        Ok(v) => {
            if v < i32::MIN as i64 || v > i32::MAX as i64 {
                warnx("non-numeric pid");
                None
            } else {
                Some(v as c_int)
            }
        }
        Err(_) => {
            warnx("non-numeric pid");
            None
        }
    }
}

fn usage() -> ! {
    let stderr = io::stderr();
    let mut out = stderr.lock();
    let _ = out.write_all(b"Usage:\n");
    let _ = out.write_all(b"    proccontrol -m mode -s (enable|disable) ");
    let _ = out.write_all(b"(-p pid | command)\n");
    let _ = out.write_all(b"    proccontrol -m mode -q [-p pid]\n");
    let _ = out.write_all(b"Modes: ");
    for (i, m) in MODES.iter().enumerate() {
        if i > 0 {
            let _ = out.write_all(b"|");
        }
        let _ = out.write_all(m.name.as_bytes());
    }
    let _ = out.write_all(b"\n");
    process::exit(1);
}

fn find_mode(name: &str) -> Option<Mode> {
    for m in MODES {
        if m.name == name {
            return Some(m.mode);
        }
    }
    None
}
