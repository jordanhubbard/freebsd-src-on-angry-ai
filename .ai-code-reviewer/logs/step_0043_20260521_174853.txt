const AT_FDCWD: i32 = -100;
const AT_SYMLINK_NOFOLLOW: i32 = 0x200;

static SIGINFO: AtomicBool = AtomicBool::new(false);

fn usage() -> ! {
    let _ = io::stderr().write_all(

fn parse_flags(flags_str: &str) -> Result<(u32, u32, bool), String> {
    let cstr = CString::new(flags_str).map_err(|_| "invalid flags".to_string())?;
    if flags_str.starts_with(|c: char| c.is_ascii_digit() && c >= '0' && c <= '7') {
        let val = u32::from_str_radix(flags_str, 8)
            .map_err(|_| format!("invalid flags: {}", flags_str))?;
        // Check for trailing non-digit characters
        if flags_str.chars().any(|c| !('0'..='7').contains(&c)) {
            return Err(format!("invalid flags: {}", flags_str));
        }
        Ok((val, 0, true))
    } else {
        let mut set: u64 = 0;
        let mut clear: u64 = 0;
        unsafe {
            if strtofflags(cstr.as_ptr(), &mut set, &mut clear) != 0 {
                return Err(format!("invalid flag: {}", flags_str));
            }
        }
        Ok((set as u32, (!clear) as u32, false))
    }
}

fn chflags_for_path(
    path: &Path,
    newflags: u32,
    hflag: bool,
    vflag: usize,
) -> Result<(), i32> {
    let cpath = CString::new(path.as_os_str().as_encoded_bytes())
        .map_err(|_| io::Error::from(io::ErrorKind::InvalidInput).raw_os_error().unwrap_or(1))?;

    let atflag = if hflag {
        AT_SYMLINK_NOFOLLOW
    } else {
        0
    };

    unsafe {
        if chflagsat(AT_FDCWD, cpath.as_ptr(), newflags, atflag) == -1 {
            return Err(io::Error::last_os_error().raw_os_error().unwrap_or(1));
        }
    }
    Ok(())
}

fn process_path(
    path: &Path,
    set: u32,
    clear: u32,
    oct: bool,
    hflag: bool,
    fflag: bool,
    vflag: usize,
) -> i32 {
    let meta = if hflag {
        fs::symlink_metadata(path)
    } else {
        fs::metadata(path)
    };

    let meta = match meta {
        Ok(m) => m,
        Err(e) => {
            if !fflag {
                eprintln!("chflags: {}: {}", path.display(), e);
            }
            return 1;
        }
    };

    let oldflags = meta.st_flags() as u32;
    let newflags = if oct { set } else { (oldflags | set) & clear };

    if newflags == oldflags {
        return 0;
    }

    match chflags_for_path(path, newflags, hflag, vflag) {
        Ok(()) => {
            if vflag > 0 || SIGINFO.load(Ordering::Relaxed) {
                let mut stdout = io::stdout().lock();
                let _ = write!(stdout, "{}", path.display());
                if vflag > 1 || SIGINFO.load(Ordering::Relaxed) {
                    let _ = write!(
                        stdout,
                        ": 0{:o} -> 0{:o}",
                        oldflags, newflags
                    );
                }
                let _ = writeln!(stdout);
            }
            0
        }
        Err(e) => {
            if !fflag {
                eprintln!(
                    "chflags: {}: {}",
                    path.display(),
                    io::Error::from_raw_os_error(e)
                );
            }
            if SIGINFO.load(Ordering::Relaxed) {
                let mut stdout = io::stdout().lock();
                let _ = writeln!(
                    stdout,
                    "{}: {}",
                    path.display(),
                    io::Error::from_raw_os_error(e)
                );
                SIGINFO.store(false, Ordering::Relaxed);
            }
            1
        }
    }
}

fn process_recursive(
    path: &Path,
    set: u32,
    clear: u32,
    oct: bool,
    hflag: bool,
    fflag: bool,
    vflag: usize,
    xflag: bool,
    follow_links: bool,
) -> i32 {
    let mut rval = 0;

    // Process the entry itself
    rval |= process_path(path, set, clear, oct, hflag, fflag, vflag);

    // If it's a directory, recurse
    let meta = match fs::symlink_metadata(path) {
        Ok(m) => m,
        Err(_) => return rval,
    };

    if meta.is_dir() {
        let entries = match fs::read_dir(path) {
            Ok(e) => e,
            Err(e) => {
                eprintln!("chflags: {}: {}", path.display(), e);
                rval = 1;
                return rval;
            }
        };

        for entry in entries {
            let entry = match entry {
                Ok(e) => e,
                Err(e) => {
                    eprintln!("chflags: {}: {}", path.display(), e);
                    rval = 1;
                    continue;
                }
            };

            let entry_path = entry.path();

            // Check xflag: don't cross device boundaries
            if xflag {
                let entry_meta = match fs::symlink_metadata(&entry_path) {
                    Ok(m) => m,
                    Err(_) => continue,
                };
                if entry_meta.dev() != meta.dev() {
                    continue;
                }
            }

            // If follow_links is true (FTS_LOGICAL), follow symlinks
            if follow_links {
                rval |= process_recursive(
                    &entry_path, set, clear, oct, hflag, fflag, vflag, xflag, follow_links,
                );
            } else {
                // Physical mode: don't follow symlinks for recursion
                let entry_meta = match fs::symlink_metadata(&entry_path) {
                    Ok(m) => m,
                    Err(_) => continue,
                };
                if entry_meta.is_dir() {
                    rval |= process_recursive(
                        &entry_path, set, clear, oct, hflag, fflag, vflag, xflag, false,
                    );
                } else {
                    rval |= process_path(&entry_path, set, clear, oct, hflag, fflag, vflag);
                }
            }
        }
    }

    rval
}

        if Rflag {
            rval |= process_recursive(
                path, set, clear, oct, hflag, fflag, vflag, xflag, follow_links,
            );
        } else {
            rval |= process_path(path, set, clear, oct, hflag, fflag, vflag);
        }
    }

    process::exit(rval);
