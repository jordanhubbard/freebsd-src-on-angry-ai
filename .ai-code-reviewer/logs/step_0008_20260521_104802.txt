const MAXHOSTNAMELEN: usize = 64;

extern "C" {
    fn getdomainname(name: *mut c_char, len: usize) -> i32;
    fn setdomainname(name: *const c_char, len: usize) -> i32;
}

    process::exit(1);

    if !args.is_empty() {

            usage();

            Ok(c) => c,
            Err(_) => usage(),
        };

}

fn nix_error() -> String {
    // Simple errno-based error string without external crates
    // We use libc's strerror via FFI
    unsafe {
        let err = libc_errno();
        let msg = libc_strerror(err);
        if msg.is_null() {
            format!("errno {}", err)
        } else {
            std::ffi::CStr::from_ptr(msg).to_string_lossy().into_owned()
        }
    }
}

    fn strerror(errnum: i32) -> *const c_char;
