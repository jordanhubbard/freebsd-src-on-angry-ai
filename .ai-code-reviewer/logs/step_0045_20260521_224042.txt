        }

            }

    }

fn kill_self(signo: i32) -> ! {
    logv(&format!("killing self with signal {}({})", util::signame(signo), signo));
    unsafe {
        let rl: ffi::rlimit = ffi::rlimit { rlim_cur: 0, rlim_max: 0 };
        ffi::setrlimit(ffi::RLIMIT_CORE, &rl);
        ffi::signal(signo, std::os::unix::raw::SIG_DFL as extern "C" fn(i32));
        let mut mask: ffi::sigset_t = std::mem::zeroed();
        ffi::sigemptyset(&mut mask);
        ffi::sigprocmask(ffi::SIG_SETMASK, &mask, std::ptr::null_mut());
        ffi::kill(process::id() as i32, signo);
    }
    process::exit(128 + signo);
}

fn log_termination(name: &str, si: &ffi::siginfo_t) {
    if si.si_code == ffi::CLD_EXITED {
        logv(&format!("{}: pid={}, exit={}", name, si.si_pid, si.si_status));
    } else if si.si_code == ffi::CLD_DUMPED || si.si_code == ffi::CLD_KILLED {
        logv(&format!("{}: pid={}, sig={}", name, si.si_pid, si.si_status));
    } else {
        logv(&format!("{}: pid={}, reason={}, status={}", name, si.si_pid, si.si_code, si.si_status));
    }
}
