// Signal handling for timeout

use std::sync::atomic::{AtomicBool, AtomicI32, Ordering};

use crate::ffi;
use crate::KILLSIG;

pub static SIG_CHLD: AtomicBool = AtomicBool::new(false);
pub static SIG_ALRM: AtomicBool = AtomicBool::new(false);
pub static SIG_TERM: AtomicBool = AtomicBool::new(false);
pub static SIG_OTHER: AtomicBool = AtomicBool::new(false);
pub static SIG_TERM_VAL: AtomicI32 = AtomicI32::new(0);
pub static SIG_OTHER_VAL: AtomicI32 = AtomicI32::new(0);

pub extern "C" fn sig_handler(signo: i32) {
    if signo == KILLSIG.load(Ordering::Relaxed) {
        SIG_TERM.store(true, Ordering::Relaxed);
        return;
    }
    match signo {
        ffi::SIGCHLD => SIG_CHLD.store(true, Ordering::Relaxed),
        ffi::SIGALRM => SIG_ALRM.store(true, Ordering::Relaxed),
        ffi::SIGHUP|ffi::SIGINT|ffi::SIGQUIT|ffi::SIGILL|ffi::SIGTRAP
        |ffi::SIGABRT|ffi::SIGEMT|ffi::SIGFPE|ffi::SIGBUS|ffi::SIGSEGV
        |ffi::SIGSYS|ffi::SIGPIPE|ffi::SIGTERM|ffi::SIGXCPU|ffi::SIGXFSZ
        |ffi::SIGVTALRM|ffi::SIGPROF|ffi::SIGUSR1|ffi::SIGUSR2 => {
            SIG_TERM_VAL.store(signo, Ordering::Relaxed);
            SIG_TERM.store(true, Ordering::Relaxed);
        }
        _ => {
            SIG_OTHER_VAL.store(signo, Ordering::Relaxed);
            SIG_OTHER.store(true, Ordering::Relaxed);
        }
    }
}
