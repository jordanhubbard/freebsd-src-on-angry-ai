extern "C" fn sig_handler(signo: i32) {
    unsafe {
        if signo == KILLSIG {
            SIG_TERM.store(true, Ordering::Relaxed);
            return;
        }
    }

    match signo {
        SIGCHLD => {
            SIG_CHLD.store(true, Ordering::Relaxed);
        }
        SIGALRM => {
            SIG_ALRM.store(true, Ordering::Relaxed);
        }
        SIGHUP | SIGINT | SIGQUIT | SIGILL | SIGTRAP | SIGABRT | SIGEMT
        | SIGFPE | SIGBUS | SIGSEGV | SIGSYS | SIGPIPE | SIGTERM | SIGXCPU
        | SIGXFSZ | SIGVTALRM | SIGPROF | SIGUSR1 | SIGUSR2 => {
            unsafe {
                SIG_TERM_VAL.store(signo, Ordering::Relaxed);
            }
            SIG_TERM.store(true, Ordering::Relaxed);
        }
        _ => {
            unsafe {
                SIG_OTHER_VAL.store(signo, Ordering::Relaxed);
            }
            SIG_OTHER.store(true, Ordering::Relaxed);
        }
    }
}

fn parse_duration(duration: &str) -> f64 {
    let mut chars = duration.chars();
    let suffix_char = chars.next_back();
    let num_str: String = chars.collect();

    let ret: f64 = match num_str.parse() {
        Ok(n) => n,
        Err(_) => {
            eprintln!("timeout: duration is not a number");
            process::exit(EXIT_INVALID);
        }
    };

    if ret.is_nan() {
        eprintln!("timeout: duration is not a number");
        process::exit(EXIT_INVALID);
    }

    match suffix_char {
        None => ret,
        Some('s') => ret,
        Some('m') => ret * 60.0,
        Some('h') => ret * 60.0 * 60.0,
        Some('d') => ret * 60.0 * 60.0 * 24.0,
        Some(_) => {
            eprintln!("timeout: duration unit suffix invalid");
            process::exit(EXIT_INVALID);
        }
    }
}

fn set_interval(iv: f64) {
    let mut tim: itimerval = unsafe { std::mem::zeroed() };
    if iv > 0.0 {
        let secs = iv as i64;
        let usecs = ((iv - secs as f64) * 1_000_000.0) as i64;
        tim.it_value.tv_sec = secs;
        tim.it_value.tv_usec = usecs;

    if unsafe { setitimer(ITIMER_REAL, &tim, std::ptr::null_mut()) } < 0 {
        eprintln!("timeout: setitimer: {}", io::Error::last_os_error());
        process::exit(1);
    }
}

                        rk.rk_fpid
