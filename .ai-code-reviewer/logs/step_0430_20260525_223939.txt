    fn getinode(disk: *const uufsd, dp: *mut dinodep, ino: u64) -> c_int;
    fn putinode(disk: *const uufsd) -> c_int;
    fn ufs_disk_close(disk: *const uufsd) -> c_int;
}

const UFS_ROOTINO: u64 = 2;

#[repr(C)]
pub struct uufsd {
    pub d_fd: c_int,
    pub d_error: *mut c_char,
    pub d_fs: uufsd_fs,
    _padding: [u8; 0],
}

#[repr(C)]
pub union uufsd_fs {
    pub fs_magic: u32,
    _padding: [u8; 0],
}

#[repr(C)]
pub struct dinodep {

}

#[repr(C)]
pub struct ufs1_dinode {
    pub di_gen: u32,
    _rest: [u8; 128 - 4],
}

#[repr(C)]
pub struct ufs2_dinode {
    pub di_gen: u32,
    _rest: [u8; 256 - 4],
}

fn usage() -> ! {
    eprintln!("usage: clri special_device inode_number ...");
    process::exit(1);
}

fn main() {
    let args: Vec<String> = std::env::args().collect();
    if args.len() < 3 {
        usage();
    }

    let fsname = match CString::new(args[1].as_bytes()) {
        Ok(s) => s,
        Err(_) => {
            eprintln!("Invalid device name");
            process::exit(1);
        }
    };

    let mut disk: uufsd = unsafe { std::mem::zeroed() };

    if unsafe { ufs_disk_fillout(&mut disk, fsname.as_ptr()) } == -1 {

                .into_owned()

        eprintln!("loading superblock: {}", err);
        process::exit(1);
    }

        let inonum: u64 = match arg.parse() {
            Ok(n) => n,
            Err(_) => {

                exitval = 1;
                continue;

            exitval = 1;

                    .to_string_lossy()
                    .into_owned()
            };
            eprintln!("getinode: {}", err);
            exitval = 1;
            continue;

        let fs_magic = unsafe { disk.d_fs.fs_magic };
        let generation: u32;
