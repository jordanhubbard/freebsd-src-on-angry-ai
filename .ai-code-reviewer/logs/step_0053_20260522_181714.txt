const BEGIN_DEF: f64 = 1.0;
const ENDER_DEF: f64 = 100.0;

const HAVE_STEP: u8 = 1;
const HAVE_ENDER: u8 = 2;

fn is_default(s: &str) -> bool {
    s.is_empty() || s == "-"
}

fn getprec(str: &str) -> i32 {
    let bytes = str.as_bytes();
    let mut dot_pos = None;
    for (i, &b) in bytes.iter().enumerate() {
        if b == b'.' {
            dot_pos = Some(i);
            break;
        }
    }
    let start = match dot_pos {
        Some(p) => p + 1,
        None => return 0,
    };
    let mut count = 0;
    for &b in &bytes[start..] {
        if b.is_ascii_digit() {
            count += 1;
        } else {
            break;
        }
    }
    count
}

fn getformat(
    format: &mut String,
    prec: i32,
    chardata: bool,
    boring: bool,
) -> (bool, bool, bool, bool) {
    if boring {
        return (false, false, false, false);
    }

    let mut intdata = false;
    let mut longdata = false;
    let mut nosign = false;

    // Look for '%' in format
    let mut pct_pos = None;
    let bytes = format.as_bytes();
    let mut i = 0;
    while i < bytes.len() {
        if bytes[i] == b'%' {
            if i + 1 < bytes.len() && bytes[i + 1] == b'%' {
                i += 2;
                continue;
            }
            pct_pos = Some(i);
            break;
        }
        i += 1;
    }

    if let Some(pos) = pct_pos {
        // There's a '%' in the format string
        let after_pct = &format[pos + 1..];
        if after_pct.is_empty() {
            // Cannot end in single '%'
            format.push('%');
            return (intdata, longdata, nosign, false);
        }

        // Parse the format specifier
        let mut p = pos + 1;
        let mut dot = false;
        let mut hash = false;
        let mut space = false;
        let mut sign = false;
        let mut numbers = false;

        while p < bytes.len() && !bytes[p].is_ascii_alphabetic() {
            if bytes[p].is_ascii_digit() {
                numbers = true;
                p += 1;
            } else if bytes[p] == b'#' && !numbers && !dot && !sign && !space && !hash {
                hash = true;
                p += 1;
            } else if bytes[p] == b' ' && !numbers && !dot && !space {
                space = true;
                p += 1;
            } else if (bytes[p] == b'+' || bytes[p] == b'-') && !numbers && !dot && !sign {
                sign = true;
                p += 1;
            } else if bytes[p] == b'.' && !dot {
                dot = true;
                p += 1;
            } else {
                // Illegal format
                let spec_start = pos;
                let spec_end = p;
                let spec = &format[spec_start..spec_end];
                errx(1, &format!("illegal or unsupported format '{}'", spec));
            }
        }

        // Check for 'l' prefix
        if p < bytes.len() && bytes[p] == b'l' {
            longdata = true;
            p += 1;
            if p < bytes.len() && bytes[p] == b'l' {
                // 'll' is not supported
                let spec_start = pos;
                let spec_end = p + 1;
                let spec = &format[spec_start..spec_end];
                errx(1, &format!("illegal or unsupported format '{}'", spec));
            }
        }

        if p >= bytes.len() {
            let spec_start = pos;
            let spec = &format[spec_start..];
            errx(1, &format!("illegal or unsupported format '{}'", spec));
        }

        let conv = bytes[p] as char;
        match conv {
            'o' | 'u' | 'x' | 'X' => {
                intdata = true;
                nosign = true;
            }
            'd' | 'i' => {
                intdata = true;
            }
            'D' => {
                if !longdata {
                    intdata = true;
                } else {
                    // longdata with D is unsupported
                    let spec_start = pos;
                    let spec = &format[spec_start..p + 1];
                    errx(1, &format!("illegal or unsupported format '{}'", spec));
                }
            }
            'O' | 'U' => {
                if !longdata {
                    intdata = true;
                    nosign = true;
                } else {
                    let spec_start = pos;
                    let spec = &format[spec_start..p + 1];
                    errx(1, &format!("illegal or unsupported format '{}'", spec));
                }
            }
            'c' => {
                if !intdata && !longdata {
                    intdata = true; // chardata handled separately
                } else {
                    let spec_start = pos;
                    let spec = &format[spec_start..p + 1];
                    errx(1, &format!("illegal or unsupported format '{}'", spec));
                }
            }
            'h' | 'n' | 'p' | 'q' | 's' | 'L' | '$' | '*' => {
                let spec_start = pos;
                let spec = &format[spec_start..p + 1];
                errx(1, &format!("illegal or unsupported format '{}'", spec));
            }
            'f' | 'e' | 'g' | 'E' | 'G' => {
                if longdata {
                    let spec_start = pos;
                    let spec = &format[spec_start..p + 1];
                    errx(1, &format!("illegal or unsupported format '{}'", spec));
                }
            }
            _ => {
                let spec_start = pos;
                let spec = &format[spec_start..p + 1];
                errx(1, &format!("illegal or unsupported format '{}'", spec));
            }
        }

        // Check for additional conversions
        p += 1;
        while p < bytes.len() {
            if bytes[p] == b'%' {
                if p + 1 < bytes.len() {
                    if bytes[p + 1] == b'%' {
                        p += 2;
                        continue;
                    }
                    // Too many conversions
                    errx(1, "too many conversions");
                } else {
                    // Trailing single '%'
                    format.push('%');
                    break;
                }
            }
            p += 1;
        }
    } else if !chardata {
        // No '%' found, append default format
        format.push_str(&format!("%.{}f", prec));
    } else {
        // chardata with no '%'
        format.push_str("%c");
        intdata = true;
    }

    (intdata, longdata, nosign, false)
}

fn putdata(
    x: f64,
    last: bool,
    sepstring: &str,
    format: &str,
    boring: bool,
    longdata: bool,
    intdata: bool,
    chardata: bool,
    nosign: bool,
) -> bool {
    let stdout = io::stdout();
    let mut out = stdout.lock();

    if boring {
        let _ = write!(out, "{}", format);
    } else if longdata && nosign {
        if x <= u64::MAX as f64 && x >= 0.0 {
            let _ = write!(out, format!("{}", format), x as u64);
        } else {
            return true;
        }
    } else if longdata {
        if x <= i64::MAX as f64 && x >= i64::MIN as f64 {
            let _ = write!(out, format!("{}", format), x as i64);
        } else {
            return true;
        }
    } else if chardata || (intdata && !nosign) {
        if x <= i32::MAX as f64 && x >= i32::MIN as f64 {
            let _ = write!(out, format!("{}", format), x as i32);
        } else {
            return true;
        }
    } else if intdata {
        if x <= u32::MAX as f64 && x >= 0.0 {
            let _ = write!(out, format!("{}", format), x as u32);
        } else {
            return true;
        }
    } else {
        let _ = write!(out, format!("{}", format), x);
    }

    if !last {
        let _ = write!(out, "{}", sepstring);
    }

    false
}
