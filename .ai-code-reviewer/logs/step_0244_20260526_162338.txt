use std::env;
use std::fs::File;
use std::io::{self, BufRead, BufReader, Read, Write};
use std::process;

#[derive(Debug, Clone, Copy, PartialEq, Eq)]
enum RandomType {
    Unset,
    Lines,
    Words,
}

fn usage() -> ! {
    eprintln!("usage: random [-elrUuw] [-f filename] [denominator]");
    process::exit(1);
}

fn main() {
    let args: Vec<String> = env::args().collect();
    let mut random_exit = false;
    let mut randomize_lines = false;
    let mut random_type = RandomType::Unset;
    let mut unique_output = true;
    let mut unbuffer_output = false;
    let mut filename = String::from("/dev/fd/0");
    let mut denom: Option<f64> = None;

    let mut i = 1;
    while i < args.len() {
        let arg = &args[i];
        if arg == "-e" {
            random_exit = true;
            i += 1;
        } else if arg == "-f" {
            randomize_lines = true;
            i += 1;
            if i < args.len() {
                filename = args[i].clone();
            }
            i += 1;
        } else if arg == "-l" {
            randomize_lines = true;
            random_type = RandomType::Lines;
            i += 1;
        } else if arg == "-r" {
            unbuffer_output = true;
            i += 1;
        } else if arg == "-u" {
            randomize_lines = true;
            unique_output = true;
            i += 1;
        } else if arg == "-U" {
            randomize_lines = true;
            unique_output = false;
            i += 1;
        } else if arg == "-w" {
            randomize_lines = true;
            random_type = RandomType::Words;
            i += 1;

                    'e' => random_exit = true,
                    'l' => {
                        randomize_lines = true;
                        random_type = RandomType::Lines;

                    'r' => unbuffer_output = true,
                    'u' => {
                        randomize_lines = true;
                        unique_output = true;
                    }
                    'U' => {
                        randomize_lines = true;
                        unique_output = false;
                    }
                    'w' => {
                        randomize_lines = true;
                        random_type = RandomType::Words;
                    }
                    'f' => {
                        // -f needs an argument, handle next arg
                        randomize_lines = true;
                        i += 1;
                        if i < args.len() {
                            filename = args[i].clone();
                        }
                        break;
                    }
                    _ => usage(),
                }
            }
            i += 1;
        } else {
            break;
        }

    let remaining = &args[i..];

                2.0
            }

                process::exit(1);

                process::exit(1);

            eprintln!("random: {}: {}", filename, e);
            process::exit(1);

    }

fn random_uniform_denom(denom: f64) -> bool {
    // Returns true with probability 1/denom
    let d = denom as u64;
    if d < 2 {
        return true;
    }
    fastrand::u64(0..d) == 0
}

fn randomize_reader(
    mut reader: BufReader<File>,
    _type: RandomType,
    unique: bool,
    denom: f64,
) -> i32 {
    let mut tokens: Vec<String> = Vec::new();

    let mut buf = String::new();
    loop {
        let bytes_read = match reader.read_line(&mut buf) {
            Ok(0) => break,
            Ok(n) => n,
            Err(_) => {
                eprintln!("random: read: error");
                process::exit(1);
            }
        };
        if bytes_read == 0 {
            break;
        }

        match _type {
            RandomType::Lines => {
                if !buf.is_empty() {
                    tokens.push(buf.clone());
                    buf.clear();
                }
            }
            RandomType::Words => {
                for word in buf.split_whitespace() {
                    tokens.push(format!("{}\n", word));
                }
                buf.clear();
            }
            RandomType::Unset => {}
        }
    }

    let stdout = io::stdout();
    let mut out = stdout.lock();

    let mut remaining = tokens;
    while !remaining.is_empty() {
        let idx = fastrand::usize(0..remaining.len());
        let token = remaining.remove(idx);
        if random_uniform_denom(denom) {
            if write!(out, "{}", token).is_err() {
                eprintln!("random: printf: write error");
                process::exit(1);
            }
        }
        if unique {
            // Token already removed from remaining
        }
    }

    if out.flush().is_err() {
        eprintln!("random: fflush: write error");
        process::exit(1);
    }

    0
}
