/// Test handler that prints encoder operations.
/// Matches the original test_handler from enc_test.c.
#[no_mangle]
pub unsafe extern "C" fn test_handler(
    op: XoEncoderOp,
    name: *const c_char,
    value: *const c_char,
    flags: XoEncoderFlags,
) -> i32 {
    let op_name = match op {
        XoEncoderOp::BeginObject => "BeginObject",
        XoEncoderOp::EndObject => "EndObject",
        XoEncoderOp::BeginArray => "BeginArray",
        XoEncoderOp::EndArray => "EndArray",
        XoEncoderOp::BeginMember => "BeginMember",
        XoEncoderOp::EndMember => "EndMember",
        XoEncoderOp::BeginValue => "BeginValue",
        XoEncoderOp::EndValue => "EndValue",
        XoEncoderOp::BeginString => "BeginString",
        XoEncoderOp::EndString => "EndString",
        XoEncoderOp::BeginNumber => "BeginNumber",
        XoEncoderOp::EndNumber => "EndNumber",
        XoEncoderOp::BeginBoolean => "BeginBoolean",
        XoEncoderOp::EndBoolean => "EndBoolean",
        XoEncoderOp::BeginNull => "BeginNull",
        XoEncoderOp::EndNull => "EndNull",
        _ => "Unknown",
    };

    let name_str = if name.is_null() {
        ""
    } else {
        std::ffi::CStr::from_ptr(name).to_string_lossy()
    };

    let value_str = if value.is_null() {
        ""
    } else {
        std::ffi::CStr::from_ptr(value).to_string_lossy()
    };

    println!(
        "op {}: [{}] [{}] [{:#x}]",
        op_name, name_str, value_str, flags
    );

    0
}
