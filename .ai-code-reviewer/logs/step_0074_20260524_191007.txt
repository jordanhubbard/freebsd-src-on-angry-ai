fn full_read(fd: RawFd, dst: &mut [u8]) -> i64 {
    let mut total: usize = 0;
    while total < dst.len() {
        let n = unsafe { libc::read(fd, dst[total..].as_mut_ptr() as *mut libc::c_void, dst.len() - total) };
        if n <= 0 { return total as i64; }
        total += n as usize;
    }
    total as i64
}

fn report_total<W: io::Write>(mut w: W) {

}

fn expand_number(s: &str) -> Option<i64> {
    let (num, suf) = match s.find(|c: char| c.is_ascii_alphabetic()) {
        Some(p) => (&s[..p], &s[p..]),
        None => (s, ""),
    };
    let mult = match suf {
        "" | "b" => 1i64, "k" => 1024, "m" => 1024*1024,
        "g" => 1024*1024*1024, "t" => 1024*1024*1024*1024, _ => return None,
    };
    num.parse::<i64>().ok().map(|n| n * mult)
}

trait TapeOps {
    fn read_blk(&mut self, dst: &mut [u8]) -> i64;
    fn verify_blk(&mut self, dst: &mut [u8], expected: usize) -> i64;
    fn write_blk(&mut self, src: &[u8]);
    fn file_mark(&mut self);
    fn rewind(&mut self);
}

struct TapeDev { fd: RawFd, name: String, max_read_size: usize }
impl TapeDev {
    fn new(fd: RawFd, name: &str) -> Self {
        TapeDev { fd, name: name.to_string(), max_read_size: 0 }
    }
}
impl TapeOps for TapeDev {
    fn read_blk(&mut self, dst: &mut [u8]) -> i64 {
        let mut mrs = if self.max_read_size == 0 { dst.len() } else { self.max_read_size };
        let mut retval: i64 = -1;
        if self.max_read_size == 0 {
            while mrs > 0 {
                retval = unsafe { libc::read(self.fd, dst.as_mut_ptr() as *mut libc::c_void, mrs) };
                if retval >= 0 {
                    self.max_read_size = mrs;
                    break;
                }
                let err = io::Error::last_os_error().raw_os_error().unwrap_or(0);
                if err != libc::EINVAL && err != libc::EFBIG { break; }
                if mrs < 512 { eprintln!("tcopy: Cannot find a sane max blocksize"); process::exit(1); }
                let i = (mrs as u64 - 1).leading_zeros() as usize;
                mrs = 1usize << (64 - i - 1);
            }
        } else {
            retval = unsafe { libc::read(self.fd, dst.as_mut_ptr() as *mut libc::c_void, self.max_read_size) };
        }
        if retval < 0 {
            eprintln!("tcopy: read error, {}, file {}, record {}", self.name, unsafe { G_FILN }, unsafe { G_RECORD });
            process::exit(1);
        }
        retval
    }
    fn verify_blk(&mut self, dst: &mut [u8], _expected: usize) -> i64 { self.read_blk(dst) }
    fn write_blk(&mut self, src: &[u8]) {
        let n = unsafe { libc::write(self.fd, src.as_ptr() as *const libc::c_void, src.len()) };
        if n < 0 || (n as usize) != src.len() {
            eprintln!("tcopy: write error, file {}, record {}", unsafe { G_FILN }, unsafe { G_RECORD });
            process::exit(5);
        }
    }
    fn file_mark(&mut self) {
        let mut op: libc::mtio_mtop = unsafe { std::mem::zeroed() };
        op.mt_op = libc::MTWEOF;
        op.mt_count = 1;
        if unsafe { libc::ioctl(self.fd, libc::MTIOCTOP, &mut op) } < 0 {
            eprintln!("tcopy: tape op (write file mark): {}", io::Error::last_os_error());
            process::exit(6);
        }
    }
    fn rewind(&mut self) {
        let mut op: libc::mtio_mtop = unsafe { std::mem::zeroed() };
        op.mt_op = libc::MTREW;
        op.mt_count = 1;
        if unsafe { libc::ioctl(self.fd, libc::MTIOCTOP, &mut op) } < 0 {
            eprintln!("tcopy: tape op (rewind): {}", io::Error::last_os_error());
            process::exit(6);
        }
    }
}

struct TapFile { fd: RawFd, name: String }
impl TapFile {
    fn new(fd: RawFd, name: &str) -> Self { TapFile { fd, name: name.to_string() } }
}
impl TapeOps for TapFile {
    fn read_blk(&mut self, dst: &mut [u8]) -> i64 {
        let mut hdr = [0u8; 4];
        let nread = full_read(self.fd, &mut hdr);
        if nread == 0 { return 0; }
        if nread != 4 { eprintln!("tcopy: Corrupt tap-file, read hdr1={}", nread); process::exit(17); }
        let u = u32::from_le_bytes(hdr) as u64;
        if u == 0 || (u >> 24) == 0xff { return 0; }
        if u as usize > dst.len() { eprintln!("tcopy: tapfile blocksize too big, 0x{:08x}", u); process::exit(17); }
        full_read(self.fd, &mut dst[..u as usize])
    }
    fn verify_blk(&mut self, dst: &mut [u8], _expected: usize) -> i64 { self.read_blk(dst) }
    fn write_blk(&mut self, src: &[u8]) {
        let len = src.len() as u32;
        let hdr = len.to_le_bytes();
        let n = unsafe { libc::write(self.fd, hdr.as_ptr() as *const libc::c_void, 4) };
        if n != 4 { eprintln!("tcopy: write error"); process::exit(5); }
        let n = unsafe { libc::write(self.fd, src.as_ptr() as *const libc::c_void, src.len()) };
        if n < 0 || (n as usize) != src.len() { eprintln!("tcopy: write error"); process::exit(5); }
    }
    fn file_mark(&mut self) {
        let eof = 0xffffffffu32.to_le_bytes();
        let n = unsafe { libc::write(self.fd, eof.as_ptr() as *const libc::c_void, 4) };
        if n != 4 { eprintln!("tcopy: write error"); process::exit(5); }
    }
    fn rewind(&mut self) {
        let f = unsafe { std::fs::File::from_raw_fd(self.fd) };
        let _ = f.seek(SeekFrom::Start(0));
        let _ = f.into_raw_fd();
    }
}

fn open_arg(name: &str, is_output: bool, rawfile: bool) -> Box<dyn TapeOps> {

    let fd = unsafe { libc::open(c_name.as_ptr(), flags, 0o666) };

    let mut st: libc::stat = unsafe { std::mem::zeroed() };
    let is_reg = unsafe { libc::fstat(fd, &mut st) == 0 && (st.st_mode & libc::S_IFMT) == libc::S_IFREG };

    else if is_reg { Box::new(TapFile::new(fd, name)) }

}
