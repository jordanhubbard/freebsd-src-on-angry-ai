use std::env;
use std::fs::File;
use std::io::{self, BufRead, BufReader};
use std::process;

const TAB: usize = 8;
const DEFNUM: usize = 1000;
const DEFCOLS: usize = 25;

fn usage() -> ! {
    eprintln!(
        "usage: column [-tx] [-c columns] [-l tblcols] [-s sep] [file ...]"
    );
    process::exit(1);
}

fn roundup(v: usize, align: usize) -> usize {
    (v + align - 1) / align * align
}

fn width(s: &str) -> usize {
    s.chars()
        .filter(|c| !c.is_control())
        .map(|c| c.width().unwrap_or(0))
        .sum()
}

fn get_termwidth() -> usize {
    // Try COLUMNS env var first, then default to 80
    // The original C code uses ioctl TIOCGWINSZ but we avoid libc dependency
    env::var("COLUMNS")
        .ok()
        .and_then(|v| v.parse().ok())
        .unwrap_or(80)
}

fn input_file(fp: &mut dyn BufRead, list: &mut Vec<String>, eval: &mut i32) {
    let mut maxentry = DEFNUM;
    let mut entries = list.len();

    let mut buf = String::new();
    while fp.read_line(&mut buf).unwrap_or(0) > 0 {
        let trimmed: String = buf.chars().skip_while(|c| c.is_whitespace()).collect();
        if trimmed.is_empty() {
            buf.clear();
            continue;
        }
        // Remove trailing newline
        let line = trimmed.trim_end_matches('\n').trim_end_matches('\r');
        if entries >= maxentry {
            maxentry += DEFNUM;
        }
        list.push(line.to_string());
        entries += 1;
        buf.clear();
    }
    let _ = eval; // suppress unused warning
}

fn c_columnate(list: &[String], termwidth: usize, maxlength: usize) {
    let numcols = termwidth / maxlength;
    let mut col = 0;
    let mut chcnt = 0;
    let mut endcol = maxlength;
    let mut entries = list.len();

    for item in list {
        print!("{}", item);
        chcnt += width(item);
        entries -= 1;
        if entries == 0 {
            break;
        }
        col += 1;
        if col == numcols {
            chcnt = 0;
            col = 0;
            endcol = maxlength;
            println!();
        } else {
            let mut cnt = roundup(chcnt + 1, TAB);
            while cnt <= endcol {
                print!("\t");
                chcnt = cnt;
                cnt = roundup(chcnt + 1, TAB);
            }
            endcol += maxlength;
        }
    }
    if chcnt > 0 {
        println!();
    }
}

fn r_columnate(list: &[String], termwidth: usize, maxlength: usize) {
    let numcols = termwidth / maxlength;
    let entries = list.len();
    let numrows = if entries % numcols == 0 {
        entries / numcols
    } else {
        entries / numcols + 1
    };

    for row in 0..numrows {
        let mut endcol = maxlength;
        let mut chcnt = 0;
        let mut col = 0;
        let mut base = row;
        while col < numcols {
            if base >= entries {
                break;
            }
            print!("{}", list[base]);
            chcnt += width(list[base]);
            base += numrows;
            if base >= entries {
                break;
            }
            let mut cnt = roundup(chcnt + 1, TAB);
            while cnt <= endcol {
                print!("\t");
                chcnt = cnt;
                cnt = roundup(chcnt + 1, TAB);
            }
            endcol += maxlength;
            col += 1;
        }
        println!();
    }
}

fn print_list(list: &[String]) {
    for item in list {
        println!("{}", item);
    }
}

struct TblEntry {
    cols: Vec<String>,
    len: Vec<usize>,
}

fn maketbl(list: &[String], separator: &str, tblcols: Option<usize>) {
    let mut tbl: Vec<TblEntry> = Vec::with_capacity(list.len());
    let mut lens: Vec<usize> = Vec::new();
    let mut maxcols = DEFCOLS;

    for line in list {
        let mut cols: Vec<String> = Vec::new();
        let mut coloff = 0;
        let mut chars = line.chars().peekable();

        // Skip leading separators
        while let Some(c) = chars.peek() {
            if separator.contains(*c) {
                chars.next();
            } else {
                break;
            }
        }

        while chars.peek().is_some() {
            if coloff >= maxcols {
                maxcols += DEFCOLS;
                while lens.len() < maxcols {
                    lens.push(0);
                }
            }

            let mut field = String::new();
            while let Some(c) = chars.next() {
                if tblcols.is_none() || coloff < tblcols.unwrap() {
                    if separator.contains(c) {
                        break;
                    }
                }
                field.push(c);
            }
            cols.push(field);
            coloff += 1;

            // Skip separators between fields
            while let Some(c) = chars.peek() {
                if separator.contains(*c) {
                    chars.next();
                } else {
                    break;
                }
            }
        }

        let lens_copy: Vec<usize> = cols.iter().map(|s| width(s)).collect();
        for i in 0..cols.len() {
            if i >= lens.len() {
                lens.push(0);
            }
            if lens_copy[i] > lens[i] {
                lens[i] = lens_copy[i];
            }
        }
        tbl.push(TblEntry { cols, len: lens_copy });
    }

    for entry in &tbl {
        for i in 0..entry.cols.len().saturating_sub(1) {
            let pad = lens[i] - entry.len[i] + 2;
            print!("{}{}", entry.cols[i], " ".repeat(pad));
        }
        if !entry.cols.is_empty() {
            let last = entry.cols.len() - 1;
            println!("{}", entry.cols[last]);
        } else {
            println!();
        }
    }
}

fn main() {
    let args: Vec<String> = env::args().collect();

    let mut termwidth = get_termwidth();
    let mut tblcols: Option<usize> = None;
    let mut separator = "\t ";
    let mut tflag = false;
    let mut xflag = false;
    let mut files: Vec<String> = Vec::new();

    let mut i = 1;
    while i < args.len() {
        match args[i].as_str() {
            "-c" => {
                i += 1;
                if i >= args.len() {
                    usage();
                }
                termwidth = args[i].parse().unwrap_or_else(|_| usage());
            }
            "-l" => {
                i += 1;
                if i >= args.len() {
                    usage();
                }
                tblcols = Some(args[i].parse().unwrap_or_else(|_| usage()));
            }
            "-s" => {
                i += 1;
                if i >= args.len() {
                    usage();
                }
                separator = &args[i];
            }
            "-t" => {
                tflag = true;
            }
            "-x" => {
                xflag = true;
            }
            "--" => {
                i += 1;
                break;
            }
            s if s.starts_with('-') => {
                usage();
            }
            _ => {
                files.push(args[i].clone());
            }
        }
        i += 1;
    }

    if tblcols.is_some() && !tflag {
        eprintln!("the -l flag cannot be used without the -t flag");
        process::exit(1);
    }

    let mut list: Vec<String> = Vec::new();
    let mut eval = 0i32;

    if files.is_empty() {
        let stdin = io::stdin();
        let mut reader = BufReader::new(stdin.lock());
        input_file(&mut reader, &mut list, &mut eval);
    } else {
        for f in &files {
            match File::open(f) {
                Ok(file) => {
                    let mut reader = BufReader::new(file);
                    input_file(&mut reader, &mut list, &mut eval);
                }
                Err(e) => {
                    eprintln!("column: {}: {}", f, e);
                    eval = 1;
                }
            }
        }
    }

    if list.is_empty() {
        process::exit(eval);
    }

    let maxlength = list.iter().map(|s| width(s)).max().unwrap_or(0);
    let maxlength = roundup(maxlength + 1, TAB);

    if tflag {
        maketbl(&list, separator, tblcols);
    } else if maxlength >= termwidth {
        print_list(&list);
    } else if xflag {
        c_columnate(&list, termwidth, maxlength);
    } else {
        r_columnate(&list, termwidth, maxlength);
    }

    process::exit(eval);
}
