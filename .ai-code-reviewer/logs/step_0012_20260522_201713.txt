fn do_mkdtemp(template: &str) -> Result<String, std::io::Error> {
    let mut buf = template.as_bytes().to_vec();
    buf.push(0);
    let result = unsafe {
        let ptr = buf.as_mut_ptr() as *mut c_char;
        mkdtemp(ptr)
    };
    if result.is_null() {
        return Err(std::io::Error::last_os_error());
    }
    let path = unsafe { std::ffi::CStr::from_ptr(result) }
        .to_string_lossy()
        .into_owned();
    Ok(path)
}

fn do_mkstemp(template: &str) -> Result<String, std::io::Error> {
    let mut buf = template.as_bytes().to_vec();
    buf.push(0);
    let fd = unsafe {
        let ptr = buf.as_mut_ptr() as *mut c_char;
        mkstemp(ptr)
    };
    if fd < 0 {
        return Err(std::io::Error::last_os_error());
    }
    let path = unsafe {
        std::ffi::CStr::from_ptr(buf.as_ptr() as *const c_char)
            .to_string_lossy()
            .into_owned()
    };
    unsafe { close(fd) };
    Ok(path)
}
