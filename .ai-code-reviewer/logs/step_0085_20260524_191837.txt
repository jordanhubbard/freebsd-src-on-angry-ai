fn full_read(fd: RawFd, dst: &mut [u8]) -> i64 {
    let mut total: usize = 0;
    while total < dst.len() {
        let n = unsafe { read(fd, dst[total..].as_mut_ptr() as *mut std::os::raw::c_void, dst.len() - total) };
        if n <= 0 { return total as i64; }
        total += n as usize;
    }
    total as i64
}

fn expand_number(s: &str) -> Option<i64> {
    let (num, suf) = match s.find(|c: char| c.is_ascii_alphabetic()) {
        Some(p) => (&s[..p], &s[p..]),
        None => (s, ""),
    };
    let mult = match suf {
        "" | "b" => 1i64, "k" => 1024, "m" => 1024*1024,
        "g" => 1024*1024*1024, "t" => 1024*1024*1024*1024, _ => return None,
    };
    num.parse::<i64>().ok().map(|n| n * mult)
}
