use std::os::raw::{c_int, size_t};

extern "C" {
    fn getpagesizes(buf: *mut size_t, bufsize: c_int) -> c_int;
    fn getpagesize() -> c_int;
    fn getosreldate() -> c_int;
    fn sysconf(name: c_int) -> c_int;
}

const _SC_NPROCESSORS_CONF: c_int = 84;

fn test_pagesizes() {
    let nelem = unsafe { getpagesizes(std::ptr::null_mut(), 0) };
    if nelem == -1 {
        eprintln!("getpagesizes(NULL, 0): {}", std::io::Error::last_os_error());
        std::process::exit(1);
    }
    let mut ps: Vec<size_t> = vec![0; nelem as usize];
    let nelem = unsafe { getpagesizes(ps.as_mut_ptr(), nelem) };
    if nelem == -1 {
        eprintln!("getpagesizes: {}", std::io::Error::last_os_error());
        std::process::exit(1);
    }
    print!("Supported page sizes:");
    for i in 0..nelem as usize {
        print!(" {}", ps[i] as i64);
    }
    println!();
}

fn test_pagesize() {
    let ps = unsafe { getpagesize() };
    println!("Pagesize: {}", ps);
}

fn test_osreldate() {
    let date = unsafe { getosreldate() };
    println!("OSRELDATE: {}", date);
}

fn test_ncpus() {
    let ncpus = unsafe { sysconf(_SC_NPROCESSORS_CONF) };
    println!("NCPUs: {}", ncpus as i64);
}

fn main() {
    test_pagesizes();
    test_pagesize();
    test_osreldate();
    test_ncpus();
}
