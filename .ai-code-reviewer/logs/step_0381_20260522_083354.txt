fn get_whom() -> String {
    let login = unsafe { getlogin() };
    if !login.is_null() {
        let s = unsafe { std::ffi::CStr::from_ptr(login) };
        return s.to_string_lossy().into_owned();
    }

    let uid = unsafe { getuid() };
    let pw = unsafe { getpwuid(uid) };
    if pw.is_null() {
        return "???".to_string();
    }
    let name = unsafe { (*pw).pw_name };
    unsafe { std::ffi::CStr::from_ptr(name) }.to_string_lossy().into_owned()
}

fn get_hostname() -> String {
    let mut hostname = [0i8; MAXHOSTNAMELEN];
    unsafe { gethostname(hostname.as_mut_ptr(), hostname.len()) };
    unsafe { std::ffi::CStr::from_ptr(hostname.as_ptr()) }
        .to_string_lossy()
        .into_owned()
}

fn get_time() -> (i32, i32) {

    unsafe { time(&mut now) };
    let lt = unsafe { localtime(&now) };
    let hour = unsafe { (*lt).tm_hour };
    let min = unsafe { (*lt).tm_min };
    (hour, min)
}

    }
