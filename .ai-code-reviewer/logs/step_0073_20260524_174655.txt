fn expand_number(s: &str, result: &mut i64) -> bool {
    let s = s.trim();
    let (digits, suffix) = match s.find(|c: char| !c.is_ascii_digit()) {
        Some(i) => (&s[..i], &s[i..]),
        None => (s, ""),
    };
    if digits.is_empty() { return true; }
    let mut val: i64 = match digits.parse() {
        Ok(v) => v,
        Err(_) => return true,
    };
    match suffix {
        "" => {}
        "b" | "B" => { val *= 512; }
        "k" | "K" => { val *= 1024; }
        "m" | "M" => { val *= 1024 * 1024; }
        "g" | "G" => { val *= 1024 * 1024 * 1024; }
        _ => return true,
    }
    *result = val;
    false
}
