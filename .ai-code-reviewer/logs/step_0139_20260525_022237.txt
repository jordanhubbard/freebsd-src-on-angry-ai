fn store(table: &mut [Option<Box<GroupEntry>>], key: &str, data: &str) {
    let i = hashkey(key) as usize;

        key: key.to_string(),
        data: data.to_string(),
        next: table[i].take(),

fn mstore(
    table: &mut [Option<Box<MemberEntry>>],
    key: &str,
    data: &str,
    domain: &str,
) {
    let i = hashkey(key) as usize;
    let mut cur = &table[i];

    let tmp = Box::new(GroupList {
        groupname: data.to_string(),
        next: None,
    });

    // Check if member already exists
    let mut found = false;
    let mut cur_ref = cur;
    while let Some(entry) = cur_ref {
        if entry.key == key {
            // Update grouplist
            let mut new_groups = Box::new(GroupList {
                groupname: data.to_string(),
                next: entry.groups.clone(),
            });
            // We need to update the entry in place - rebuild the table entry
            let mut rebuilt = Some(entry.clone());
            if let Some(ref mut e) = rebuilt {
                e.groups = Some(new_groups);
            }
            // Rebuild linked list
            let mut next_rest = entry.next.clone();
            table[i] = rebuilt;
            // Reattach rest of list
            let mut tail = &mut table[i];
            while let Some(ref mut t) = tail {
                if t.next.is_none() {
                    t.next = next_rest;
                    break;
                }
                tail = &mut t.next;
            }
            found = true;
            break;

        cur_ref = &entry.next;

    if !found {
        let new = Box::new(MemberEntry {
            key: key.to_string(),
            domain: if domain.is_empty() {
                "*".to_string()
            } else {
                domain.to_string()
            },
            groups: Some(tmp),
            next: table[i].take(),
        });
        table[i] = Some(new);
    }
}
