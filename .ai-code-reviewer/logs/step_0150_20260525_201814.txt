fn diff_files(file1: &Path, file2: &Path, opts: &DiffOpts) -> i32 {
    let label1 = if opts.labels.len() > 0 {
        opts.labels[0].clone()
    } else {
        file1.to_string_lossy().to_string()
    };
    let label2 = if opts.labels.len() > 1 {
        opts.labels[1].clone()
    } else {
        file2.to_string_lossy().to_string()
    };
    
    // Check if files exist
    let exists1 = file1.exists();
    let exists2 = file2.exists();
    
    if !exists1 && !exists2 {
        if opts.new_file || opts.unidirectional_new_file {
            eprintln!("diff: {}: No such file or directory", label1);
            eprintln!("diff: {}: No such file or directory", label2);
        }
        return 2;
    }
    
    if !exists1 {
        if opts.new_file || opts.unidirectional_new_file {
            println!("{} only", label1);
        }
        return 1;
    }
    
    if !exists2 {
        if opts.new_file || opts.unidirectional_new_file {
            println!("{} only", label2);
        }
        return 1;
    }
    
    // Read file contents
    let lines1 = read_lines(file1).unwrap_or_else(|e| {
        eprintln!("diff: {}: {}", label1, e);
        Vec::new()
    });
    let lines2 = read_lines(file2).unwrap_or_else(|e| {
        eprintln!("diff: {}: {}", label2, e);
        Vec::new()
    });
    
    // Compare lines
    if lines1 == lines2 {
        if opts.report_identical {
            println!("Files {} and {} are identical", label1, label2);
        }
        return 0;
    }
    
    if opts.brief {
        println!("Files {} and {} differ", label1, label2);
        return 1;
    }
    
    // Compute diff
    let edits = compute_diff(&lines1, &lines2);
    
    match opts.format {
        DiffFormat::Normal => print_normal_diff(&label1, &label2, &edits, &lines1, &lines2),
        DiffFormat::Unified => print_unified_diff(&label1, &label2, &edits, &lines1, &lines2, opts.context),
        DiffFormat::Context => print_context_diff(&label1, &label2, &edits, &lines1, &lines2, opts.context),
        DiffFormat::SideBySide => print_side_by_side_diff(&label1, &label2, &lines1, &lines2, opts.width),
        _ => print_normal_diff(&label1, &label2, &edits, &lines1, &lines2),
    }
    
    1
}

fn diff_dirs(dir1: &Path, dir2: &Path, opts: &DiffOpts) -> i32 {
    let mut status = 0;
    
    let entries1 = match fs::read_dir(dir1) {
        Ok(e) => e,
        Err(e) => {
            eprintln!("diff: {}: {}", dir1.display(), e);
            return 2;
        }
    };
    let entries2 = match fs::read_dir(dir2) {
        Ok(e) => e,
        Err(e) => {
            eprintln!("diff: {}: {}", dir2.display(), e);
            return 2;
        }
    };
    
    let mut names1: Vec<String> = Vec::new();
    for entry in entries1 {
        if let Ok(entry) = entry {
            let name = entry.file_name().to_string_lossy().to_string();
            if name != "." && name != ".." {
                names1.push(name);
            }
        }
    }
    names1.sort();
    
    let mut names2: Vec<String> = Vec::new();
    for entry in entries2 {
        if let Ok(entry) = entry {
            let name = entry.file_name().to_string_lossy().to_string();
            if name != "." && name != ".." {
                names2.push(name);
            }
        }
    }
    names2.sort();
    
    for name in &names1 {
        let path1 = dir1.join(name);
        let path2 = dir2.join(name);
        
        if !names2.contains(name) {
            if path1.is_dir() && opts.recursive {
                println!("Only in {}: {}", dir1.display(), name);
            } else if !path1.is_dir() {
                println!("Only in {}: {}", dir1.display(), name);
            }
            status = 1;
            continue;
        }
        
        if path1.is_dir() && path2.is_dir() && opts.recursive {
            let s = diff_dirs(&path1, &path2, opts);
            if s > status { status = s; }
        } else {
            let s = diff_files(&path1, &path2, opts);
            if s > status { status = s; }
        }
    }
    
    for name in &names2 {
        if !names1.contains(name) {
            let path2 = dir2.join(name);
            if path2.is_dir() && opts.recursive {
                println!("Only in {}: {}", dir2.display(), name);
            } else if !path2.is_dir() {
                println!("Only in {}: {}", dir2.display(), name);
            }
            status = 1;
        }
    }
    
    status
}

fn read_lines(path: &Path) -> io::Result<Vec<String>> {
    let file = fs::File::open(path)?;
    let reader = BufReader::new(file);
    let mut lines = Vec::new();
    for line in reader.lines() {
        lines.push(line?);
    }
    Ok(lines)
}

#[derive(Debug, Clone, PartialEq)]
enum Edit {
    Equal(usize, usize),  // count of equal lines
    Delete(usize, usize), // start in file1, count
    Insert(usize, usize), // start in file2, count
}

fn compute_diff(lines1: &[String], lines2: &[String]) -> Vec<Edit> {
    let n = lines1.len();
    let m = lines2.len();
    
    // LCS-based diff using Myers-like algorithm
    // For simplicity, use a basic LCS approach
    let mut dp = vec![vec![0usize; m + 1]; n + 1];
    
    for i in 1..=n {
        for j in 1..=m {
            if lines1[i-1] == lines2[j-1] {
                dp[i][j] = dp[i-1][j-1] + 1;
            } else {
                dp[i][j] = dp[i-1][j].max(dp[i][j-1]);
            }
        }
    }
    
    // Backtrack to find edits
    let mut edits = Vec::new();
    let mut i = n;
    let mut j = m;
    
    let mut pending_del: Option<(usize, usize)> = None; // (start, count) in file1
    let mut pending_ins: Option<(usize, usize)> = None; // (start, count) in file2
    
    while i > 0 ||
