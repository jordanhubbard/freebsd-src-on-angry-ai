pub fn parse_duration(d: &str) -> f64 {
    let b = d.as_bytes();
    let mut end = b.len();
    let mut suf = '\0';
    for (i, &c) in b.iter().enumerate() {
        if !c.is_ascii_digit() && c != b'.' && c != b'-' && c != b'+' && c != b'e' && c != b'E' {
            end = i; suf = c as char; break;
        }
    }
    let ret: f64 = match d[..end].parse() {
        Ok(v) => v,
        Err(_) => {
            let _ = io::stderr().lock().write_all(b"timeout: duration is not a number\n");
            process::exit(EXIT_INVALID);
        }
    };
    if suf == '\0' { return ret; }
    if end + 1 != b.len() {
        let _ = io::stderr().lock().write_all(b"timeout: duration unit suffix too long\n");
        process::exit(EXIT_INVALID);
    }
    let ret = match suf {
        's' => ret, 'm' => ret * 60.0, 'h' => ret * 3600.0, 'd' => ret * 86400.0,
        _ => {
            let _ = io::stderr().lock().write_all(b"timeout: duration unit suffix invalid\n");
            process::exit(EXIT_INVALID);
        }
    };
    if ret < 0.0 || ret >= 100000000.0 {
        let _ = io::stderr().lock().write_all(b"timeout: duration out of range\n");
        process::exit(EXIT_INVALID);
    }
    ret
}
