b_n_cut(FILE *fp, const char *fname)
{

			if ((clen = mbrlen(lbuf, lbuflen, &mbs)) < 0) {
				if (!warned) {

					warned = 1;
				}
				memset(&mbs, 0, sizeof(mbs));
				clen = 1;
			}

			if (col < maxval && !positions[1 + col]) {

				    !positions[1 + i])

					canwrite &= positions[1 + i];

			} else {
